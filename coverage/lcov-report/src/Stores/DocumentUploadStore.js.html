import { EventEmitter } from 'events';
import AppDispatcher from '../Dispatchers/AppDispatcher';
import HouseholdMemberActions from '../Actions/HouseholdMemberActions';
 
// Object or array to save data to
let storage = {
	aDocs: [],
	bDocs: [],
	allDocs: [],
	documentType: ""
};
 
const DocumentUploadStore = Object.assign({}, EventEmitter.prototype, {
 
	getDocuments: () => storage.allDocs,
 
	getADocuments: () => storage.aDocs,
 
	getBDocuments: () => storage.bDocs,
 
	getDocumentType: () => storage.documentType,
 
	setDocumentType: (documentType) => {
		storage.documentType = documentType;
	},
 
	getSubsetDocuments: (documentType) => {
		if (documentType === "A"){
			return storage.aDocs;
		}
		else if (documentType === "B"){
			return storage.bDocs;
		}
		return [];
	},
 
	resetUploadedDocs: () => {
		storage.aDocs = [];
		storage.bDocs = [];
		storage.allDocs = [];
	},
 
	deleteAllCurrentUploads: () => {
		storage.allDocs.forEach((d) => {
			HouseholdMemberActions.deleteIdentityDocument(d);
		});
		return [];
	},
 
	dispatcherId: AppDispatcher.register(payload => {
		let action = payload.action;
 
		switch(action.type) {
			case 'IDENTITY_DOCUMENT_UPLOADED':
				_uploadFromServer(action.payload);
				DocumentUploadStore.emit('change');
				break;
			case 'IDENTITY_DOCUMENTS_UPDATED':
				_updateStoreFromServer(action.payload);
				DocumentUploadStore.emit('change');
				break;
			case 'IDENTITY_DOCUMENT_DELETED':
				_deleteFromServer(action.payload);
				DocumentUploadStore.emit('change');
				break;
			default:
				break;
		}
	})
 
});

function _deleteFromServer(fileDeleted){
	storage.aDocs = storage.aDocs.filter(function(f){
		return f.name !== fileDeleted.name;
	});
	storage.bDocs = storage.bDocs.filter(function(f){
		return f.name !== fileDeleted.name;
	});
	storage.allDocs = storage.allDocs.filter(function(f){
		return f.name !== fileDeleted.name;
	});
}
 
function _uploadFromServer(file){

	if (DocumentUploadStore.getDocumentType() === "A"){

}
