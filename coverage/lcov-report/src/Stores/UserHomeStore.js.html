import { EventEmitter } from 'events';
import moment from 'moment';
import AppDispatcher from '../Dispatchers/AppDispatcher';
import ApplicationActions from '../Actions/ApplicationActions.js';
import UserHomeActions from '../Actions/UserHomeAction.js';
import { deepGet, deepSetObj } from '../Components/Forms/FormHOC.jsx';
import {_deleteCookie} from '../Containers/App.jsx';
 
var userHomeDetails = {};
var UserHomeStore = Object.assign({}, EventEmitter.prototype, {
 
	getUserHomeData: () => userHomeDetails,
 
	isIntake: () => userHomeDetails.applicationData && userHomeDetails.applicationData.submissionType === "intake",
 
	isRenewal: () => userHomeDetails.applicationData && userHomeDetails.applicationData.submissionType === "renewal",
 
	isRAC: () => userHomeDetails.applicationData && userHomeDetails.applicationData.submissionType === "rac",
 
	isCoverCalifornia: () => userHomeDetails.applicationData && userHomeDetails.applicationData.renewalType === "ccp",
 
	isMediCalCover: () => userHomeDetails.applicationData && userHomeDetails.applicationData.renewalType === "medical",
 
	isRenewalIsMedicalAndCoveredCA: () => userHomeDetails.applicationData && userHomeDetails.applicationData.renewalType === "ccp_medical",
 
	applicationStartMode: () => userHomeDetails.applicationData && userHomeDetails.applicationData.applicationMode === "start",
 
	isCaseClosed: () => userHomeDetails && userHomeDetails.closedCase === true,
 
	isFormerFosterYouth: () => userHomeDetails && userHomeDetails.formerFosterYouth === true,
 
	applicationContinueMode: () => userHomeDetails.applicationData && userHomeDetails.applicationData.applicationMode === "edit",
 
	applicationCompletedMode: () => userHomeDetails.applicationData && userHomeDetails.applicationData.applicationMode === "view",
 
	isSpecialEnrollmentPeriod: () => userHomeDetails.specialEnrollmentPeriod,
 
	getPendingAppIdList: () => userHomeDetails.pendingAppIdList && userHomeDetails.pendingAppIdList,
 
	emptyPendingAppIdList: () => userHomeDetails.pendingAppIdList && userHomeDetails.pendingAppIdList.filter((application) => application.hasOwnProperty('pendingAppId')).length === 0,
 
	getApplicationDeadlineDate: () => userHomeDetails.impDates && userHomeDetails.impDates.submissionEndDate,
 
	getApplicationEnrollmentStartDate: () => userHomeDetails.impDates && userHomeDetails.impDates.startDate,
 
	getApplicationEnrollmentEndDate: () => userHomeDetails.impDates && userHomeDetails.impDates.endDate,
 
	getCoverageStartDate: () => userHomeDetails.impDates && userHomeDetails.impDates.coverageStartDate,
 
	getCoverageEndDate: () => userHomeDetails.impDates && moment().endOf("year"),
 
	getRenewalEndDate: () => userHomeDetails.impDates && userHomeDetails.impDates.renewalEndDate,
 
	getButtonLabel: () => userHomeDetails.applicationData && userHomeDetails.applicationData.homeButtonLabel,
 
	getApplicationData: () => userHomeDetails.applicationData,
 
	//add date for special enrollment time  logic
 
	getNotCurrentYearAppData: () => {
		let validDate = ((userHomeDetails.impDates && userHomeDetails.impDates.renewalStartDate) && (userHomeDetails.impDates && userHomeDetails.impDates.renewalEndDate)) ? (
			moment().isBetween(userHomeDetails.impDates.renewalStartDate, userHomeDetails.impDates.renewalEndDate, 'days', true)) : false;
 
		return ( userHomeDetails.pendingAppIdList
			&& userHomeDetails.pendingAppIdList.length > 1
			&& validDate) ? userHomeDetails.pendingAppIdList.filter((app) => app.isCurrentYear === false) : false;
	},
 
	getKeepOrSwitchDetails: () => userHomeDetails.manageApplication && userHomeDetails.manageApplication.keepOrSwitchDetls,
 
	hasMultiCoverageYears: () => userHomeDetails.manageApplication && userHomeDetails.manageApplication.keepOrSwitchDetls && userHomeDetails.manageApplication.keepOrSwitchDetls.length > 1,
 
	dispatcherId: AppDispatcher.register(payload => {
		var action = payload.action;
 
		switch(action.type) {
			case 'APPLICATION_WITHDRAWN':
				window.location.href = window.location.href;
			case 'GET_USER_HOME_DETAILS':
				userHomeDetails = _transformUserHomeData(action.payload);
				UserHomeStore.emit('change');
				break;
		}
	})
 
});
 
export default UserHomeStore;
 
function _transformUserHomeData(data) {
	let transforms = {},
		dateFields = [
			'impDates.coverageStartDate',
			'impDates.endDate',
			'impDates.planSelectionDate',
			'impDates.startDate',
			'impDates.renewalEndDate',
			'impDates.submissionEndDate',
			'impDates.formerFosterYouthDOB',
			'impDates.renewalStartDate'
		],
		applicationData = data.pendingAppIdList.filter((application) => {
			return application.isCurrentYear === true;
		}).map((application) => {
			setTimeout(() => {
				UserHomeActions.setEnrollmentYear(application.enrlYear);
				typeof application.pendingAppId !== "undefined" ?
					ApplicationActions.setApplicationId(application.pendingAppId) : (
					_deleteCookie('pendingAppId')
					)
			}, 1);
			return application;
		});
 
	/** assign date fields as moment */
	dateFields.forEach(field => {
		if(typeof deepGet(field, data) === "string") {
			Object.assign(transforms, deepSetObj(field, moment(deepGet(field, data), 'L'), Object.assign({}, data, transforms)));
		}
	});
	return Object.assign({}, data, transforms, {applicationData: applicationData[0]});
}
