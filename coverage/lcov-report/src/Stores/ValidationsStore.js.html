/*
* This file is to be used as a starting point for creating new Actions

// Import dependencies
import { EventEmitter } from 'events';
import AppDispatcher from '../Dispatchers/AppDispatcher';
 
// Object or array to save data to
var usernameRequirements = [
	{
		regex: /^.{8,}$/,
		i18nLabel: 'account-creation-create-username-requirements-minlength'
	},
	{
		regex: /^.{1,50}$/,
		i18nLabel: 'account-creation-create-username-requirements-maxlength'
	},
	{
		regex: /^[a-zA-Z0-9-\.]+$/,
		i18nLabel: 'account-creation-create-username-requirements-content'
	}
];
 
var passwordRequirements = [
	/** match 1 or more characters by default, to highlight dictionary word rule until after server response */
	{
		regex: /.+/,
		i18nLabel: 'account-creation-create-password-requirements-dictionary-word'
	},
	{
		regex: /^.{8,}$/,
		i18nLabel: 'account-creation-create-password-requirements-minlength'
	},
	{
		regex: /^.{1,16}$/,
		i18nLabel: 'account-creation-create-password-requirements-maxlength'
	},
	{
		regex: /[0-9]+/,
		i18nLabel: 'account-creation-create-password-requirements-includes-number'
	},
	{
		regex: /[\!\@\#\$\%\^]+/,
		i18nLabel: 'account-creation-create-password-requirements-includes-special-char'
	},
	{
		regex: /[A-Z]+/,
		i18nLabel: 'account-creation-create-password-requirements-includes-uppercase-char'
	},
	{
		regex: /[a-z]+/,
		i18nLabel: 'account-creation-create-password-requirements-includes-lowercase-char'
	}
];
 
var ValidationsStore = Object.assign({}, EventEmitter.prototype, {
 
	getUsernameRequirements: () => usernameRequirements,
 
	getPasswordRequirements: () => passwordRequirements,
	
	dispatcherId: AppDispatcher.register(payload => {
		var action = payload.action;
		
		switch(action.type) {
			/**
			 *  replaces impossible regex with universally matching regex
			 */
			case 'PASSWORD_VALID':
				passwordRequirements[0].regex = /.+/;
				ValidationsStore.emit('change');
				break;
			/**
			 * sets dictionary match impossible to match regex to cause regex matches to fail
			 */
			case 'INVALID_PASSWORD':
				passwordRequirements[0].regex = /$a/;
				ValidationsStore.emit('change');
				break;
		}
	})
 
});
 
export default ValidationsStore;
