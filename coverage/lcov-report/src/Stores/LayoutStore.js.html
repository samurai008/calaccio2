import { EventEmitter } from 'events';
import AppDispatcher from '../Dispatchers/AppDispatcher';
 
// Object or array to save data to
var data = {
	mobileNavOpen: false,
	fabIsOpen: false,
	localeMenuIsOpen: false,
	progressModalIsOpen: false
};
 
var LayoutStore = Object.assign({}, EventEmitter.prototype, {
 
	getState: () => {
		return data;
	},
 
	dispatcherId: AppDispatcher.register(payload => {
		var action = payload.action;
		
		// Do something on dispatcher events, like update the store.
		switch(action.type) {
			case 'MOBILE_NAV_OPEN':
				data.mobileNavOpen = true;
				LayoutStore.emit('change');
				break;
			case 'MOBILE_NAV_CLOSE':
				data.mobileNavOpen = false;
				LayoutStore.emit('change');
				break;
			case 'MOBILE_NAV_TOGGLE':
				data.mobileNavOpen = ! data.mobileNavOpen;
				LayoutStore.emit('change');
				break;
			case 'TOGGLE_FAB': 
				data.fabIsOpen = ! data.fabIsOpen;
				data.localeMenuIsOpen = false;
				LayoutStore.emit('change');
				break;
			case 'OPEN_LOCALE_MENU': 
				data.localeMenuIsOpen = true;
				data.fabIsOpen = false;
				LayoutStore.emit('change');
				break;
			case 'CLOSE_LOCALE_MENU': 
				data.localeMenuIsOpen = false;
				LayoutStore.emit('change');
				break;
			case 'TOGGLE_LOCALE_MENU': 
				data.localeMenuIsOpen = ! data.localeMenuIsOpen;
				data.fabIsOpen = false;
				LayoutStore.emit('change');
				break;
			case 'OPEN_PROGRESS_MODAL': 
			case 'APP_SIGNATURE_IN_PROGRESS':
			case 'CREATE_ACCOUNT_PENDING':
			case 'IDENTITY_VERIFICATION_QUESTIONS_IN_PROGRESS':
			case 'IDENTITY_VERIFICATION_RESPONSE_IN_PROGRESS':
				data.progressModalIsOpen = true;
				LayoutStore.emit('change');
				break;
			case 'CLOSE_PROGRESS_MODAL': 
			case 'APP_SIGNATURE_SAVED':
			case 'PIN_IS_NOT_VALID':
			case 'CREATE_ACCOUNT_SUCCESS':
			case 'CREATE_ACCOUNT_USERNAME_EXISTS':
			case 'CREATE_ACCOUNT_DUPLICATE_ACCOUNT':
			case 'CREATE_ACCOUNT_ACCESS_CODE_INVALID':
			case 'IDENTITY_VERIFICATION_QUESTIONS':
			case 'IDENTITY_VERIFICATION_RESPONSE':
			case 'IDENTITY_VERIFICATION_ERROR':
			case 'SERVER_ERROR':
			case 'HOUSEHOLD_TAX_DONE':
			case 'HOUSEHOLD_INFO_DONE':
			case 'INCOME_STATUS_UPDATE':
			case 'PERSONS_SERVER_ERROR':
				data.progressModalIsOpen = false;
				LayoutStore.emit('change');
				break;
		}
 
	})
 
});
 
export default LayoutStore;
