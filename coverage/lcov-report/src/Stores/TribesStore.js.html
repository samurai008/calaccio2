import { EventEmitter } from 'events';
import AppDispatcher from '../Dispatchers/AppDispatcher';
 
// Object or array to save data to
// Start with SAMPLE data for local dev.
var storeData = {
	"AK": [{
		"tribeCode": "2364",
		"tribeName": "Alatna Village"
	}, {
		"tribeCode": "2373",
		"tribeName": "Anvik Village"
	}]
};
 
var TribesStore = Object.assign({}, EventEmitter.prototype, {
 
	getAll: () => storeData,
 
	getTribesByState: (state) => {
		return storeData[state];
	},
	getTribeCode: (state, name) => {
		if(typeof storeData[state] === 'undefined' && Array.isArray(storeData[state]) === false) return null;
 
		var foundTribe = storeData[state].filter(tribe => tribe.tribeName === name);
		return foundTribe.length > 0 ? foundTribe[0].tribeCode : null; 
	},
	getTribeName: (state, code) => {
		if(typeof storeData[state] === 'undefined' && Array.isArray(storeData[state]) === false) return null;
 
		var foundTribe = storeData[state].filter(tribe => tribe.tribeCode === code);
		return foundTribe.length > 0 ? foundTribe[0].tribeName : null;
	},
	dispatcherId: AppDispatcher.register(payload => {
		var action = payload.action;
		
		// Do something on dispatcher events, like update the store.
		switch(action.type) {
			case 'TRIBE_NAMES_LIST_UPDATED':
				_updateTribesListFromServer(action.payload);
				// Store only emits from dispatcher callbacks
				TribesStore.emit('change');
				break;
		}
 
	})
});
 
const _updateTribesListFromServer = (data) => {
	if( ! data || ! data.state) return;
	// Index tribes list by state
	storeData[data.state] = data.tribes;
};
 
export default TribesStore;
