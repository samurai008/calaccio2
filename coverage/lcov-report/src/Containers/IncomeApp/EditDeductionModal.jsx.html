// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage, FormattedDate } from 'react-intl';
import Modal, { ModalHeader } from '../../Components/Modal.jsx';
import DeductionEntry from './DeductionEntry.jsx';
 
class EditDeductionModal extends Component {
 
	constructor(props) {
		super(props);
 
		// Set initial state
		this.state = {
			deductionId: this.props.deductionId || null,
			deductionType: this.props.deductionType || null,
			showBackButton: false,
			showCloseButton: false
		};
 
	}
 
	handleCancelEditDeduction() {
		this.props.onCancel();
	}
 
	handleSaveDeduction() {
		this.props.onSave();
	}
 
	render() {
		return (
			<Modal
				className="add-deduction-modal"
				shouldCloseOnOverlayClick={false}
				isOpen={this.props.isOpen}
				onRequestClose={this.props.onRequestClose}
				contentLabel={this.props.person ? this.props.intl.formatMessage({ id: '2.8.6-edit-deduction-modal-header' }, { firstName: this.props.person.firstName }) : ''}
				header={
					<ModalHeader
						closeButton={this.state.showCloseButton}
						backButton={this.state.showBackButton}
					>
						{
							this.props.person
							? (
								<FormattedMessage
									id="2.8.6-edit-deduction-modal-header"
									values={{
										firstName: this.props.person ? this.props.person.firstName : ''
									}}
								/>
							) : ''
						}
					</ModalHeader>
				}
			>
				<DeductionEntry
					person={this.props.person}
					deductionType={this.props.deductionType}
					onRequestConfirmDelete={this.props.onRequestConfirmDelete}
					onCancel={this.handleCancelEditDeduction.bind(this)}
					onSave={this.handleSaveDeduction.bind(this)}
					formColumnClasses="col-xs-12 col-md-8"
					deductionId={this.props.deductionId}
					intl={this.props.intl}
					enrollmentYear={this.props.enrollmentYear}
					isInModal={true}
					{...this.props}
				/>
			</Modal>
		);
	}
}
 
EditDeductionModal.propTypes = {
	isOpen: React.PropTypes.bool,
	deductionType: React.PropTypes.string,
	deductionId: React.PropTypes.oneOfType([
		React.PropTypes.number,
		React.PropTypes.string
	]),
	person: React.PropTypes.object,
	onCancel: React.PropTypes.func,
	onSave: React.PropTypes.func,
	onSaveIncome: React.PropTypes.func,
	onRequestConfirmDelete: React.PropTypes.func,
	enrollmentYear: React.PropTypes.number.isRequired,
	intl: intlShape.isRequired
};
 
EditDeductionModal.defaultProps = {
	onCancel: () => {},
	onSave: () => {},
	onSaveIncome: () => {},
	onRequestConfirmDelete: onConfirm => {}
};
 
EditDeductionModal.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(EditDeductionModal);
