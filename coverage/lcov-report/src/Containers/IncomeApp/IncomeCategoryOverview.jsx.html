// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage, FormattedDate } from 'react-intl';
import IncomeCategoriesStore from '../../Stores/IncomeApp/IncomeCategoriesStore.js';
import IncomeTypesStore from '../../Stores/IncomeApp/IncomeTypesStore.js';
import HouseholdMemberActions from '../../Actions/HouseholdMemberActions.js';
import { ContentCard } from '../../Components/Layout/Layout.jsx';
import IncomeType from '../../Components/IncomeApp/IncomeType.jsx';
import ConditionalField from '../../Components/ConditionalField.jsx';
import EditIncomeModal from './EditIncomeModal.jsx';
import { DeleteModal } from '../../Components/Modal.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
import ContentHeader from '../../Components/ContentHeader.jsx';
import moment from '../../TimeshifterMoment';
 
class IncomeCategoryOverview extends Component {
 
	constructor(props) {
		super(props);
 
		// Set initial state
		this.state = {
			totalCategories: IncomeCategoriesStore.getAllCategories().length,
			editIncomeModalIsOpen: false,
			confirmDeleteIncomeModalIsOpen: false,
			editIncomeId: null,
			editIncomeType: null
		};
 
		this.state = Object.assign(this.state, this.getCategoryState(this.props.params.categoryType));
	}
 
	componentWillReceiveProps(nextProps) {
		this.setState(this.getCategoryState(nextProps.params.categoryType));
	}
 
	getCategoryState(categoryType) {
		var category = categoryType
						? IncomeCategoriesStore.getCategoryByType(categoryType)
						: IncomeCategoriesStore.getFirstCategory();
		return {
			category: category,
			previousCategory: IncomeCategoriesStore.getPreviousCategory(category.categoryType),
			nextCategory: IncomeCategoriesStore.getNextCategory(category.categoryType),
			incomeTypes: IncomeTypesStore.getAllByCategory(category.categoryType),
			currentPage: IncomeCategoriesStore.getCategoryKey(category.categoryType) + 1
		}
	}
 
	onBack() {
		this.context.router.push(this.state.previousCategory ? '/income/person/' + this.props.person.personId + '/income/' + this.state.previousCategory.categoryType : '/income/review');
	}
 
	onForward() {
		this.context.router.push('/income/person/' + this.props.person.personId + (
									this.state.nextCategory
										? '/income/' + this.state.nextCategory.categoryType
										: '/deductions'
							));
	}
 
	handleAddIncome(typeCode) {
		if(this.props.viewOnlyMode === true) return;
		this.context.router.push('/income/person/' + this.props.params.personId + '/income/add-income/' + typeCode);
	}
 
	handleEditIncome(incomeType, incomeId) {
		this.setState({ editIncomeId: incomeId, editIncomeType: incomeType, editIncomeModalIsOpen: true});
	}
 
	handleCloseEditIncomeModal() {
		this.setState({ editIncomeId: null, editIncomeType: null, editIncomeModalIsOpen: false});
	}
 
	requestConfirmDeleteIncome(onConfirm) {
		this.onConirmDeleteIncome = onConfirm;
		this.setState({ editIncomeModalIsOpen: false, confirmDeleteIncomeModalIsOpen: true });
	}
 
	cancelDeleteIncome() {
		this.setState({ editIncomeModalIsOpen: true, confirmDeleteIncomeModalIsOpen: false });
	}
 
	confirmDeleteIncome() {
		this.onConirmDeleteIncome();
		this.setState({ confirmDeleteIncomeModalIsOpen: false });
	}
 
	render() {
		const classNames = ['income-overview'],
			incomeTypes = this.state.incomeTypes.map((cat, key) => {
				/* don't display special types unless the user is a member of the group, or if the member is, and the type is PC which doesn't get displayed */
				if((this.props.person && (!this.props.person.memberOfIndianTribeIndicator || this.props.person.memberOfIndianTribeIndicator === 'N') &&	cat.incomeType == 'AA')
					|| cat.incomeType == 'PC') {
					return;
				}
 
				var incomes = this.props.person && Array.isArray(this.props.person.incomes) ?
								this.props.person.incomes.filter((income) => {
									return income.incomeType === cat.incomeType
										/** show PC income entries with AA */
										|| (this.props.person && this.props.person.memberOfIndianTribeIndicator === 'Y' && cat.incomeType == 'AA' && income.incomeType == 'PC')
								})
								: [];
				return (
					<li key={key}>
						<IncomeType
							id={'income-overview-type-' + cat.incomeType}
							item={cat}
							incomes={incomes}
							onAdd={this.handleAddIncome.bind(this)}
							onEdit={this.handleEditIncome.bind(this)}
							showSummaryRow={true}
							person={this.props.person}
							viewOnlyMode={this.props.viewOnlyMode}
						/>
					</li>
				)
			});
 
		// Add category specific classes
		classNames.push(this.state.category.categoryType);
 
		return (
			<div className={classNames.join(' ')}>
				<ContentCard>
					<ContentHeader
						text={
							this.props.person
							? <FormattedMessage
								id="2.1-employment-&-self-employment-income-header-dynamic"
								values={{
									firstName: this.props.person.firstName
								}}
								/>
							: ''
						}
					/>
 
					<div className="content">
						<div className="intro">
							<h5 className="category-count">
								<FormattedMessage
									id="category-page-of-page-count"
									values={{
										currentPage: this.state.currentPage,
										totalPages: this.state.totalCategories
									}}
									/>
							</h5>
							<h2>
								<FormattedHTMLMessage
									id={this.state.category.categoryType + '-category-title'}
									/>
							</h2>
							<p>
								<FormattedMessage
									id={this.state.category.categoryType + '-category-instructions'}
									values={{
										firstName: this.props.person ? this.props.person.firstName : '',
										currentYear: <FormattedDate value={moment().year(this.props.enrollmentYear)} year="numeric" />
									}}
								/>
							</p>
						</div>
 
						<div className="income-categories">
							<ul>{incomeTypes}</ul>
						</div>
 
						<ConditionalField condition={
							this.props.intl.doesMessageExist(this.state.category.categoryType + '-category-disclaimer')
						}>
							<div className="disclaimer">
								<FormattedHTMLMessage
									tagName="div"
									id={this.state.category.categoryType + '-category-disclaimer'}
									/>
							</div>
						</ConditionalField>
					</div>
 
					<ForwardBack
						id="income-category-forwardback"
						forwardText={<FormattedMessage id="2.1-employment-&-self-employment-income-button7" />}
						backText={<FormattedMessage id="2.1-employment-&-self-employment-income-button8" />}
						onForward={this.onForward.bind(this)}
						onBack={this.onBack.bind(this)}
					/>
 
				</ContentCard>
 
				<EditIncomeModal
					isOpen={this.state.editIncomeModalIsOpen}
					person={this.props.person}
					onSave={this.handleCloseEditIncomeModal.bind(this)}
					onCancel={this.handleCloseEditIncomeModal.bind(this)}
					onRequestConfirmDelete={this.requestConfirmDeleteIncome.bind(this)}
					incomeId={this.state.editIncomeId}
					incomeType={this.state.editIncomeType}
					intl={this.props.intl}
					enrollmentYear={this.props.enrollmentYear}
					viewOnlyMode={this.props.viewOnlyMode}
				/>
 
				<DeleteModal
					isOpen={this.state.confirmDeleteIncomeModalIsOpen}
					onRequestClose={this.cancelDeleteIncome.bind(this)}
					onRequestConfirm={this.confirmDeleteIncome.bind(this)}
					contentLabel={this.props.intl.messageIfExists('modals.delete.header')}
					title={this.props.intl.messageIfExists('modals.delete.header')}
					confirmText={this.props.intl.messageIfExists('modals.delete.confirmButtonText')}
					cancelText={this.props.intl.messageIfExists('modals.delete.cancelButtonText')}
				>
					<FormattedHTMLMessage id="modals.delete.text" />
				</DeleteModal>
 
			</div>
		);
	}
}
 
IncomeCategoryOverview.propTypes = {
	intl: intlShape.isRequired,
	person: React.PropTypes.object,
	params: React.PropTypes.object,
	enrollmentYear: React.PropTypes.number.isRequired,
	viewOnlyMode: React.PropTypes.bool
};
 
IncomeCategoryOverview.defaultProps = {
	viewOnlyMode: false
};
 
IncomeCategoryOverview.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(IncomeCategoryOverview);
