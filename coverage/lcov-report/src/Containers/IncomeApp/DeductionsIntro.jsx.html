// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage, FormattedDate } from 'react-intl';
import { ContentIntro, ContentCard } from '../../Components/Layout/Layout.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
 
class DeductionsIntro extends Component {
 
	constructor(props) {
		super(props);
 
		// Set initial state
		this.state = {};
	}
 
	onBack() {
		this.context.router.push('/income/person/' + this.props.person.personId + '/income/ATI');
	}
 
	onForward() {
		this.context.router.push('/income/person/' + this.props.person.personId + '/deductions/overview');
	}
 
	render() {
		// const classNames = ['tax-deductions-intro'];
		return (
			<div className="application-start image-header deductions">
				<ContentIntro className="deductions_intro-content">
					<div className="deductions_intro-content_img hidden-xs" />
					<div className="deductions_intro-content_img--mobile hidden-sm hidden-md hidden-lg" />
					<h1 className="deductions_intro-conten_header deductions_intro-conten_title">
								{
									this.props.person
									? <FormattedHTMLMessage
											id="tax-deductions-intro-subheadline"
											values={{
												firstName: this.props.person.firstName
											}}
										/>
									: ''
								}
							</h1>
				</ContentIntro>
				<ContentCard>
					<div className="deductions_content-card">
						<p className="">
							{
								this.props.person
								? <FormattedHTMLMessage
									id="tax-deductions-intro-body"
									values={{
										firstName: this.props.person.firstName
									}}
									/>
								: ''
							}
						</p>
					</div>
 
					<ForwardBack
						id="deducations-intro-forwardback"
						forwardText={<FormattedMessage id="tax-deductions-intro-back-continue" />}
						backText={<FormattedMessage id="tax-deductions-intro-back-button" />}
						onForward={this.onForward.bind(this)}
						onBack={this.onBack.bind(this)}
					/>
				</ContentCard>
 
		</div>
 
		);
	}
}
 
DeductionsIntro.propTypes = {
	person: React.PropTypes.object,
	intl: intlShape.isRequired
};
 
DeductionsIntro.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(DeductionsIntro);
