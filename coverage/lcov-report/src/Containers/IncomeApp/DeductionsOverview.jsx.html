// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage, FormattedDate } from 'react-intl';
import DeductionCategoriesStore from '../../Stores/IncomeApp/DeductionCategoriesStore.js';
import PeopleStore from '../../Stores/PeopleStore.js';
import { ContentCard } from '../../Components/Layout/Layout.jsx';
import DeductionCategory from '../../Components/IncomeApp/DeductionCategory.jsx';
import { DeleteModal } from '../../Components/Modal.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
import EditDeductionModal from './EditDeductionModal.jsx';
 
 
class DeductionsOverview extends Component {
 
	constructor(props) {
		super(props);
 
		// Set initial state
		this.state = {
			categories: DeductionCategoriesStore.getAllCategories(),
			editDeductionModalIsOpen: false,
			confirmDeleteDeductionModalIsOpen: false,
			editDeductionId: null,
			editDeductionType:  null
		};
 
	}
 
 
	onBack() {
		this.context.router.push('/income/person/' + this.props.person.personId + '/deductions');
	}
 
	onSave() {
		this.context.router.push('/income/person/' + this.props.person.personId + '/review');
	}
 
	handleAddDeduction(typeCode) {
		if(this.props.viewOnlyMode === true) return;
		this.context.router.push('/income/person/' + this.props.person.personId + '/deductions/add-deduction/' + typeCode);
	}
 
	handleEditDeduction(typeCode, deductionId) {
		this.setState({ editDeductionId: deductionId, editDeductionType: typeCode, editDeductionModalIsOpen: true});
	}
 
	handleCloseEditDeductionModal() {
		this.setState({ editDeductionId: null, editDeductionType: null, editDeductionModalIsOpen: false});
	}
 
	requestConfirmDeleteDeduction(onConfirm) {
		this.onConfirmDeleteDeduction = onConfirm;
		this.setState({ editDeductionModalIsOpen: false, confirmDeleteDeductionModalIsOpen: true });
	}
 
	cancelDeleteDeduction() {
		this.setState({ editDeductionModalIsOpen: true, confirmDeleteDeductionModalIsOpen: false });
	}
 
	confirmDeleteDeduction() {
		this.onConfirmDeleteDeduction();
		this.setState({confirmDeleteDeductionModalIsOpen: false});
	}
	render() {
		const classNames = ['deductions-overview'],
			categories = this.state.categories.map((cat, key) => {
				var deductions = this.props.person && Array.isArray(this.props.person.deductions) ?
											this.props.person.deductions.filter(deduction => deduction.incomeType === cat.categoryType)
											: [];
				return <DeductionCategory id={'deductions-overview-category-' + cat.categoryType} key={key} item={cat} deductions={deductions} onAdd={this.handleAddDeduction.bind(this)} onEdit={this.handleEditDeduction.bind(this)} viewOnlyMode={this.props.viewOnlyMode} />
			}),
			hasSEIncome = this.props.person ? PeopleStore.personHasIncomeType(this.props.person.personId, 'SE') : false;
		return (
			<div className={classNames.join(' ')}>
				<div className="title-border-line"></div>
				<ContentCard>
					<div className="header">
						<h1>
							{
								this.props.person
								? <FormattedMessage
									id="tax-deductions-overview-header"
									values={{
										firstName: this.props.person.firstName
									}}
									/>
								: ''
							}
						</h1>
					</div>
					<div className="content">
						<div className="intro">
							<h2>
								<FormattedHTMLMessage
									id="tax-deductions-overview-title"
									/>
							</h2>
							<p>
								<FormattedMessage
									id="tax-deductions-overview-subtitle"
									values={{
										firstName: this.props.person ? this.props.person.firstName : ''
									}}
								/>
							</p>
						</div>
 
						<div className="disclaimer">
							{
								hasSEIncome ? <FormattedHTMLMessage
										tagName="div"
										id="tax-deductions-overview-disclaimer"
										values={{
											firstName: this.props.person ? this.props.person.firstName : ''
										}}
									/>
									: ''
							}
 
						</div>
 
						<div className="deduction-categories-list">
							{categories}
						</div>
					</div>
 
					<ForwardBack
						id="deductions-overview-forwardback"
						forwardText={<FormattedMessage id="tax-deductions-overview-save-button" />}
						backText={<FormattedMessage id="tax-deductions-overview-back-button" />}
						onForward={this.onSave.bind(this)}
						onBack={this.onBack.bind(this)}
					/>
 
				</ContentCard>
 
				<EditDeductionModal
					isOpen={this.state.editDeductionModalIsOpen}
					person={this.props.person}
					onSave={this.handleCloseEditDeductionModal.bind(this)}
					onCancel={this.handleCloseEditDeductionModal.bind(this)}
					onRequestConfirmDelete={this.requestConfirmDeleteDeduction.bind(this)}
					deductionId={this.state.editDeductionId}
					deductionType={this.state.editDeductionType}
					intl={this.props.intl}
					enrollmentYear={this.props.enrollmentYear}
					viewOnlyMode={this.props.viewOnlyMode}
				/>
 
				<DeleteModal
					isOpen={this.state.confirmDeleteDeductionModalIsOpen}
					onRequestClose={this.cancelDeleteDeduction.bind(this)}
					onRequestConfirm={this.confirmDeleteDeduction.bind(this)}
					contentLabel={this.props.intl.messageIfExists('modals.delete.header')}
					title={this.props.intl.messageIfExists('modals.delete.header')}
					confirmText={this.props.intl.messageIfExists('modals.delete.confirmButtonText')}
					cancelText={this.props.intl.messageIfExists('modals.delete.cancelButtonText')}
				>
					<FormattedHTMLMessage id="modals.delete.text" />
				</DeleteModal>
			</div>
		);
	}
}
 
DeductionsOverview.propTypes = {
	person: React.PropTypes.object,
	intl: intlShape.isRequired,
	viewOnlyMode: React.PropTypes.bool
};
DeductionsOverview.defaultProps = {
	viewOnlyMode: false
};
 
DeductionsOverview.contextTypes = {
    router: React.PropTypes.object.isRequired
};
export default injectIntl(DeductionsOverview);
