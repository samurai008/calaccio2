// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage, FormattedDate } from 'react-intl';
import { browserHistory } from 'react-router';
import Modal, { ModalHeader } from '../../Components/Modal.jsx';
import IncomeEntry from './IncomeEntry.jsx';
 
class EditIncomeModal extends Component {
 
	constructor(props) {
		super(props);
 
		// Set initial state
		this.state = {
			incomeId: this.props.incomeId || null,
			incomeType: this.props.incomeType || null,
			showBackButton: false,
			showCloseButton: false
		};
 
	}
 
	componentDidUpdate() {
		if(this.refs.modalContainer && this.refs.modalContainer.refs.modal && this.refs.modalContainer.refs.modal.portal.refs.content) {
			setTimeout(() => {
				this.refs.modalContainer.refs.modal.portal.refs.content.focus();
			}, 0);
		}
	}
 
 
	handleCancelEditIncome() {
		this.props.onCancel();
	}
 
	handleSaveIncome() {
		this.props.onSave();
	}
 
	render() {
		var classNames = ['add-income-modal'],
			content = null,
			incomeTypes = null,
			typeTitleString = (this.props.incomeType) ? this.props.intl.formatMessage({ id: this.props.incomeType + '-income-type-title' }) : null,
			typeTitle = (typeTitleString !== null && this.props.person) ? <FormattedMessage id="2.8.3-edit-income-modal-header-dynamic" values={{ typeTitle: typeTitleString, firstName: this.props.person.firstName }} /> : '';
 
		return (
			<Modal
				className={classNames.join(' ')}
				shouldCloseOnOverlayClick={false}
				isOpen={this.props.isOpen}
				onRequestClose={this.props.onRequestClose}
				contentLabel={this.props.person ? this.props.intl.formatMessage({ id: '2.8.3-edit-income-modal-header-dynamic' }, { firstName: this.props.person.firstName }) : ''}
				header={
					<ModalHeader
						closeButton={this.state.showCloseButton}
						backButton={this.state.showBackButton}
					>
						{typeTitle}
					</ModalHeader>
				}
				ref="modalContainer"
			>
				<IncomeEntry
					person={this.props.person}
					incomeType={this.props.incomeType}
					onRequestConfirmDelete={this.props.onRequestConfirmDelete}
					onCancel={this.handleCancelEditIncome.bind(this)}
					onSave={this.handleSaveIncome.bind(this)}
					formColumnClasses="col-xs-12 col-md-8"
					incomeId={this.props.incomeId}
					intl={this.props.intl}
					enrollmentYear={this.props.enrollmentYear}
					isInModal={true}
					{...this.props}
				/>
			</Modal>
		);
	}
}
 
EditIncomeModal.propTypes = {
	person: React.PropTypes.object,
	incomeType: React.PropTypes.string,
	incomeId: React.PropTypes.oneOfType([
		React.PropTypes.number,
		React.PropTypes.string
	]),
	onCancel: React.PropTypes.func,
	onSave: React.PropTypes.func,
	onSaveIncome: React.PropTypes.func,
	onRequestConfirmDelete: React.PropTypes.func,
	enrollmentYear: React.PropTypes.number.isRequired,
	intl: intlShape.isRequired
};
 
EditIncomeModal.defaultProps = {
	onSaveIncome: () => {},
	onRequestConfirmDelete: onConfirm => {}
};
 
EditIncomeModal.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(EditIncomeModal);
