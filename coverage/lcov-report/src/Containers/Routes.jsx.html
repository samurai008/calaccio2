/**
 * onChange handler for the top-level Route. When called when a Route changes, generates a new UUID
 * value for use as the Correlation Id for all REST requests from Actions from the current page.
 * 
 * Since the UUID value is not changed again until the next Route change, all REST requests
 * originating from Action calls from the same page will be stamped with the same UUID 
 * Correlation Id value.
 * 
 * This Correlation Id value is used by Operations to correlate related REST requests coming from a
 * specific page for a user.
 * 
 * @param previousRoute
 * @param nextRoute
 */
const routeChangeHandler = (previousRoute, nextRoute) => {
	ConfigStore.createNewUUIDForRouteChange();
	//console.log("Router change for:[" + nextRoute.location.pathname + "], new CorrelationId: "
	//	+ ConfigStore.getRouteChangeBasedUUIDCorrelationId());
	ConfigStore.setPageIdAndNameForRouteChange(nextRoute.location.pathname);
 
}
