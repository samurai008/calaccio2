import React from 'react';
import moment from '../../TimeshifterMoment';

import ConditionalField from '../../Components/ConditionalField.jsx';
import ApplicationStore from '../../Stores/ApplicationStore.js';
import ConfigStore from '../../Stores/ConfigStore.js';
 
const generatePortalUrl = key => {
	var portalURL = ConfigStore.getURL('portalURL'),
		pageURL = ConfigStore.get(key);
 
		return (portalURL && pageURL !== null) ? pageURL
						.replace('{portalURL}', portalURL)
						.replace('{encryptedId}', ApplicationStore.getPendingAppId()) : null;
};

const Announcements = props => {
	return (
		<div className="announcements">

const FooterLinks = (props, context) => {
	var i18nValues = {
		currentYear: moment().year()
	},
	showManageApplication = Object.keys(props.manageApplication).filter((key) => props.manageApplication[key]).length !== 0,
 
	showMoreActions = Object.keys( props.moreActions).filter((key) =>  props.moreActions[key]).length !== 0,
 
	showAnnouncements = typeof props.announcements !== 'undefined' && Object.keys( props.announcements).filter((key) =>  props.announcements[key]).length !== 0,
 
	announcements = props.announcements && props.announcements.viewAnnouncement === true ? (
		<li key="viewAnnouncement">
			<a id="account-home-footer-viewAnnouncement-btn" href={generatePortalUrl(urls.viewAnnouncement)}>
				<FormattedMessage
					id={`account-home-footer-button-viewAnnouncement`}
				/>
			</a>
		</li>) : null,
 
	announcementMessage = (props.announcements && props.announcements.announcementText) ? props.announcements.announcementText : <FormattedMessage id={`account-home-footer-button-noAnnouncements`}/>,
 
	adminUser = props.userType('consumer') === false,
 
	applyForNotCurrentAppBtn = () => {
		if (props.notCurrentYearApp && adminUser) {
			let data = props.notCurrentYearApp[0],
			handlerClick = data.homeButtonLabel === 'begin_application' ? props.notCurrentYearAppStart(data.enrlYear) : props.notCurrentYearAppContinue(data.enrlYear, data.pendingAppId);
			return <li key='applyFor'>
				<button id={`account-home-footer-button-manageApplication-applyFor-btn-${data.homeButtonLabel}`} onClick={handlerClick} type="button" className="btn btn-link">
					<FormattedMessage
						id={`account-home-footer-button-manageApplication-applyFor-btn-${data.homeButtonLabel}`}
						values={{enrlYear: data.enrlYear}}
					/>
				</button>
			</li>;
 
		}
		return null;
	},
 
	manageApplicationKeys = [
		'notCurrentYear',
		"eligResult",
		"enrlSummary",
		"changePlan",
		"selectHealthPlan",
		"keepOrSwitch",
		"keepOrSwitchDetls",
		"changePremiumAssit",
		"rac",
		"reportAChange",
		"continueRAC",
		"continueRenewal",
		"cancelRAC",
		"provideVerif",
		"discontCCHIP",
		"withdrawApplication",
		"viewCaseHistory",
	].filter((key) => props.manageApplication[key] === true)
	.map((key) => {
		if (urls[key] === key) {
			let handler = (['continueRenewal', 'continueRAC'].indexOf(key) > -1) ? (
				() => context.router.push((generatePortalUrl(urls[key])))) : (
					(typeof props[key + 'Modal'] !== "undefined") ? props[key + 'Modal'] : props['handle' + key]
				);
			return (
				<li key={key}>
					<button
						id={`account-home-footer-manageApp-btn-${key}`}
						onClick={handler}
						type="button"
						className="btn btn-link"
					>
						<FormattedMessage id={`account-home-footer-button-manageApplication-${key}`}/>
					</button>
				</li>
			);
		}
		else {
			return (
				<li key={key}>
					<a id={`account-home-footer-manageApp-btn-${key}`} href={generatePortalUrl(urls[key])}>
						<FormattedMessage id={`account-home-footer-button-manageApplication-${key}`}/>
					</a>
				</li>
			);
		}
	}),
 
	manageApplication = [applyForNotCurrentAppBtn(), ...manageApplicationKeys],
 
	moreActions = [
		"myProfile",
		"secureMailBox",
		"authRep",
		"manageDelegate",
		"shopAndCompare",
		"cceHome",
		"consentVerif",
		"emplrContactInfo",
		"cancelCoverage",
		"downloadPDF",
		"adobePDFReader",
	].filter((key) => props.moreActions[key] === true)
	.map((key) => {
		if(key === 'secureMailBox' && props.userType('consumer') === true) {
			return(
				<li key={key}>
					<a id={`account-home-footer-moreActions-btn-${key}`} href={generatePortalUrl(urls[key])}>
						<FormattedMessage
							id={`account-home-footer-button-moreActions-${key}`}
							values={{secureMailCount: props.moreActions.secureMailCount ? props.moreActions.secureMailCount : ''}}
						/>
					</a>
				</li>
			);
		}
		return(
			<li key={key}>
				<a
					id={`account-home-footer-moreActions-btn-${key}`}
					href={generatePortalUrl(urls[key])}
					target={(['downloadPDF', 'adobePDFReader', 'cceHome'].indexOf(key) > -1) ? "_blank" : ''}
				>
					<FormattedMessage
						id={`account-home-footer-button-moreActions-${key}`}
					/>
				</a>
			</li>
		);
	});
 
	return (
		<div className="footer-links">
			<div className="row">
				<ConditionalField condition={showAnnouncements}>
					<div className="col-xs-12 col-sm-4">
						<FormattedMessage id="account-home-announcements-header" tagName="h3" />
						<ul>
							<li>
								<Announcements date={props.announcements && props.announcements.announcementDate} content={announcementMessage}/>
							</li>
							{announcements}
						</ul>
					</div>
				</ConditionalField>
				<ConditionalField condition={showManageApplication}>
					<div className="col-xs-12 col-sm-4">
						<FormattedMessage id="account-home-manage-header" tagName="h3" />
						<ul>
							{manageApplication}
						</ul>
					</div>
				</ConditionalField>
				<ConditionalField condition={showMoreActions}>
					<div className="col-xs-12 col-sm-4">
						<FormattedMessage id="account-home-more-options-header" tagName="h3" />
						<ul>
							{moreActions}
						</ul>
					</div>
				</ConditionalField>
			</div>
		</div>
	)
};
 
 
FooterLinks.propTypes = {
	announcements: React.PropTypes.object,
	manageApplication: React.PropTypes.object,
	moreActions: React.PropTypes.object,
	userType: React.PropTypes.func,
	notCurrentYearAppStart: React.PropTypes.func,
	notCurrentYearAppContinue: React.PropTypes.func
};
 
FooterLinks.contextTypes = {
	router: React.PropTypes.object.isRequired
};
 
export default FooterLinks;
