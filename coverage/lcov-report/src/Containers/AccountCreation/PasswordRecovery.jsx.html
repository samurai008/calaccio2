import { Form } from 'formsy-react';
import ConditionalField from '../../Components/ConditionalField.jsx';
import TextInput from '../../Components/Inputs/TextInput.jsx';
import PhoneInput from '../../Components/Inputs/PhoneInput.jsx';
import SegmentedControl from '../../Components/Inputs/SegmentedControl.jsx';
import CheckBox from '../../Components/Inputs/CheckBox.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
 
class PasswordRecovery extends Component {
 
	constructor(props) {
		super(props);
		// Set initial state
		this.state = {
			formIsValid: false,
			formData: {}
		};
	}
 
	onFormValid() {
		this.setState({ formIsValid: true });
		this.props.onFormValid();
	}
 
	onFormInValid() {
		this.setState({ formIsValid: false });
		this.props.onFormInValid();
	}
 
	handleFieldChange(name, value) {
		var delta = {};
		delta[name] = value;
		this.setState({ formData: Object.assign(this.state.formData, delta) });
	}
 
	getFormValue(name) {
		return typeof this.state.formData[name] !== 'undefined' ? this.state.formData[name] : ''
	}
 
	handleSubmitForm(data) {

	}
 
	onForward() {
 
	}
 
	onBack() {
 
	}
 
	render() {
		var className = ['password-recovery'].concat(
				this.getFormValue('contactMethod').indexOf('email') >= 0 ? 'has-email' : [],
				this.getFormValue('contactMethod').indexOf('phone') >= 0 ? 'has-phone' : []
			);
 
		return (
			<div className={className.join(' ')}>
				<span className="step-progress">
					<FormattedMessage id="account-creation-create-step-progress" values={{ page: 4, total: 4 }} />
				</span>
				<h1><FormattedMessage id="password-recovery-header" /></h1>
				<p><FormattedMessage id="password-recovery-header-text" /></p>
				<Form onValidSubmit={this.handleSubmitForm.bind(this)} onValid={this.onFormValid.bind(this)} onInvalid={this.onFormInValid.bind(this)}>
					<FormattedMessage id="password-recovery-method-label" tagName="label" />
					<SegmentedControl
						className="preferred-contact-control"
						name="contactMethod"
						id="contactMethod"
						onChange={this.handleFieldChange.bind(this)}
						options={{
							email: this.props.intl.messageIfExists('password-recovery-method-email-label'),
							phone: this.props.intl.messageIfExists('password-recovery-method-phone-label')
						}}
						multiSelect={true}
						required
					/>
					<div className="row">
						<div className="col-xs-12 col-sm-6">
							<ConditionalField condition={this.getFormValue('contactMethod').indexOf('email') >= 0}>
								<TextInput 
									className="email-input"
									label={<FormattedMessage id="password-recovery-email-label" />}
									aria-label={this.props.intl.messageIfExists('password-recovery-email-label')}
									name="email"
									id="email"
									htmlFor="email"
									value=""
									onChange={this.handleFieldChange.bind(this)}
									validations="isEmail"
									validationError={this.props.intl.messageIfExists('password-recovery-email-error')}
									required
								/>
							</ConditionalField>
							<ConditionalField condition={this.getFormValue('contactMethod').indexOf('phone') >= 0}>
								<PhoneInput 
									label={<FormattedMessage id="password-recovery-phone-label" />}
									aria-label={this.props.intl.messageIfExists('password-recovery-phone-label')}
									name="phone"
									id="phone"
									htmlFor="phone"
									value=""
									onChange={this.handleFieldChange.bind(this)}
									validations={{
										matchRegexp: /^\(\d{3}\)\s\d{3}-\d{4}$/
									}}
									validationError={this.props.intl.messageIfExists('password-recovery-phone-error')}
									required
								/>
								<CheckBox
									label={<FormattedMessage id="password-recovery-allow-text-label" />}
									name="allowTextMessages"
									id="allowTextMessages"
									htmlFor="allowTextMessages"
									onChange={this.handleFieldChange.bind(this)}
								/>
							</ConditionalField>
							<ConditionalField condition={this.state.formIsValid}>
								<FormattedMessage id="password-recovery-contact-message" tagName="p" />
							</ConditionalField>
						</div>
					</div>
				</Form>
				<ForwardBack
					id="password-recovery-forwardback"
					forwardText={this.getFormValue('contactMethod') ? <FormattedMessage id="password-recovery-next" /> : <FormattedMessage id="password-recovery-skip" />}
					backText={<FormattedMessage id="password-recovery-back" />}
					onForward={this.onForward.bind(this)}
					onBack={this.onBack.bind(this)}
				/>
			</div>
		);
	}
}
