import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage } from 'react-intl';
import FormHOC from '../../Components/Forms/FormHOC.jsx';
import { Modal, ConfirmModal } from '../../Components/Modal.jsx';
import RadioInput from '../../Components/Inputs/RadioInput.jsx';
import moment from '../../TimeshifterMoment';
import UserHomeActions from '../../Actions/UserHomeAction.js';
 
const BeginApplicationModal = injectIntl(FormHOC((props) => {
	let radioBtns = props.pendingAppIdList.reduce((obj, currentElement, i) => {
			obj[currentElement.enrlYear] = props.intl.messageIfExists('account-home-modal-Begin-application-radio-1', {currentYear: currentElement.enrlYear});
			return obj;
	}, {});
    return (
        <div>
            <ConfirmModal
					isOpen={props.openBeginAppModal}
					onRequestClose={props.closeBeginAppModal}
					onRequestConfirm={() => {
						UserHomeActions.setEnrollmentYear(props.getFormValue('enrollmentYear'));
						props.confirmBeginAppModal();
					}}
					fullScreenMobile={true}
					className={props.className}
					confirmDisabled={props.confirmDisabled}
					title={<FormattedMessage id="account-home-modal-Begin-application-header" />}
					contentLabel={props.intl.formatMessage({ id: '1.0.1-no-income-confirmation-modal-header' })}
					confirmText={<FormattedMessage id="account-home-modal-Begin-application-button-confirm" />}
					cancelText={<FormattedMessage id="account-home-modal-Begin-application-button-cancel" />}
					>
					<RadioInput
						name="enrollmentYear"
						id="enrollmentYear"
						className="opt-in-radio text-left"
						label=""
						inline={false}
						onChange={props.handleFieldChange}
						value={props.getFormValue('enrollmentYear')}
						options={radioBtns}
						required
					/>
            </ConfirmModal>
        </div>
    )
}));
 
BeginApplicationModal.propTypes = {
	// intl: intlShape.isRequired
	openBeginAppModal: React.PropTypes.bool,
	closeBeginAppModal: React.PropTypes.func,
	getFormValue: React.PropTypes.func,
	pendingAppIdList: React.PropTypes.array
};
 
export default BeginApplicationModal;
