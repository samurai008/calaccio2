const generatePortalUrl = (key) => {
	var portalURL = ConfigStore.getURL('portalURL'),
		pageURL = ConfigStore.get(key);
 
	return (portalURL && pageURL !== null) ? pageURL

					.replace('{encryptedId}', ApplicationStore.getPendingAppId()) : null;
 
};

	handleIntake() {

		if(UserHomeStore.applicationStartMode() === true) {
			if(UserHomeStore.getPendingAppIdList().length > 1 && AuthStore.userHasRole('consumer') === false) {

				return;

		if(UserHomeStore.applicationContinueMode() === true) {
			this.viewConfirmationFilter("/application-menu");

		}

	shouldRacModalShow() {
		let individualAgentCec = ['consumer', 'agent', 'cec'].filter(role => AuthStore.userHasRole(role)).length > 0,
			// when next coverage year begins
			cutOff = UserHomeStore.getCoverageStartDate(),
			isBeforeCoveragePeriod = moment().isBefore(cutOff),
			appIdList = UserHomeStore.getPendingAppIdList(),
			hasMultiCoverageYears = (Array.isArray(appIdList)
				&& appIdList.length === 2
				&& appIdList.filter((app) => app.isCurrentYear === false)
					.every((app) => {
						return (app.homeButtonLabel === 'begin_rac');
					}
				)
			);
		// to display rac modal, for individual, agent or cec, it must be 12/31/<previous coverage year> or before, with multiple coverage years
		// for admin users, mulitple coverage years is the only criteria
		return (individualAgentCec && isBeforeCoveragePeriod && hasMultiCoverageYears) || (hasMultiCoverageYears && AuthStore.userHasRole('admin'));
	}
