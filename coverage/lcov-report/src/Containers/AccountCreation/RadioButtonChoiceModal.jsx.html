import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage } from 'react-intl';
import FormHOC from '../../Components/Forms/FormHOC.jsx';
import Modal, { ConfirmModal } from '../../Components/Modal.jsx';
import RadioInput from '../../Components/Inputs/RadioInput.jsx';
import moment from '../../TimeshifterMoment';
import ConfigStore from '../../Stores/ConfigStore';
import Header from '../../Components/Header.jsx';
import UserHomeActions from '../../Actions/UserHomeAction.js';
 
const RadioButtonChoiceModal = injectIntl(FormHOC((props) => {
	let radioBtns = [];
	if(Array.isArray(props.pendingAppIdList)) {
		radioBtns = props.pendingAppIdList.reduce((obj, currentElement, i) => {
			obj[currentElement.enrlYear] = props.intl.messageIfExists(props.radioLabel, {year: currentElement.enrlYear});
			return obj;
		}, {});
	}
	return (
		<div>
			<ConfirmModal
				isOpen={props.isOpen}
				onRequestClose={props.cancelModal}
				onRequestConfirm={() => {
					props.onConfirm(props.getFormValue('enrollmentYear'));
				}}
				fullScreenMobile={true}
				contentLabel={props.intl.messageIfExists(props.headerTextId)}
				confirmText={<FormattedMessage id="account-home-modal-Begin-application-button-confirm" />}
				cancelText={<FormattedMessage id="account-home-modal-Begin-application-button-cancel" />}
				className={ConfigStore.get('isMobile') ? "fullscreen-confirm-modal" : "squish"}
				confirmDisabled={props.getFormValue('enrollmentYear') === null}
				title={props.intl.messageIfExists(props.headerTextId)}
				id='keep-or-switch-modal'
				>
				{
					ConfigStore.get('isMobile') ? <h3 className="text-center"><FormattedMessage id={props.headerTextId}/></h3> : null
				}
				<p className="text-left">
					{props.intl.messageIfExists(props.paragraphTextId, {year: Object.keys(radioBtns)[1]})}
				</p>
				<RadioInput
					name="enrollmentYear"
					id="enrollmentYear"
					className="text-left opt-in-radio"
					label=""
					inline={false}
					onChange={props.handleFieldChange}
					value={props.getFormValue('enrollmentYear')}
					options={radioBtns}
					required
				/>
			</ConfirmModal>
		</div>
	)
}));
 
RadioButtonChoiceModal.propTypes = {
	// intl: intlShape.isRequired
	isOpen: React.PropTypes.bool,
	cancelModal: React.PropTypes.func,
	pendingAppIdList: React.PropTypes.array,
	handleFieldChange: React.PropTypes.func,
	paragraphTextId: React.PropTypes.string,
	headerTextId: React.PropTypes.string,
	radioLabel: React.PropTypes.string,
	onConfirm: React.PropTypes.func
};
 
export default RadioButtonChoiceModal;
