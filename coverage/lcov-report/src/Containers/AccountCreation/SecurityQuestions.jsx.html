// Import dependencies
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage } from 'react-intl';
import { Form } from 'formsy-react';
import SecurityQuestion from '../../Components/AccountCreation/SecurityQuestion.jsx';
 
class SecurityQuestions extends Component {
 
	constructor(props) {
		super(props);
		// Set initial state
		this.state = {
			formData: {},
			formIsValid: false
		};
	}
 
	onForward() {
 
	}
 
	updateQuestion(name, val) {
		var delta = {};
		delta[name] = Object.assign(this.state.formData[name] || {}, { question: val });
		this.setState({ formData: Object.assign({}, this.state.formData, delta) });
		console.log(this.state.formData, val)
	}
 
	updateAnswer(name, val) {
		var delta = {};
		delta[name] = Object.assign(this.state.formData[name] || {}, { answer: val });
		this.setState({ formData: Object.assign({}, this.state.formData, delta) });
		// console.log(this.state.formData);
	}
 
	// Used to enable successive inputs. Deprecated. To add back, add attribute disabled={! (i === 0 || this.isSQComplete('sq' + (i - 1)))} to <SecurityQuestion...>
	isSQComplete(name) {
		return this.state.formData[name] && this.state.formData[name].question && this.state.formData[name].answer;
	}
 
	onFormValid() {
		// this is placeholder to know when the form is valid. It will likely change once we understand the SQ backend spec
		var complete = this.isSQComplete('sq0') && this.isSQComplete('sq1') && this.isSQComplete('sq2') && this.isSQComplete('sq3') && this.isSQComplete('sq4');
		this.setState({ formIsValid: complete });
	}
 
	onFormInValid() {
		this.setState({ formIsValid: false });
	}
 
	handleSubmitForm(data) {
		// console.log('handleSubmitForm', data);
	}
 
	render() {
		var questions = Array.apply(null, Array(5)).map((v, i) => {
			return <SecurityQuestion 
				key={i}
				name={'sq' + i}
				label={'Security Question ' + (i + 1)}
				questions={[
					'Lorem Ipsum',
					'Dolor Sit Amet',
					'Eu praesent sadipscing'
				]}
				htmlFor="securityQuestion"
				disabled={! (i === 0 || this.isSQComplete('sq' + (i - 1)))}
				requiredAnswerError={this.props.intl.messageIfExists('security-questions-required-answer-error')}
				answerErrorMaxLength={this.props.intl.messageIfExists('security-questions-required-answer-error')}
				onChangeQuestion={this.updateQuestion.bind(this)}
				onChangeAnswer={this.updateAnswer.bind(this)}
				modalHeader={this.props.intl.messageIfExists('security-questions-modal-header', { number: (i + 1) })}
			/>
		});
 
		return (
			<div className="security-questions">
				<span className="step-progress">
					<FormattedMessage id="account-creation-create-step-progress" values={{ page: 3, total: 4 }} />
				</span>
				<h1><FormattedMessage id="security-questions-header" /></h1>
				<p><FormattedMessage id="security-questions-header-text" /></p>
				<Form onValidSubmit={this.handleSubmitForm.bind(this)} onValid={this.onFormValid.bind(this)} onInvalid={this.onFormInValid.bind(this)}>
					{questions}
				</Form>
				<div className="row">
					<div className="col-xs-12 col-sm-4 col-sm-push-8 col-md-3 col-md-push-9 col-lg-3 col-lg-push-9">
						<button 
							id="security-questions-next"
							className="btn btn-primary btn-block" 
							type="submit" 
							disabled={!this.state.formIsValid}
						>
							<FormattedMessage id="security-questions-next" />
						</button>
					</div>
				</div>
			</div>
		);
	}
}
 
SecurityQuestions.propTypes = {
	intl: intlShape.isRequired
}
 
export default injectIntl(SecurityQuestions);
