import React, { Component } from 'react';
import AppDispatcher from '../Dispatchers/AppDispatcher';
import { IntlProvider, addLocaleData } from 'react-intl';
import Routes from './Routes.jsx';
import i18n from '../i18n.en.json';
import i18n_es from '../i18n.es.json';
import enLocaleData from 'react-intl/locale-data/en';
import esLocaleData from 'react-intl/locale-data/es';
 
 
// Add locale data for the languages we support
addLocaleData(enLocaleData);
addLocaleData(esLocaleData);
 
class AppIntl extends Component {
	
	constructor(props) {
		super(props);
		
		var locale = function() {
			var match = document.cookie.match(/(^|;)\s*ahbx_locale\s*=\s*([^;]+)/),
				matched = (match) ? match.pop() : 'en';
			// only support en / es
			return (matched === 'en' || matched === 'es') ? matched : 'en';
		}();
 
		this.state = {
			locale: locale,
			messages: (locale == 'en') ? i18n : i18n_es
		};
 
		/** set messages file based on locale **/
		AppDispatcher.register(payload => {
			if(payload.action.type === 'LOCALE_CHANGE') {
				var locale = payload.action.payload,
					cookie = "ahbx_locale=" + locale + ";path=/";
 
				// don't set the domain to avoid creating multiple copies of the cookie at slightly different domain definitions, unless IE, which requires it.
				//if(/MSIE|Trident|Edge/.test(window.navigator.userAgent) === true) {
			//		cookie += 'domain=' + window.location.hostname;
			//	}
				document.cookie = cookie;
				this.setState({ locale: locale, messages: (locale == 'en') ? i18n : i18n_es });
			}
		});		
	}
 
	render() {
		return (
			<IntlProvider key={this.state.locale} locale={this.state.locale} messages={this.state.messages}>
				<Routes />
			</IntlProvider>
		);
	}
 
}
 
export default AppIntl;
