// Import Libs & Components
import React, { Component } from 'react';
import AppDispatcher from '../../Dispatchers/AppDispatcher';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import { OptInFormPage } from '../../Components/Forms/OptInForm.jsx';
import ApplicationActions from '../../Actions/ApplicationActions.js';
 
class OptIn extends Component {
	/**
	 *
	 * @param props
	 */
	constructor(props) {
		super(props);
 
		// Set initial state
		this.state = {
			formIsValid: false,
			formData: Object.assign({}, props.applicationDetails)
		};
 
	}
 
	onFormChange(formData) {
		this.setState({formData: Object.assign({}, this.state.formData, formData)});
	}
 
	onFormValid() {
		this.setState({ formIsValid: true });
	}
 
	onFormInValid() {
		this.setState({ formIsValid: false });
	}
 
	handleSubmitForm() {
		// disable submit button during submission
		this.setState({ formIsValid: false});
 
		// TODO: Figure out what to do on submission. Just update?
		let appAction = this.state.formData.pendingAppId === null ? 'startApplication' : 'updateApplication';
		ApplicationActions[appAction](this.state.formData);
	}
 
	componentDidMount() {
		this.dispatchListener = AppDispatcher.register((payload) => {
			let action = payload.action;
			switch (action.type) {
				case 'APPLICATION_CREATE_SUCCESS':
				case 'APPLICATION_UPDATE_SUCCESS':
					this.context.router.push('/renewal/review/household');
					break;
				default:
					break;
			}
		});
	}
 
	componentWillUnmount() {
		AppDispatcher.unregister(this.dispatchListener);
	}
 
	/**
	 * component output
	 * @returns {XML}
	 */
	render() {
 
		return (
			<OptInFormPage 
				className="application-start"
				id="opt-in-form-page"
				formData={this.state.formData}
				onFormValid={this.onFormValid.bind(this)} 
				onFormInvalid={this.onFormInValid.bind(this)}
				formIsValid={this.state.formIsValid && this.state.formData.subsidizedApplication !== null}
				handleSubmitForm={this.handleSubmitForm.bind(this)}
				onChange={this.onFormChange.bind(this)}
			/>
		);
	}
}
 
OptIn.propTypes = {
	intl: intlShape.isRequired
};
 
OptIn.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(OptIn);
