import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage, FormattedDate } from 'react-intl';
import { ContentIntro, ContentCard, CardNavigation } from '../../Components/Layout/Layout.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
import { ConfirmModal } from '../../Components/Modal.jsx';
import UserHomeAction from '../../Actions/UserHomeAction';
import UserHomeStore from '../../Stores/UserHomeStore';
 
class Start extends Component {
 
	constructor(props) {
		super(props);
 
		// Set initial state
		this.state = {
			isNoChangesConfirmModalOpen: false
		};
	}
 
	onStart() {
		UserHomeAction.startRenewal({});
		this.context.router.push('/household/build');
	}
 
	onNoChanges() {
		this.setState({ isNoChangesConfirmModalOpen: true });
	}
 
	handleCloseConfirmModal() {
		UserHomeAction.startRenewal({});
		this.setState({ isNoChangesConfirmModalOpen: false });
		this.context.router.push('/household/build');
	}
 
	 handleConfirmSkip() {
		UserHomeAction.startRenewal({});
		this.setState({ isNoChangesConfirmModalOpen: false });
		this.context.router.push('/renewal/review/household');
	}
 
	closeRenewalConfirmModel() {
		this.setState({
			isNoChangesConfirmModalOpen: false
		});
	}
 
	render() {
		return (
			<div className="renewal-start">
			
				<ContentIntro className="illustration-text">
					<div className="illustration ic_renewal_dt hidden-xs" />
					<div className="illustration ic_renewal_dt hidden-sm hidden-md hidden-lg" />
					<FormattedMessage id="renewal-start-intro-header" tagName="h1" />
					<FormattedMessage id="renewal-start-intro-text" tagName="p" />
				</ContentIntro>
 
				<ContentCard>
					<CardNavigation
						text={this.props.intl.messageIfExists('renewal-account-home-link')}
						link="/account-home"
					/>
 
					<div className="row renewal-content">
						<div className="col-xs-12">
							<FormattedHTMLMessage id="renewal-start-text" tagName="p" />
						</div>
					</div>
 
					<ForwardBack
						id="renewal-forwardback"
						forwardText={<FormattedMessage id="renewal-get-started" />}
						backText={<FormattedMessage id="renewal-no-changes-to-report" />}
						onForward={this.onStart.bind(this)}
						onBack={this.onNoChanges.bind(this)}
						backButtonCustomClassNames="btn-link"
					/>
 
				</ContentCard>
 
				<ConfirmModal
					title={<div>{this.props.intl.messageIfExists('renewal-start-no-change-modal-title')}<button className="btn btn-close" onClick={this.closeRenewalConfirmModel.bind(this)} ><FormattedMessage id="modals.header.close" /></button></div>}
					className="renewal-confirm-no-changes-modal"
					isOpen={this.state.isNoChangesConfirmModalOpen}
					warningIcon={true}
					onRequestClose={this.handleCloseConfirmModal.bind(this)}
					onRequestConfirm={this.handleConfirmSkip.bind(this)}
					shouldCloseOnOverlayClick={false}
					contentLabel={this.props.intl.formatMessage({ id: 'renewal-start-no-change-modal-content-label' })}
					confirmText={this.props.intl.messageIfExists('renewal-start-no-change-modal-confirm')}
					cancelText={this.props.intl.messageIfExists('renewal-start-no-change-modal-cancel')}
				>
					<FormattedHTMLMessage id="renewal-start-no-change-modal-text" />
				</ConfirmModal>
 
			</div>
		);
	}
}
 
Start.contextTypes = {
	router: React.PropTypes.object.isRequired
};
 
export default injectIntl(Start);
