	 *

		super(props);
		
		// Set initial state
		this.state = {
			lifeEvent: HouseholdMemberActions.getLifeEvent(),
			subsidyConfirmationModalIsOpen: false
		};
 
		this.binds = {
			updateFromPeopleStore: () => {
				this.setState({ lifeEvent: PeopleStore.getLifeEvent() });
			}
		}
	}
 
	componentDidMount() {
		PeopleStore.on('change', this.binds.updateFromPeopleStore);
	}
 
	componentWillUnmount() {
		PeopleStore.removeListener('change', this.binds.updateFromPeopleStore);
	}
 
	checkSinglePersonFosterCare() {
		let persons = this.props.dynamicQuestionsForPerson.persons;
		return typeof persons !== "undefined" && Array.isArray(persons) && persons.length === 1 && persons[0].outputpersonlistname[0] === 'CP2' && persons[0].outputpersonlisttype[0] === '5';
	}
 
	isUnsubsidizedApplication() {
		let persons = this.props.dynamicQuestionsForPerson;
		return typeof persons !== "undefined" && typeof persons === "object" &&  persons.outputglobalresname[0] === 'CP1' && persons.outputglobalresvalue[0] === '1';
	}
 
	onForward() {
		if (this.isUnsubsidizedApplication() || this.checkSinglePersonFosterCare()){
			this.context.router.push('/renewal/review/individual');
		}
		else {
			this.context.router.push('/renewal/review/tax-information');
		}
	}
 
	
 
	onBack() {
		this.context.router.push('/renewal');
	}
 
	handleSubsidyConfirmationClick(e) {
		e.preventDefault();
		this.setState({ subsidyConfirmationModalIsOpen: true });
	}
 
	onRequestCloseSubsidyConfirmationModal() {
		this.setState({ subsidyConfirmationModalIsOpen: false });
	}
 
	onRequestConfirmSubsidyConfirmation() {
		// Change application status from Unsubsidized to Subsidized and navigate to 4.0 HH Menu with alerts to update information for each HHM.
		ApplicationActions.updateApplication({ subsidizedApplication: 'Y' });
		this.context.router.push('/household/menu');
	}
