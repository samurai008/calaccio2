// Import Libs & Components
import React, { Component } from 'react';
import { Link } from 'react-router';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import ApplicationActions from '../../Actions/ApplicationActions';
import { ContentIntro, ContentCard, CardNavigation } from '../../Components/Layout/Layout.jsx';
import HouseholdMemberList from '../../Components/Household/HouseholdMemberList.jsx';
import ToolTip from '../../Components/ToolTip.jsx';
import { ReviewSection, ReviewRow } from '../../Components/ReviewSection.jsx';
import MemberAvatar from '../../Components/MemberAvatar.jsx';
import MemberReviewContent from '../../Components/MemberReviewContent.jsx';
import { ConfirmModal } from '../../Components/Modal.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
import AppDispatcher from '../../Dispatchers/AppDispatcher';
 
class ApplicationReview extends Component {
	/**
	 *
	 * @param props
	 */
	constructor(props) {
		super(props);
		
		// Set initial state, modal is not open by default
		this.state = {
			subsidyConfirmationModalIsOpen: false
		};
	}
 
	handleSubsidyConfirmationClick(e) {
		e.preventDefault();
		this.setState({ subsidyConfirmationModalIsOpen: true });
	}
 
	onRequestCloseSubsidyConfirmationModal() {
		this.setState({ subsidyConfirmationModalIsOpen: false });
	}
 
	onRequestConfirmSubsidyConfirmation() {
		// Change application status from Unsubsidized to Subsidized and navigate to 4.0 HH Menu with alerts to update information for each HHM.
		ApplicationActions.updateApplication({ subsidizedApplication: 'Y', agreeToConsent: 'Y' });
		this.dispatchListener = AppDispatcher.register((payload) => {
			let action = payload.action;
			switch (action.type) {
				case 'APPLICATION_UPDATE_SUCCESS':
					this.context.router.push('/household/menu');
					break;
				default:
					break;
			}
		});
		
	}
 
	onForward() {
		this.context.router.push('/review/household');
	}
 
	onBack() {
		
	}
	
	/**
	 * component output
	 * @returns {XML}
	 */
	render() {
 
		var householdMembers = this.props.householdMembers 
									? this.props.householdMembers.map((item, i) => {
										return (
											<MemberAvatar key={i} className="household-member" person={item} avatarSize="xs" />
										);
									})
									: [];
 
		return (
			<div className="review-page application-review">
				
				<ContentIntro className="hidden-xs" />
				
				<ContentCard>
					<CardNavigation
						text={this.props.intl.messageIfExists('application-review-card-nav-link')}
						link="/application-menu"
					>
						<em><FormattedMessage id="application-review-card-nav-text" /></em>
					</CardNavigation>
 
					<FormattedMessage id="application-review-header" tagName="h1" />
					<p>
						<FormattedMessage id="application-review-text" />
						&nbsp;
						<ToolTip text={this.props.intl.messageIfExists('application-review-tooltip')} id={this.props.intl.messageIfExists('application-review-tooltip')} />
					</p>
					<br/>
 
					{
						typeof this.props.applicationDetails !== "undefined" && this.props.applicationDetails.subsidizedApplication === 'N'
						? (
							<ReviewSection>
								<ReviewRow id="application-review-subsidies">
									<FormattedMessage id="application-review-applying-for-subsidies-label" tagName="h5" />
									<FormattedMessage id="application-review-applying-for-subsidies-value-no" tagName="p" />
									<br />
									<button className="btn-link" onClick={this.handleSubsidyConfirmationClick.bind(this)}><FormattedMessage id="application-review-applying-for-subsidies-link" tagName="em" /></button>
							
								</ReviewRow>
							</ReviewSection>
						) : null
					}
 
					<FormattedMessage id="application-review-household-members-header" tagName="h2" />
 
					<ReviewSection className="review-household-members">
						<ReviewRow id="review-household-members" linkUrl="/household/build">
							<FormattedMessage id="application-review-household-members-list-header" tagName="h5" />
							<div className="household-members">
								{householdMembers}
							</div>
						</ReviewRow>
					</ReviewSection>
 
					{
						this.props.householdMembers.map((member, i) => {
							return (
								<MemberReviewContent
									id={'member-review-' + i}
									key={i}
									person={member}
									otherHouseholdMembers={this.props.householdMembers.filter(p => p.personId !== member.personId)}
									editUrlBase="/household/edit-review/{personId}/"
								/>
							);
						})
					}
					
					<ForwardBack
						id="application-review-forwardback"
						forwardText={<FormattedMessage id="application-review-btn-save" />}
						backText={<FormattedMessage id="application-review-btn-back" />}
						showBack={false}
						onForward={this.onForward.bind(this)}
						onBack={this.onBack.bind(this)}
					/>
				</ContentCard>
 
				<ConfirmModal
					isOpen={this.state.subsidyConfirmationModalIsOpen}
					cancelText={this.props.intl.messageIfExists('application-review-subsidy-confirmation-modal-btn-cancel')}
					confirmText={this.props.intl.messageIfExists('application-review-subsidy-confirmation-modal-btn-confirm')}
					onRequestClose={this.onRequestCloseSubsidyConfirmationModal.bind(this)}
					onRequestConfirm={this.onRequestConfirmSubsidyConfirmation.bind(this)}
					contentLabel={this.props.intl.messageIfExists('application-review-subsidy-confirmation-modal-header')}
				>
					<FormattedMessage id="application-review-subsidy-confirmation-modal-header" tagName="h1" />
					<FormattedMessage id="application-review-subsidy-confirmation-modal-text" tagName="p" />
				</ConfirmModal>
				
			</div>
		);
	}
}
 
ApplicationReview.propTypes = {
	intl: intlShape.isRequired
};
 
ApplicationReview.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(ApplicationReview);
