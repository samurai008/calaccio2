// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage } from 'react-intl';
import PeopleStore from '../../Stores/PeopleStore';
import { ContentIntro, ContentCard, CardNavigation } from '../../Components/Layout/Layout.jsx';
import { ReviewSection, ReviewRow, ReviewDT, ReviewDD } from '../../Components/ReviewSection.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
import MoneyValue from '../../Components/MoneyValue.jsx';
import MemberReview from '../../Components/MemberReview.jsx';
import HouseholdMemberIncomeSummary from '../../Components/IncomeApp/HouseholdMemberIncomeSummary.jsx';
import ApplicationStore from '../../Stores/ApplicationStore.js';
 
const IncomeReview = props => {
	var phi = PeopleStore.getHouseholdNetIncome();
 
	return (
		<div className="review-page household-income-review">
			
			<ContentIntro className="hidden-xs" />
			
			<ContentCard>
					{
						ApplicationStore.isNewApplication() === true
						?
							<CardNavigation
								text={props.intl.messageIfExists('application-review-card-nav-link')}
								link={props.cardNavLink}
							>
								<em><FormattedMessage id="application-review-card-nav-text" /></em>
							</CardNavigation>
						: 
							<CardNavigation
								text={props.intl.messageIfExists('household-review-header')}
								link="/review/household"
							>
								<em><FormattedMessage id="application-review-card-nav-text" /></em>
							</CardNavigation>
					}
 
				<FormattedMessage id="income-review-header" tagName="h1" />
					
				<br/>
 
				<ReviewSection>
					<ReviewRow id="review-income-projected-household-income" linkUrl="/income/review">
						<div className="row money-row phi">
							<div className="col-xs-12">
								<ReviewDT><FormattedMessage id="income-review-phi-label" /></ReviewDT>
								<ReviewDD>
									<MoneyValue
										amount={phi}
										timeframe="incomeSummary.timeframe.annually"
										showTimeframeIfZero={true}
										useSlash={props.currencyUseSlash}
									/>
								</ReviewDD>
							</div>
						</div>
					</ReviewRow>
				</ReviewSection>
				{
					props.householdMembers.map((member, i) => {
						var moneyValue = props.showMemberPAI && member.projectedAnnualIncome >= 0
											? (
												<ReviewDD className="total">
													<MoneyValue
														amount={member.projectedAnnualIncome}
														timeframe="incomeSummary.timeframe.annually"
														showTimeframeIfZero={true}
														useSlash={true}
													/>
												</ReviewDD>
											)
											: null;
 
						return (
							<ReviewSection key={member.personId}>
								<ReviewRow id={'review-income-person-' + i} linkUrl={'/income/person/' + member.personId + '/review'}>
									<div className="row money-row">
										<MemberReview 
											className="col-xs-12 col-sm-12" 
											person={member}
											avatarChildren={moneyValue}
										>
											<HouseholdMemberIncomeSummary
												incomes={member.incomes}
												deductions={member.deductions}
												currencyUseSlash={props.currencyUseSlash}
											/>
										</MemberReview>
									</div>
								</ReviewRow>
								{
									member.totalMonthlyIncome > 0
									?
										<ReviewRow  id={'review-income-person-cmi' + i}>
											<div className="row money-row">
												<div className="col-xs-6">
													<ReviewDT><FormattedMessage id="income-review-cmi-label" /></ReviewDT>
												</div>
												<div className="col-xs-6">
													<ReviewDD>
														<MoneyValue
															amount={member.totalMonthlyIncome}
															timeframe="incomeSummary.timeframe.monthly"
															showTimeframeIfZero={true}
															useSlash={props.currencyUseSlash}
														/>
													</ReviewDD>
												</div>
											</div>
										</ReviewRow>
									: null
								}
								{
									member.projectedAnnualIncome > 0
									?
										<ReviewRow  id={'review-income-person-pai' + i}>
											<div className="row money-row">
												<div className="col-xs-6">
													<ReviewDT><FormattedMessage id="income-review-pai-label" /></ReviewDT>
												</div>
												<div className="col-xs-6">
													<ReviewDD>
														<MoneyValue
															amount={member.projectedAnnualIncome}
															timeframe="incomeSummary.timeframe.annually"
															showTimeframeIfZero={true}
															useSlash={props.currencyUseSlash}
														/>
													</ReviewDD>
												</div>
											</div>
										</ReviewRow>
									: null
								}
							</ReviewSection>
						);
					})
				}
				
				<ForwardBack
					id="review-income-review-forwardback"
					forwardText={<FormattedMessage id="application-review-btn-save" />}
					backText={<FormattedMessage id="application-review-btn-back" />}
					showBack={true}
					forwardLink={props.forwardLink}
					backLink={props.backLink}
				/>
			</ContentCard>
			
		</div>
	);
}
 
IncomeReview.propTypes = {
	intl: intlShape.isRequired,
	householdMembers: React.PropTypes.array
};
 
IncomeReview.defaultProps = {
	cardNavLink: '/household/menu',
    forwardLink: '/review/individual',
    backLink: '/review/tax-information',
    showMemberPAI: false,
    currencyUseSlash: false
};
 
IncomeReview.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(IncomeReview);
