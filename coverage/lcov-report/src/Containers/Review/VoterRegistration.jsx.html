// Import Libs & Components
import React, { Component } from 'react';
import AppDispatcher from '../../Dispatchers/AppDispatcher.js';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import { ContentIntro, ContentCard, CardNavigation } from '../../Components/Layout/Layout.jsx';
import { VoterRegistrationForm } from '../../Components/Review/VoterRegistrationForm.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
import EligibilityActions from '../../Actions/EligibilityActions.js';
import VoterRegistrationStore from '../../Stores/VoterRegistrationStore.js';
import ApplicationStore from '../../Stores/ApplicationStore.js';
class VoterRegistration extends Component {
	/**
	 *
	 * @param props
	 */
	constructor(props) {
		super(props);
 
		// Set initial state, modal is not open by default
		this.state = {
			formIsValid: true,
			formData: VoterRegistrationStore.getDetails()
		};
 
		// will hold a ref to the external link a tag
		this.externalLink = null;
 
		this.binds = {
			handleRegistrationUpdate: () => {
				this.setState({formData: VoterRegistrationStore.getDetails()});
			}
		}
	}
 
	componentDidMount() {
		VoterRegistrationStore.on('change', this.binds.handleRegistrationUpdate);
		EligibilityActions.getVoterRegistration();
		// Listen for save, and redirect to portal
		this.dispatcherId = AppDispatcher.register(data => {
			switch(data.action.type) {
				case 'VOTER_REGISTRATION_SAVED':
					setTimeout(() => { this.context.router.push(this.props.forwardLink) }, 1);
					break;
			}
		});
	}
 
	/**
	 * actions to perform when routing away
	 */
	componentWillUnmount() {
		VoterRegistrationStore.removeListener('change', this.binds.handleRegistrationUpdate);
		AppDispatcher.unregister(this.dispatcherId);
	}
 
	onFormValid() {
		this.setState({ formIsValid: true });
	}
 
	onFormInValid() {
		this.setState({ formIsValid: true });
	}
 
	onFormChange(formData) {
		if(typeof formData === "object" && typeof formData.voterRegistration === "object" && formData.voterRegistration.registerToVote === 'X') this.handleExternalVoterRegistration();
		this.setState(Object.assign(this.state.formData, formData));
	}
 
	/**
	 * clicks the hidden link
	 */
	handleExternalVoterRegistration() {
		this.externalLink.click();
	}
 
	onForward() {
		EligibilityActions.updateVoterRegistration(this.state.formData, this.props.intl.locale);
	}
 
	render() {
		return (
			<div className="voter-registration">
				<ContentIntro>
					<div className="illustration ic_vote_illustration_desktop hidden-xs" />
					<div className="illustration ic_vote_illustration_mobile hidden-sm hidden-md hidden-lg" />
					<h1><FormattedMessage id="voter-registration-header" /></h1>
					<p><FormattedHTMLMessage id="voter-registration-text" /></p>
				</ContentIntro>
 
				<ContentCard>
					{
						ApplicationStore.isNewApplication() === true
						?
							<CardNavigation
								text={this.props.intl.messageIfExists('application-menu')}
								link="/application-menu"
							/>
						: 
							<CardNavigation
								text={this.props.intl.messageIfExists('household-review-header')}
								link="/review/household"
							/>
 
					}
					<div className="row">
						<div className="col-xs-12">
 
								<VoterRegistrationForm
									formData={this.state.formData}
									onChange={this.onFormChange.bind(this)}
									onFormValid={this.onFormValid.bind(this)}
									onFormInValid={this.onFormInValid.bind(this)}
								/>
 
								{/* this link will be "clicked" when the first option is chosen in the form, not displayed */}
								<a
									className="hidden"
									ref={r =>	this.externalLink = r }
									href={this.props.intl.messageIfExists('voter-registration-external-link-href')}
									target="_blank">
									<FormattedMessage id="voter-registration-external-link-label" />
								</a>
 
								<ForwardBack
									id="voter-registration-forwardback"
									forwardText={<FormattedMessage id="voter-registration-next-label" />}
									forwardIsDisabled={!this.state.formIsValid}
									backText={<FormattedMessage id="voter-registration-back-label" />}
									showBack={true}
									onForward={this.onForward.bind(this)}
									backLink={this.props.backLink}
								/>
						</div>
					</div>
				</ContentCard>
			</div>
		);
	}
}
 
VoterRegistration.propTypes = {
	intl: intlShape.isRequired
};
 
VoterRegistration.defaultProps = {
	forwardLink: '/review/application-signature',
	backLink: '/review/individual'
};
 
VoterRegistration.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(VoterRegistration);
