// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage } from 'react-intl';
import { ContentIntro, ContentCard, CardNavigation } from '../../Components/Layout/Layout.jsx';
import { ReviewSection, ReviewRow, ReviewDT, ReviewDD } from '../../Components/ReviewSection.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
import TaxReviewContent from '../../Components/Review/TaxReviewContent.jsx';
import ApplicationStore from '../../Stores/ApplicationStore.js';
 
const TaxReview = props => {
	return (
		<div className="review-page application-review">
			
			<ContentIntro className="hidden-xs" />
			
			<ContentCard>
					{
						ApplicationStore.isNewApplication() === true
						?
							<CardNavigation
								text={props.intl.messageIfExists('application-review-card-nav-link')}
								link={props.cardNavLink}
							>
								<em><FormattedMessage id="application-review-card-nav-text" /></em>
							</CardNavigation>
						: 
							<CardNavigation
								text={props.intl.messageIfExists('household-review-header')}
								link="/review/household"
							>
								<em><FormattedMessage id="application-review-card-nav-text" /></em>
							</CardNavigation>
					}
 
				<FormattedMessage id="tax-review-final-header" tagName="h1" />
				{
					props.subHeaderTextId ? (
						<FormattedMessage id={props.subHeaderTextId} tagName="p" />
					) : null
				}
					
				<br/>
 
				<TaxReviewContent
					id="tax-review-content"
					householdMembers={props.householdMembers}
					editUrlBase={props.editUrlBase}
				/>
 
				<ForwardBack
					id="tax-review-forwardback"
					forwardText={<FormattedMessage id="application-review-btn-save" />}
					backText={<FormattedMessage id="application-review-btn-back" />}
					showBack={true}
					forwardLink={props.forwardLink}
					backLink={props.backLink}
				/>
			</ContentCard>
			
		</div>
	);
}
 
TaxReview.propTypes = {
	intl: intlShape.isRequired,
	householdMembers: React.PropTypes.array,
	editUrlBase: React.PropTypes.string
};
 
TaxReview.defaultProps = {
	forwardLink: '/review/income',
	backLink: '/review/household',
	cardNavLink: '/household/menu',
	subHeaderTextId: '',
	editUrlBase: '/household/tax-information/edit-review/'
};
 
TaxReview.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(TaxReview);
