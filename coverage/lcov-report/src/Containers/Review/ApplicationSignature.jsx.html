// Import Libs & Components
import React, { Component } from 'react';
import AppDispatcher from '../../Dispatchers/AppDispatcher.js';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import { ContentIntro, ContentCard, CardNavigation } from '../../Components/Layout/Layout.jsx';
import ApplicationSignatureForm from '../../Components/Review/ApplicationSignatureForm.jsx';
import ConditionalField from '../../Components/ConditionalField.jsx';
import { ForwardBack } from '../../Components/ActionsRow.jsx';
import EligibilityActions from '../../Actions/EligibilityActions.js';
import NavActions from '../../Actions/NavActions.js';
import ApplicationSignatureStore from '../../Stores/ApplicationSignatureStore.js';
import ApplicationStore from '../../Stores/ApplicationStore.js';
import ConfigStore from '../../Stores/ConfigStore.js';
import AccountStore from '../../Stores/AccountStore.js';
import AuthStore from '../../Stores/AuthStore.js';
import { deepGet } from '../../Components/Forms/FormHOC.jsx';
import Modal, { NoticeModal } from '../../Components/Modal.jsx';
var isValidPin = true;
 
class ApplicationSignature extends Component {
	/**
	 *
	 * @param props
	 */
	constructor(props) {
		super(props);
 
		// Set initial state, modal is not open by default
		this.state = {
			formIsValid: false,
			formData: ApplicationStore.isRACApplication() === true ? {} : ApplicationSignatureStore.getDetails(),
			submissionInProgress: false,
			isAccessModalOpen: false,
			accessCode: 0,
			progressModalIsOpen: true,
			isConsentYearsModalOpen: false,
			consentYearsModalNotYetOpened: true,
		};
 
		this.binds = {
			handleSignatureUpdate: this.onSignatureUpdate.bind(this)
		};
	}
	onSignatureUpdate() {
		let appDetails = ApplicationSignatureStore.getDetails();
 
		// if in RAC, only update with changelog
		if(ApplicationStore.isRACApplication() === true) {
			this.setState(Object.assign(this.state.formData,  { appSignature: { persons: appDetails.appSignature.persons }}));
		}
		else {
			// for intake applications for non-consumers, display the access code modal instead of redirecting.
			// add check for access code- if the app is started as a consumer, but finished by an admin, we won't have an access code, so don't want to show the popup
			if (ApplicationStore.isNewApplication() && AuthStore.userHasRole('consumer') === false && this.state.submissionInProgress === true && typeof appDetails.appSignature.accessCode  !== 'undefined') {
				return this.setState({
					submissionInProgress: false,
					isAccessModalOpen:    true,
					accessCode:           appDetails.appSignature.accessCode
				});
			}
		}
 
		// if the signature store updates, this condition should be reached:
		// - in the case that the user is not an admin in intake
		// - the user is or is not an admin in RAC
		// if a signature submission is in progress, reset the state and redirect out.
		if(this.state.submissionInProgress === true) {
			this.setState({
				submissionInProgress: false
			});
			setTimeout(() => {
				NavActions.redirectToEligibility();
			}, 0);
		}
	}
 
	componentDidMount() {
		ApplicationSignatureStore.on('change', this.binds.handleSignatureUpdate);
		this.dispatchListener = AppDispatcher.register((payload) => {
 
			let action = payload.action;
			switch (action.type) {
				case 'PIN_IS_NOT_VALID':
					isValidPin =  false;
					this.setState({submissionInProgress: false});
				default:
					break;
			}
		});
 
		EligibilityActions.getApplicationSignature();
	}
 
	/**
	 * actions to perform when routing away
	 */
	componentWillUnmount() {
		ApplicationSignatureStore.removeListener('change', this.binds.handleSignatureUpdate);		
		AppDispatcher.unregister(this.dispatchListener);
	}
 
	onFormValid() {
		this.setState({ formIsValid: true });
	}
 
	onFormInValid() {
		this.setState({ formIsValid: false });
	}
 
	handleCloseModal(){
		let modalWasConsentYears = this.state.isConsentYearsModalOpen === true;
 
		this.setState({
			isAccessModalOpen: false,
			isConsentYearsModalOpen: false,
			consentYearsModalNotYetOpened: false
		});
 
		if (!modalWasConsentYears) NavActions.redirectToEligibility();
	}
 
	onPinRefSubmit(){
		isValidPin = true;
	}
 
	onFormChange(formData) {
		if(typeof formData.changeLogDelta !== "undefined") {
			let existingChangeLog = Object.assign({}, this.state.formData),
			person = existingChangeLog.appSignature.persons.find(person => person.personId === formData.changeLogDelta.personId);
 
			// replace the changelog object at that index, for the person
			if(typeof person !== "undefined" ) {
				person.changeLog[formData.changeLogDelta.index] = formData.changeLogDelta.changeLog;
			}
			Object.assign(formData, existingChangeLog);
			delete formData.changeLogDelta;
		}
 
		if(this.state.consentYearsModalNotYetOpened && formData.appSignature.consentForYears === '0') {
			this.state.isConsentYearsModalOpen = true;
		}
 
		this.setState(Object.assign(this.state.formData, formData));
	}
 
	onForward() {
		EligibilityActions.updateApplicationSignature(this.state.formData, this.props.intl.locale);
		this.setState({submissionInProgress: true});
	}
 
	handleScroll(e) {
		if(e.target.scrollHeight - e.target.scrollTop <= 325) this.setState({formData: Object.assign(this.state.formData, {termsAndConditionsRead: true})});
	}
	/**
	 * component output
	 * @returns {XML}
	 */
	render() {
 
		var appFormProps = {
				isRAC: ! ApplicationStore.isNewApplication() && ApplicationStore.isRACApplication(),
				isRenewal: this.props.isRenewal
			};
		return (
			<section>
				<div className="application-signature">
					<ContentIntro className="hidden-xs" />
					<ContentCard>
					{
						ApplicationStore.isNewApplication() === true
						?
							<CardNavigation
								text={this.props.intl.messageIfExists('application-menu')}
								link="/application-menu">
								<em><FormattedMessage id="household-build-menu-text" /></em>
							</CardNavigation>	
						: 
							<CardNavigation
								text={this.props.intl.messageIfExists('household-review-header')}
								link="/review/household"
							>
								<em><FormattedMessage id="application-review-card-nav-text" /></em>
							</CardNavigation>
					}					
 
						<div className="row">
							<div className="col-xs-12">
								<FormattedMessage id={'application-signature-header' + (appFormProps.isRAC ? '-rac' : (appFormProps.isRenewal ? '-renewal' : ''))} tagName="h1" />
								<p><FormattedMessage id={'application-signature-subheader' + (appFormProps.isRAC ? '-rac' : (appFormProps.isRenewal ? '-renewal' : ''))} tagName="b" /></p>
								<br/>
								<br/>
								<ApplicationSignatureForm
									formData={this.state.formData}
									onChange={this.onFormChange.bind(this)}
									onFormValid={this.onFormValid.bind(this)}
									onFormInValid={this.onFormInValid.bind(this)}
									onScroll={this.handleScroll.bind(this)}
									householdMembers={this.props.householdMembers}
									applicationDetails={this.props.applicationDetails}
									showTextStep={this.props.showTextStep}
									isValidPin={isValidPin}
									onPinRefSubmit={this.onPinRefSubmit}
									{...appFormProps}
								/>
								<ForwardBack
									id="application-signature-forwardback"
									forwardText={<FormattedMessage id="application-signature-next-label" />}
									forwardIsDisabled={
										this.state.formIsValid === false
										|| deepGet('appSignature.agreeToLegalReportingRequirements', this.state.formData) === false
										|| deepGet('appSignature.agreeToLegalTermsAndConditions', this.state.formData) === false
										|| (AuthStore.userHasRole('admin') === true && deepGet('appSignature.confirmApplicantPermission', this.state.formData) === false)
										|| this.state.submissionInProgress
									}
									backText={<FormattedMessage id="application-signature-back-label" />}
									showBack={true}
									onForward={this.onForward.bind(this)}
									backLink={this.props.backLink}
								/>
							</div>
						</div>
					</ContentCard>
				</div>
 
				<Modal
					className="access-code-modal"
					isOpen={this.state.isAccessModalOpen}
					contentLabel={this.props.intl.messageIfExists('address-input-address-confirmation-modal-header')}
					shouldCloseOnOverlayClick={false}
					fullScreenMobile={true}
					onRequestClose={this.handleCloseModal.bind(this)}
				>
					<div className="modal-header">
						<FormattedMessage id="application-signature-access-code-alert" tag="h1"/>
					</div>
					<FormattedMessage id="application-signature-access-code-alert-explain" tag="h1"/>
					<p style={{marginTop: '2em', textAlign: 'center'}}>
						<FormattedMessage id="application-signature-access-code-access" tag="span"/>
						{this.state.accessCode}
					</p>
					<div className="footer modal-buttons text-center">
						<button 
							id="address-confirmation-btn-confirm" 
							className="btn btn-primary" 
							type="submit" 
							onClick={this.handleCloseModal.bind(this)}
						>
							<FormattedMessage id="global-element-multitab-error-message-modal-confirm"/>
						</button>
					</div>
				</Modal>
 
				<NoticeModal
					isOpen={this.state.isConsentYearsModalOpen && this.state.consentYearsModalNotYetOpened}
					fullScreenMobile={true}
					onRequestClose={this.handleCloseModal.bind(this)}
					title=" "
					contentLabel=" "
					confirmText={this.props.intl.messageIfExists('global-element-multitab-error-message-modal-confirm')}
				>
					<FormattedHTMLMessage id="application-signature-consent-years-modal-text" tag="h1"/>
				</NoticeModal>
				
			</section>
		);
	}
}
 
ApplicationSignature.propTypes = {
	intl: intlShape.isRequired,
	householdMembers: React.PropTypes.array.isRequired,
	showTextStep: React.PropTypes.bool,
	backLink: React.PropTypes.string,
	applicationDetails: React.PropTypes.object
};
 
ApplicationSignature.defaultProps = {
	showTextStep: false,
	backLink: '/review/voter-registration',
	isRenewal: false
};
 
ApplicationSignature.contextTypes = {
	router: React.PropTypes.object.isRequired
};
 
export default injectIntl(ApplicationSignature);
