import React, { Component } from 'react';
import HouseholdMemberActions from '../../Actions/HouseholdMemberActions.js';
import IncomeActions from '../../Actions/IncomeApp/IncomeActions.js';
import DynamicQuestionsActions from '../../Actions/DynamicQuestionsActions';
import PeopleStore from '../../Stores/PeopleStore.js';
import { ContentIntro, ContentCard, CardNavigation } from '../../Components/Layout/Layout.jsx';
import ApplicationStore from '../../Stores/ApplicationStore.js';
import DynamicQuestionsStore from '../../Stores/DynamicQuestionsStore';
import { Modal, NoticeModal, ConfirmModal, ModalHeader, ProgressModal } from '../../Components/Modal.jsx';
import AppDispatcher from '../../Dispatchers/AppDispatcher.js';
 
class Main extends Component {
	constructor(props) {
		super(props);
 
		// Set initial state
		this.state = {
			householdMembers: PeopleStore.getAll(),
			applicationDetails: ApplicationStore.getDetails(),
			dynamicQuestionsForPerson: {}
		};
 
		this.binds = {
			householdUpdate: () => {
				// update the local state list of household members from the store
				this.setState({ householdMembers: PeopleStore.getAll() });
			},
			getAllHouseholdMembers: () => {
				// call the household members endpoint to update people in the store
				HouseholdMemberActions.getAll();
				HouseholdMemberActions.getEnrollmentPeriodDetails();
			},
			applicationUpdate: () => {
				this.setState({ applicationDetails: ApplicationStore.getDetails() });
			},
			DynamicQuestionsRacRetrieved: () =>{
				this.setState({ dynamicQuestionsForPerson : DynamicQuestionsStore.getDynamicQuestions() });
				}
		};
	}
 
	componentWillMount () {
		DynamicQuestionsActions.retrieveQuestions();
	}
 
	componentDidMount() {
		// listen to change events from PeopleStore to update household members on each store change.
		PeopleStore.on('change', this.binds.householdUpdate);
		// listen specifically for the income-list event which signals that PeopleStore's income data was updated
		// this is to chain calls to income endpoint and people endpoint without their updates to PeopleStore colliding.
		PeopleStore.on('income-list', this.binds.getAllHouseholdMembers);
		// get the income data from the endpoint, which will update PeopleStore
		IncomeActions.getAll();
		DynamicQuestionsStore.on('change', this.binds.DynamicQuestionsRacRetrieved);
		ApplicationStore.on('change', this.binds.applicationUpdate);
		this.dispatcherId = AppDispatcher.register(data => {
			switch (data.action.type) {
				case 'SAVE_EXIT_REQUESTED':
					setTimeout(function(){
						AppDispatcher.dispatch({
							action: {
								type: 'GENERIC_SAVE_EXIT',
								payload: {}
							}
						});
					}, 1);
					break;
			}
		});
	}
 
	componentWillUnmount() {
		// remove listener for all store events on unmount
		PeopleStore.removeListener('change', this.binds.householdUpdate);
		PeopleStore.removeListener('income-list', this.binds.getAllHouseholdMembers);
		ApplicationStore.removeListener('change', this.binds.applicationUpdate);
		DynamicQuestionsStore.removeListener('change', this.binds.DynamicQuestionsRacRetrieved);
		AppDispatcher.unregister(this.dispatcherId);
	}
 
	render() {
		return (
			<div className="household">
				{
					React.cloneElement(this.props.children, {
						householdMembers: this.state.householdMembers,
						applicationDetails: this.state.applicationDetails,
						dynamicQuestionsForPerson: this.state.dynamicQuestionsForPerson
					})
				}
			</div>
		);
	}
}
 
export default Main;
