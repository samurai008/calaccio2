// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import { ContentIntro, ContentCard } from '../../Components/Layout/Layout.jsx';
import MenuItem from '../../Components/ApplicationMenu/MenuItem.jsx';
import ApplicationStore from '../../Stores/ApplicationStore.js';
import ApplicationActions from '../../Actions/ApplicationActions.js';
import ConfigStore from '../../Stores/ConfigStore.js';
import AppDispatcher from '../../Dispatchers/AppDispatcher.js';
 
class ApplicationMenu extends Component {
	/**
	 *
	 * @param props
	 */
	constructor(props) {
		super(props);
 
		// Set initial state, modal is not open by default
		this.state = {
 
		};
	}
 
	componentWillMount() {
	}
	
	componentDidMount() {
		ApplicationActions.loadApplication();
		this.dispatcherId = AppDispatcher.register(data => {
			switch (data.action.type) {
				case 'SAVE_EXIT_REQUESTED':
					setTimeout(function(){
						AppDispatcher.dispatch({
							action: {
								type: 'GENERIC_SAVE_EXIT',
								payload: {}
							}
						});
					}, 1);
					break;
			}
		});
	}
 
	componentWillUnmount() {
		AppDispatcher.unregister(this.dispatcherId);
	}
 
	render() {
		return (
			<div className="application-menu menu-page">
				<ContentIntro />
				<ContentCard>
					<div className="row">
						<div className="col-xs-12">
							<FormattedMessage id="application-menu-header" tagName="h1" />
							<div className="menu-items">
								<MenuItem
									id="application-menu-item-introduction"
									title={<FormattedMessage id="application-menu-item-introduction" />}
									state={this.props.applicationDetails.applicationMenu.introduction}
									link= {this.props.applicationDetails.applicationMenu.introduction === "review" ? "/application-start" : "/application-start"}
									disabled={this.props.applicationDetails.applicationMenu.introduction === "na"}
								/>
								<MenuItem
									id="application-menu-item-household"
									title={<FormattedMessage id="application-menu-item-household" />}
									state={this.props.applicationDetails.applicationMenu.householdInfo}
									link="/household/menu"
									error={this.props.applicationDetails.applicationMenu.houseHoldInfoDone === false && this.props.applicationDetails.applicationMenu.householdInfo !== "start" ? <FormattedMessage id="household-menu-update-warning" /> : ''}
									disabled={this.props.applicationDetails.applicationMenu.householdInfo === "na"}
								/>
								<MenuItem
									id="application-menu-item-individual"
									title={<FormattedMessage id="application-menu-item-individual" />}
									state={this.props.applicationDetails.applicationMenu.individualInfo}
									link={this.props.applicationDetails.applicationMenu.individualInfo === "review" ? "/review/individual" : "/individual/menu"} 
									disabled={this.props.applicationDetails.applicationMenu.individualInfo === "na"}
								/>
								<MenuItem
									id="application-menu-item-review"
									title={<FormattedMessage id="application-menu-item-review" />}
									state={this.props.applicationDetails.applicationMenu.reviewAndSubmit}
									link={ApplicationStore.isRenewalApplication() ? '/renewal/review/household' : '/review/household'}
									disabled={this.props.applicationDetails.applicationMenu.reviewAndSubmit === "na"}
								/>
							</div>
						</div>
					</div>
				</ContentCard>
			</div>
		);
	}
}
 
ApplicationMenu.propTypes = {
	intl: intlShape.isRequired
};
 
ApplicationMenu.contextTypes = {
    router: React.PropTypes.object.isRequired
};
ApplicationMenu.defaultProps = {
	applicationDetails: {
		applicationMenu: {}
	}
}
 
export default injectIntl(ApplicationMenu);
