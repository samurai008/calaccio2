	onFormChange(data) {
		if(data.primaryContactId !== this.state.formData.primaryContactId) {
			// blow away primary contact info in formData if person changes
			let person = PeopleStore.getPersonById(data.primaryContactId);
			person.primaryContact = 'Y';
			let formData = Object.assign({}, this.state.formData, data, {
				selectedPersonsNotLivingWithPC: [],
				primaryContactData: person
			});
			return this.setState({ formData: formData });
		}
 
		if(data.isUSNational === "") {
			data.immigrationDocumentStatus = "";
			Object.keys(data).forEach( key => {
				if (key.indexOf('militaryStatus') !== -1) data[key] = null;
			});
		}
 
 
		if(Array.isArray(data.applyingForHealthCoverage)) {
			data.applyingForHealthCoverage.forEach( personId => {
				let person = PeopleStore.getPersonById(personId);
				if(person.applyingForHealthCoverage === 'N' && this.state.formData.applyingForHealthCoverage.indexOf(personId) === 'Y'){
					data['militaryStatus[' + personId + ']'] = null;
				}
			});
		}
 
 
 
		if(Array.isArray(data.isUSNational) && Array.isArray(data.immigrationDocumentStatus)) {
			data.isUSNational.forEach( personId => {
				let personIndex = data.immigrationDocumentStatus.indexOf(personId);
				if( personIndex !== -1) {
					data.immigrationDocumentStatus.splice(personIndex, 1);
				}
				data['militaryStatus[' + personId + ']'] = null;
			});
		}
		
		//Reset AI/AN data if member removed from that list
		if(typeof data.nativeAmerican !== "undefined" && Array.isArray(data.nativeAmerican.isNativeAmerican)) {
			//List of current personIds for nativeAmerican, and make sure tribe meta data aligns
			Object.keys(data.nativeAmerican).forEach( key => {
				var matches = key.match(/\[(.*?)\]/);
				if (matches) {
					var personId = matches[1];
				}
 
				//If member isNativeAmerican but !isTribeMember, clear tribeState/tribeName
				if (key.indexOf('isTribeMember[' + personId + ']') !== -1 && data.nativeAmerican[key] === "N") {
					delete data.nativeAmerican['tribeState[' + personId + ']'];
					delete data.nativeAmerican['tribeName[' + personId + ']'];
				}
				
				if (key.indexOf('isTribeMember[' + personId + ']') !== -1 && data.nativeAmerican.isNativeAmerican.indexOf(personId) === -1) {
					delete data.nativeAmerican[key];
				}
				if (key.indexOf('tribeState[' + personId + ']') !== -1 && data.nativeAmerican.isNativeAmerican.indexOf(personId) === -1) {
					delete data.nativeAmerican[key];
				}
				if (key.indexOf('tribeName[' + personId + ']') !== -1 && data.nativeAmerican.isNativeAmerican.indexOf(personId) === -1) {
					delete data.nativeAmerican[key];
				}
			});
		}
 
		if(data.primaryContactData && data.primaryContactData.contact && typeof data.primaryContactData.contact.cellPhoneNumber !== "undefined") {
            // if user previously entered a cell phone number and has since cleared the field, clear the value for isCoverageDetailsRequired
            if (data.primaryContactData.contact.cellPhoneNumber.length === 0 && typeof data.primaryContactData.contact.isCoverageDetailsRequired !== "undefined") {
                delete data.primaryContactData.contact.isCoverageDetailsRequired;
            }
        }
 
		this.setState({ formIsDirty: true, formData: Object.assign(this.state.formData, data) });
	}

	shouldShowRoute(index) {
		// Skip some routes
		// do not show foster care if there are no household members applying for healthcare between 18 -26
		// householdMembers doesn't always get updated in time to show applying for health coverage, but formdata does
		let haveFosterCareEligibleMembers = this.state.householdMembers.some(m => {
				if(parseInt(m.age) >= 18 && parseInt(m.age) <= 26 
				&& typeof this.state.formData.applyingForHealthCoverage !== 'undefined' 
				&& this.state.formData.applyingForHealthCoverage.indexOf(m.personId) >= 0)
				return true;
			}),
			householdSize = this.state.householdMembers.length,
			haveMemberAgedAtLeastTwenty = this.state.householdMembers.some(member => member.age <= 20),
			isSubsidized = ApplicationStore.getDetails().subsidizedApplication,
			path = this.pathOrder[index]; 
 
		if (path === 'primary-caretaker' && (haveMemberAgedAtLeastTwenty === false || householdSize === 1)) {
			return false;
		}
 
		if(['primary-caretaker','foster-care','pregnancy'].indexOf(path) >= 0 && isSubsidized === 'N') {
			return false;
		}
 
		if(path === 'foster-care' && haveFosterCareEligibleMembers === false) {
 
			return false;
		}
 
		if(path === 'household-member-addresses' && householdSize < 2) {
			return false;
		}
 
		if(path === 'confirm-identity' && ApplicationStore.isIdentityVerified() === true) {
			return false;
		}
 
		// skip qle page during open enrollment for new applications
		if(path === 'qualifying-life-event' && ApplicationStore.isNewApplication() === true && ConfigStore.getEnrollmentPeriod() === 'OPN') {
			return false;
		}
		
		return true;
	}

	onBack() {
		if(this.state.mode === 'edit') {
			// don't persist an invalid SSN
			if(typeof this.state.formData.primaryContactData.citizenship !== "undefined" && typeof this.state.formData.primaryContactData.citizenship.ssn !== "undefined") {
