// Import Libs & Components
import React, { Component } from 'react';
import { Link } from 'react-router';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import { ContentIntro, ContentCard, CardNavigation } from '../../Components/Layout/Layout.jsx';
import AppDispatcher from '../../Dispatchers/AppDispatcher.js';
 
class TaxOverview extends Component {
	/**
	 *
	 * @param props
	 */
	constructor(props) {
		super(props);
 
		// Set initial state, modal is not open by default
		this.state = {
		};
	}
 
	componentDidMount() {
		this.dispatcherId = AppDispatcher.register(data => {
			switch (data.action.type) {
				case 'SAVE_EXIT_REQUESTED':
					setTimeout(function(){
						AppDispatcher.dispatch({
							action: {
								type: 'GENERIC_SAVE_EXIT',
								payload: {}
							}
						});
					}, 1);
					break;
			}
		});
	}
 
	componentWillUnmount() {
		AppDispatcher.unregister(this.dispatcherId);
	}
 
	handleBeginClick(e) {
		this.context.router.push('/household/tax-information/primary-filer')
	}
 
	/**
	 * component output
	 * @returns {XML}
	 */
	render() {
 
		return (
			<div className="household-information tax-information">
 
				<ContentIntro className="illustration-text">
					<div className="illustration ic_papersmagglass hidden-xs" />
					<div className="illustration ic_papersmagglass hidden-sm hidden-md hidden-lg" />
					<h1><FormattedMessage id="tax-overview-intro-header" values={{ firstName: this.props.person ? this.props.person.firstName : '' }} /></h1>
					<p><FormattedMessage id="tax-overview-intro-text" /></p>
				</ContentIntro>
 
				<ContentCard>
					<CardNavigation
						text={this.props.intl.messageIfExists('household-build-menu-link')}
						link="/household/menu"
					/>
 
					<FormattedHTMLMessage id="tax-overview-paragraph" tagName="p" />
 
					<br/>
					<br/>
					<br/>
					<br/>
 
					<div className="row menu-actions-row">
						<div className="col-xs-12 col-sm-3 col-sm-push-9 col-md-3 col-md-push-9 col-lg-3 col-lg-push-9">
							<input
								type = "button" 
								id="tax-information-begin"
								value={this.props.intl.messageIfExists('0.1-income-overview-button')}
								className="btn btn-primary btn-block"
								onClick={this.handleBeginClick.bind(this)}
							>
							</input>
						</div>
					</div>
 
				</ContentCard>
 
			</div>
		);
	}
}
 
TaxOverview.propTypes = {
	intl: intlShape.isRequired
};
 
TaxOverview.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(TaxOverview);
