	componentDidMount() {

		this.dispatcherId = AppDispatcher.register(data => {
			switch (data.action.type) {
				case 'SAVE_EXIT_REQUESTED':
					setTimeout(function(){
						AppDispatcher.dispatch({
							action: {
								type: 'GENERIC_SAVE_EXIT',
								payload: {}
							}
						});
					}, 1);
					break;
			}
		});
	}
 
	componentWillUnmount() {
		PeopleStore.removeListener('change', this.binds.updatePersonFromStore);
		ApplicationStore.removeListener('change', this.binds.updateApplicationState);
		AppDispatcher.unregister(this.dispatcherId);

	}

		ApplicationStore.isRACApplication() ? this.setState({deletePersonRacModalIsOpen: false}) : this.setState({deletePersonModalIsOpen: false});
		this.context.router.push('/household/build');
	}

		this.context.router.push('/household/build');
	}
