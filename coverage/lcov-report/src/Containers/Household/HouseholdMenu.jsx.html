// Import Libs & Components
import React, { Component } from 'react';
import AppDispatcher from '../../Dispatchers/AppDispatcher.js';
import { Link } from 'react-router';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import ApplicationStore from '../../Stores/ApplicationStore.js';
import PeopleStore from '../../Stores/PeopleStore.js';
import { ContentIntro, ContentCard, CardNavigation } from '../../Components/Layout/Layout.jsx';
import MenuItem from '../../Components/ApplicationMenu/MenuItem.jsx';
import MemberAvatar from '../../Components/MemberAvatar.jsx';
import ApplicationActions from '../../Actions/ApplicationActions.js';
import DynamicSubsidizedStatusApplicationAction from '../../Actions/DynamicSubsidizedStatusApplicationAction.js';
import { CancelModal, ConfirmModal } from '../../Components/Modal.jsx';
 
class HouseholdMenu extends Component {
	/**
	 *
	 * @param props
	 */
	constructor(props) {
		super(props);
 
		// Set initial state, modal is not open by default
		this.state = {
			householdMenu: this.props.applicationDetails.applicationMenu.householdInfoMenu,
			isConfirmModalOpen: false,
			basicInfoHasError: this.props.applicationDetails.basicInfoHasError,
			taxInfoHasError: this.props.applicationDetails.taxInfoHasError,
			incomeInfoHasError: this.props.applicationDetails.incomeInfoHasError,
			single: false,
			isFoster: ''
 
		};
		this.binds = {
			updateState: this.updateState.bind(this),
			getDynamicQuestions: DynamicSubsidizedStatusApplicationAction.retrieveQuestions.bind(DynamicSubsidizedStatusApplicationAction)
		};
 
	}
	
	updateState() {
		let applicationDetails = ApplicationStore.getDetails();
		
		this.setState({ 
			householdMenu: Object.assign(this.state.householdMenu, applicationDetails.applicationMenu.householdInfoMenu),
			basicInfoHasError: applicationDetails.basicInfoHasError,
			taxInfoHasError: applicationDetails.taxInfoHasError,
			incomeInfoHasError: applicationDetails.incomeInfoHasError,
		});
 
		if (this.props.householdMembers && this.props.householdMembers.length === 1 && typeof this.props.householdMembers[0].fosterCare !== "undefined"){
			this.setState({
				single: true,
				isFoster: this.props.householdMembers[0].fosterCare.isFosterCare
			});
		}
	}
	
	componentDidMount() {
		ApplicationActions.loadApplication();
		ApplicationStore.on('change', this.binds.updateState);
		ApplicationStore.on('change', this.binds.getDynamicQuestions);
		// ensure the race conditions from saving the information from the last page of the flow 
		// (e.g. household information, income) and updating the household menu state.  Listening
		// for PeopleStore changes ensures the household menu updates after the last page of the flow is complete
		PeopleStore.on('change', this.binds.updateState); 
		
		this.dispatchListener = AppDispatcher.register(data => {
			switch (data.action.type) {
				/** update the application status again after an income status update */
				case 'INCOME_STATUS_UPDATE':
				case 'HOUSEHOLD_TAX_DONE':
				case 'HOUSEHOLD_INFO_DONE':
					ApplicationActions.loadApplication();
					break;
				case 'SAVE_EXIT_REQUESTED':
					setTimeout(function(){
						AppDispatcher.dispatch({
							action: {
								type: 'GENERIC_SAVE_EXIT',
								payload: {}
							}
						});
					}, 1);
					break;
				default:
					break;
			}
		});
	}
 
	componentWillUnmount() {
		ApplicationStore.removeListener('change', this.binds.updateState);
		ApplicationStore.removeListener('change', this.binds.getDynamicQuestions);
		PeopleStore.removeListener('change', this.binds.updateState);
		AppDispatcher.unregister(this.dispatchListener);
	}
	
  handleDoneClick(e) {
		this.context.router.push('/application-menu');
 
		// for people added after income entry, in subsidized applications, display confirmation dialog re: adding additional income
		/*if (PeopleStore.isNewPersonAddedWithoutIncomeCheck() && ApplicationStore.getIncomeStatus() === "review" && ApplicationStore.isSubsidized === 'Y') {
			this.setState({
				isConfirmModalOpen: true
			});
		}
		else {
			this.context.router.push('/application-menu');
		}*/
	}
 
	handleCofirmModal() {
		PeopleStore.resetNewPersonAddedWithoutIncomeCheck();
		this.context.router.push('/income/review');
	}
 
	closeConfirmModal() {
		PeopleStore.resetNewPersonAddedWithoutIncomeCheck();
		this.setState({ isConfirmModalOpen: false });
		this.context.router.push('/application-menu');
	}
	/**
	 * component output
	 * @returns {XML}
	 */
	render() {
		var householdMembers = this.props.householdMembers
			? this.props.householdMembers.map((item, i) => {
				return (
					<MemberAvatar key={i} className="household-member" person={item} avatarSize="xs" />
				);
			})
			: [];
		return (
			<div className="household-menu menu-page">
 
				<ContentIntro className="hidden-xs" />
 
				<ContentCard>
					<CardNavigation
						text={this.props.intl.messageIfExists('application-menu')}
						link="/application-menu"
					/>
					<div className="row">
						<div className="col-xs-12">
							<FormattedMessage id="household-menu-header" tagName="h1" />
							<div className="menu-items">
								<MenuItem
									id="household-menu-item-members"
									title={<FormattedMessage id="household-menu-item-members" />}
									state={this.state.householdMenu.householdMembers}
									disabled={this.state.householdMenu.householdMembers === "na"}
									link= "/household/build"
								>
									<div className="household-members">
										{householdMembers}
									</div>
								</MenuItem>
								<MenuItem
									id="household-menu-item-information"
									title={<FormattedMessage id="household-menu-item-information" />}
									state={this.state.householdMenu.basicInfo}
									disabled={this.state.householdMenu.basicInfo === "na"}
									error={this.state.basicInfoHasError === true ? <FormattedMessage id="household-menu-update-warning" /> : ''}
									link={this.state.householdMenu.basicInfo === "review" ? "/household/information/review" : "/household/begin"}
								/>
								{(this.state.isFoster === 'Y' && this.state.single) || ApplicationStore.getDetails().subsidizedApplication !== 'Y' ? '' :
                                <div>
								<MenuItem
									id="household-menu-item-tax-information"
									title={<FormattedMessage id="household-menu-item-tax-information" />}
									state={this.state.householdMenu.taxInfo}
									error={this.state.taxInfoHasError === true ? <FormattedMessage id="household-menu-update-warning" /> : ''}
									link={this.state.householdMenu.taxInfo === "review" ? "/household/tax-review": "/household/tax-overview"}
									disabled={this.state.householdMenu.taxInfo === "na"}
								/>
								<MenuItem
									id="household-menu-item-income"
									title={<FormattedMessage id="household-menu-item-income" />}
									state={this.state.householdMenu.incomeInfo}
									error={this.state.incomeInfoHasError === true ? <FormattedMessage id="household-menu-update-warning" /> : ''}
									link={this.state.householdMenu.incomeInfo === "review" ? "/income/review" : "/income"}
									disabled={this.state.householdMenu.incomeInfo === 'na'}
								/>
								</div>
                                }
 
							</div>
						</div>
					</div>
					<div className="row menu-actions-row">
						<div className="col-xs-12 col-sm-6 col-sm-push-6 col-md-3 col-md-push-9 col-lg-3 col-lg-push-9">
							<button 
								id="household-menu-done"
								className="btn btn-primary btn-block" 
								type="submit" 
								onClick={this.handleDoneClick.bind(this)}
								disabled={this.props.applicationDetails.applicationMenu.houseHoldInfoDone !== true}
							>
								<FormattedMessage id="household-menu-button-done" />
							</button>
						</div>
					</div>
				</ContentCard>
				<ConfirmModal
					className="address-confirmation-modal"
					isOpen={this.state.isConfirmModalOpen}
					contentLabel={this.props.intl.messageIfExists('household-add-confirm-modal-header')}
					shouldCloseOnOverlayClick={false}
					fullScreenMobile={true}
					onRequestConfirm={this.handleCofirmModal.bind(this)}
					onRequestClose={this.closeConfirmModal.bind(this)}
					confirmText={this.props.intl.messageIfExists('household-add-confirm-modal-confirm-text')}
					cancelText={this.props.intl.messageIfExists('household-add-confirm-modal-cancel-text')}
				>
					<div className="text-center">
						<h3><FormattedMessage
							id="household-add-confirm-modal-header"/>
						</h3>
					</div>
					<br />
					<span>
						<FormattedMessage id={"household-add-confirm-modal-paragraph"}/>
					</span>
					<br />
				</ConfirmModal>
			</div>
		);
	}
}
 
HouseholdMenu.propTypes = {
	intl: intlShape.isRequired
};
 
HouseholdMenu.contextTypes = {
    router: React.PropTypes.object.isRequired
};
 
export default injectIntl(HouseholdMenu);
