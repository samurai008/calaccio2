export const isInAgeRange = (min, max, dob, current = false) => {
	const now = moment.isMoment(current) ? current : moment();
	const start = now.clone().subtract(max, 'year');
	const end = now.clone().subtract(min, 'year');
	return dob.isBetween(start, end, 'day', '[]');
}
