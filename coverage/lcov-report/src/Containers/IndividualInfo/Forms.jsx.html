						expirationDate: ApplicationStore.getExpirationDate()
						}}  
					/>
				</ConfirmModal>
 
				<NoticeModal
					isOpen={this.state.confirmValidSaveAndExitModalIsOpen === true}
					onRequestClose={this.handleConfirmSaveAndExitModal.bind(this)}
					shouldCloseOnOverlayClick={false}
					fullScreenMobile={true}
					title={<FormattedMessage id="global-element-confirm-save-and-exit-modal-header" />}
					contentLabel={this.props.intl.formatMessage({ id: 'global-element-confirm-save-and-exit-modal-header' })}
					confirmText={<FormattedMessage id="sr-close"/>}
				>
					<FormattedHTMLMessage id="global-element-confirm-save-and-exit-modal-body" 
					values={{
						expirationDate: ApplicationStore.getExpirationDate()
						}}  
					/>
				</NoticeModal>
			</div>
		);
	}
