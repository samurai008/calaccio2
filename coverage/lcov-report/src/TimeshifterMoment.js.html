import originalUnmodifiedMoment from 'moment';
import ConfigStore from './Stores/ConfigStore.js';
/**
 * Timeshifter-aware wrapper around moment.js
 * 
 * This is a wrapper for the moment.js library which provides support for the CalHEERS
 * TimeShifter functionality. If an offset is set in ConfigStore, this shifted date/time
 * is applied to the returned instance of moment().
 */
var originalMoment = originalUnmodifiedMoment;
 
var moment = function() {
	var result;
	var timeshifterConfig = ConfigStore.get("timeshifter");
	
	// SIR 100636: if arguments were passed, use these instead of Timeshifter configuration. For example, if
	// you call moment('01/01/2017', 'L') the intent is to create a moment instance with these values
	// and ignore any Timeshifter config, whether set or not
	if(arguments.length > 1) {
		result = originalMoment.apply(originalMoment, arguments);
	}
	//if Timeshifter offset enabled, use the currently set offset date
	else if(timeshifterConfig !== null && timeshifterConfig.timeshifterEnabledForUser == 'true'){
		//console.log("TimeshifterMoment called with timeshifter offset enabled");
		result = new originalMoment(timeshifterConfig.shiftedDateMillisFromEpoch);    
	}
	//else call the original anyway
	else {
		//console.log("TimeshifterMoment called without args or Timeshifter enabled");
		result = originalMoment.apply(originalMoment, arguments);
	}
 
	return result;
};
 
//copy all other functions from original moment to the new timeshifterMoment:
//prototype includes functions that are callable on a moment() instance
moment.prototype             = originalMoment.prototype;

moment.fn                    = originalMoment.fn;
moment.min                   = originalMoment.min;

moment.duration              = originalMoment.duration;
moment.isMoment              = originalMoment.isMoment;
moment.weekdays              = originalMoment.weekdays;
moment.parseZone             = originalMoment.parseZone;
moment.localeData            = originalMoment.localeData;
moment.isDuration            = originalMoment.isDuration;
moment.monthsShort           = originalMoment.monthsShort;
moment.weekdaysMin           = originalMoment.weekdaysMin;
moment.defineLocale          = originalMoment.defineLocale;

//replace isMoment with check for TimeShifterMoment (moment) and original moment (originalMoment)
moment.isMoment = function(obj){
	return obj instanceof moment || obj instanceof originalMoment || (obj != null && obj._isAMomentObject != null);
}
