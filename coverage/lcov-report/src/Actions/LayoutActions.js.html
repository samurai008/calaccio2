import AppDispatcher from '../Dispatchers/AppDispatcher';
 
var LayoutActions = {
	
	toggleMobileNav: function() {
		this.dispatch('MOBILE_NAV_TOGGLE');
	},
 
	toggleFAB: function() {
		this.dispatch('TOGGLE_FAB');
	},
 
	toggleLocaleMenu: function() {
		this.dispatch('TOGGLE_LOCALE_MENU');
	},
 
	openProgressModal: function() {
		this.dispatch('OPEN_PROGRESS_MODAL');
	},
 
	closeProgressModal: function() {
		this.dispatch('CLOSE_PROGRESS_MODAL');
	},
 
	dispatch: function(actionType, payload) {
		// Dispatch the event, the appropriate Stores will do the rest.
		AppDispatcher.dispatch({
			action: {
				type: actionType,
				payload: payload
			}
		});
	}
 
};
 
export default LayoutActions;
