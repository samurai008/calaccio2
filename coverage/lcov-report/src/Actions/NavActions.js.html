import AppDispatcher from '../Dispatchers/AppDispatcher';
 
var NavActions = {
	completeSection: function(section) {
		this.dispatch('NAV_SECTION_COMPLETE', section);
	},
	completeNavItem: function(section, navItem) {
		this.dispatch('NAV_ITEM_COMPLETE', { section: section, item: navItem });
	},
	activateNavItemByRoute: function(nextState) {
		this.dispatch('NAV_ITEM_ACTIVATE_ROUTE', nextState.location.pathname);
	},
	changeSectionTitle: function(section, newTitle) {
		this.dispatch('NAV_SECTION_CHANGE_TITLE',{ section: section, title: newTitle });
	},
	enableSection: function(section) {
		this.dispatch('NAV_SECTION_LINK_ENABLE', section);
	},
	enableAndCompleteSection: function(section) {
		this.dispatch('NAV_SECTION_COMPLETE_LINK_ENABLE', section);
	},
	changeLocale: function(locale) {
		this.dispatch('LOCALE_CHANGE', locale);
	},
	saveAndExit: function(payload) {
		this.dispatch('SAVE_EXIT_REQUESTED', payload);
	},
	redirectToPersonalData: function() {
		this.dispatch('REDIRECT_TO_PORTAL', 'personal-data');
	},
	redirectToEligibility: function() {
		this.dispatch('REDIRECT_TO_PORTAL', 'eligibility');
	},
	logOutUser: function() {
		this.dispatch('LOG_OUT_USER');
	},
	completeIncome: function() {
		this.completeSection('income');
		this.redirectToEligibility();
	},
	dispatch: function(actionType, data) {
		// Dispatch the event, the appropriate Stores will do the rest.
		AppDispatcher.dispatch({
			action: {
				type: actionType,
				payload: data
			}
		});
	}
 
};
 
export default NavActions;
