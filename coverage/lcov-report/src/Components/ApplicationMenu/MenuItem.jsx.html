import React from 'react';
import { FormattedMessage } from 'react-intl';
 
const MenuItem = (props, context) => {
	var className = ['menu-item'].concat(
			props.state,
			props.disabled ? 'disabled' : [],
			props.className ? props.className.split(' ') : []
		),
		buttonClassName = ['btn', 'btn-block'].concat(
			props.disabled ? ['disabled', 'hidden-xs'] : []
		),
		titleColumnClassName = ['col-md-9'].concat(
			['review', 'edit'].indexOf(props.state) >= 0 ? 'col-xs-8' : 'col-xs-12'
		),
		contentColumnClassName = ['col-xs-12'],
		actionColumnClassName = ['action', 'col-md-3'].concat(
			['review', 'edit'].indexOf(props.state) >= 0 ? 'col-xs-4' : 'col-xs-12'
		),
		handleActionClick = e => {
			e.preventDefault();
			Eif(props.link) {
				context.router.push(props.link);
			}
			props.onRequestAction();
		};
 
	switch(props.state) {
		case 'start':
		case 'continue':
		case 'add':
		case 'addincome':
		case 'update':
		case 'na':
			buttonClassName.push('btn-primary');
			break;
		case 'review': 
		case 'edit':
			buttonClassName.push('btn-link');
			break;
			buttonClassName.push('btn-primary');
			break;
	}
	// default na to start string
	let buttonState = props.state === 'na' ? 'start' : props.state;
	// override for income
	let buttonMessageString = 'menu-button-' + (props.id.indexOf('income') >= 0 && buttonState === 'start' ?  'addincome' : buttonState);
	
	return  (
		<div className={className.join(' ')}>
			<div className="row">
				<div className={titleColumnClassName.join(' ')}>
					<div className="title">{props.title}</div>
					{props.error ? <p className="text-danger"><span className="ic_error" /> {props.error}</p> : null}
				</div>
				<div className={actionColumnClassName.join(' ')}>
					<a id={props.id + '-' + props.state} href="#" role="button" className={buttonClassName.join(' ')} onClick={handleActionClick} onKeyDown={() => {}}>
						<FormattedMessage id={buttonMessageString} />
					</a>
				</div>
			</div>
			<div className="row">
				<div className={contentColumnClassName.join(' ')}>
					{props.children}
				</div>
			</div>
		</div>
	);
};
 
MenuItem.propTypes = {
	title: React.PropTypes.oneOfType([
		React.PropTypes.string.isRequired,
		React.PropTypes.element
	]),
	error: React.PropTypes.oneOfType([
		React.PropTypes.string.isRequired,
		React.PropTypes.element
	]),
	state: React.PropTypes.oneOf(['start', 'continue', 'review', 'update', 'add','addincome','edit', 'na']).isRequired,
	className: React.PropTypes.string,
	disabled: React.PropTypes.bool,
	onRequestAction: React.PropTypes.func,
	link: React.PropTypes.string
};
 
MenuItem.defaultProps = {
	disabled: false,
	onRequestAction: () => {}
};
 
MenuItem.contextTypes = {
	router: React.PropTypes.object.isRequired
};
 
export default MenuItem;
