// Import Libs & Components
import React, { Component } from 'react';
import NavSection from './NavSection.jsx';
 
class NavSectionsList extends Component {
 
	constructor(props) {
		super(props);
		// Set initial state
		this.state = {};
	}
 
	render() {
		var sections = this.props.sections.map((section,idx) => { return ( <NavSection section={section} key={idx} idx={idx} />) });
		return (
			<nav role="navigation" className="panel-group" aria-multiselectable="true">
				{sections}
			</nav>
		);
	}
}
 
NavSectionsList.propTypes = {
	sections: React.PropTypes.array.isRequired
};
 
export default NavSectionsList;
