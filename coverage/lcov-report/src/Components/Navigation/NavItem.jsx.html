// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage } from 'react-intl';
import { Link } from 'react-router';
import { navItemShape } from '../../types.js';
 
class NavItem extends Component {
 
	constructor(props) {
		super(props);
		// Set initial state
		this.state = {};
	}
 
	render() {
		var active = (this.props.item.active) ? 'active' : '',
			last = this.props.isLast ? ' last' : '',
			classNames = 'list-group-item ' + active,
			linkTitle = this.props.intl.formatMessage({ id: 'navigation-' + this.props.section + '-sub-item-' + this.props.item.title }),
			completed = (this.props.item.completed) ? <span className="nav-icon icon-sub-complete"/> : '',
			ariaLabel = linkTitle + (completed ? ' ' + this.props.intl.formatMessage({ id: 'navigation-item-complete' }) : ''),
			link = (active || !this.props.item.completed)  ? linkTitle : <Link to={'/' + this.props.section + '/' + this.props.item.title}>{linkTitle}</Link>;
		return (
			<li className={classNames} aria-label={ariaLabel}>
				<span className={'icon icon-subnav-progress ' + active + last}/>
				{link}
				{completed}
			</li>
		)
	}
}
 
NavItem.propTypes = {
	item: navItemShape.isRequired,
	isLast: React.PropTypes.bool,
	section: React.PropTypes.string,
	intl: intlShape
};
 
NavItem.defaultProps = {
	isLast: false,
	section: ''
};
 
export default injectIntl(NavItem);
