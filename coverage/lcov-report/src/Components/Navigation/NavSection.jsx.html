// Import Libs & Components
import React, { Component } from 'react';
import NavItemsList from './NavItemsList.jsx';
import { intlShape, injectIntl, FormattedMessage } from 'react-intl';
import ConfigStore from '../../Stores/ConfigStore';
 
class NavSection extends Component {
 
	constructor(props) {
		super(props);
		// Set initial state
		this.state = {};
	}
 
	render() {
		var portalURL = ConfigStore.get('portalURL'),
			portalURLResolved = portalURL !== null ? portalURL.replace('{appURL}', window.location.origin) : '',
			sectionExpanded = (this.props.section.active && this.props.section.items.length > 0).toString(),
			sectionCompletedSpan = (this.props.section.completed) ? <span className="nav-icon icon-complete"/> : '',
			/* build the string id */
			pageURL = ConfigStore.get('navigation-' + this.props.section.title + '-url'),
			sectionLinkHref =  (portalURL && pageURL !== null) ? pageURL.replace('{portalURL}', portalURLResolved) : null,
			/* ensure section is completed, and resolved href doesn't just contain the ID (i.e. not resolved) */
			sectionLinkResolved = (this.props.section.linkEnabled && sectionLinkHref !== null) ? sectionLinkHref : '#',
			linkTitle = this.props.intl.formatMessage({ id: 'navigation-' + this.props.section.title }),
			ariaLabel = linkTitle + (this.props.section.completed ? ' ' + this.props.intl.formatMessage({ id: 'navigation-item-complete' }) : '');
		return (
			<div className={'panel panel-' + ((this.props.section.active) ? 'active' : 'default')}>
				<div className="panel-heading"  id={'collapseListGroupHeading' + this.props.idx}>
					<h4 className="panel-title">
						<a
							href={sectionLinkResolved}
							className=""
							data-toggle="collapse"
							aria-expanded={sectionExpanded}
							aria-controls={'collapseListGroup' + this.props.idx}
							aria-label={ariaLabel}
						>
							<span className={'nav-icon icon-' + this.props.section.title}/>
							<FormattedMessage id={'navigation-' + this.props.section.title}/>
							{sectionCompletedSpan}
						</a>
					</h4>
				</div>
				<NavItemsList items={this.props.section.items} expanded={sectionExpanded} listGroupId={this.props.idx} section={this.props.section.title} />
			</div>
		)
	}
}
 
NavSection.propTypes = {
	section: React.PropTypes.object.isRequired,
	idx: React.PropTypes.number.isRequired,
	intl: intlShape.isRequired
};
 
export default injectIntl(NavSection);
