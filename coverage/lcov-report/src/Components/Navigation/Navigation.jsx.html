// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl } from 'react-intl';
import NavSectionsStore from '../../Stores/NavSectionsStore.js';
import LayoutActions from '../../Actions/LayoutActions';
import NavSectionsList from './NavSectionsList.jsx';
 
class Navigation extends Component {
 
	constructor(props) {
		super(props);
		// Set initial state
		this.state = {
			navSections: NavSectionsStore.getAllSections()
		};
		this.binds = {
			onNavSectionsStoreChange: this.handleNavChange.bind(this)
		};
	}
 
	handleOverlayClick() {
		LayoutActions.toggleMobileNav();
	}
	handleNavChange() {
		this.setState({navSections: NavSectionsStore.getAllSections()});
	}
	componentDidMount() {
		NavSectionsStore.on('change', this.binds.onNavSectionsStoreChange);
	}
	componentWillUnmount() {
		NavSectionsStore.removeListener('change', this.binds.onNavSectionsStoreChange);
	}
	render() {
		return (
			<div>
				<div
					tabIndex="0"
					role="dialog"
					aria-label={this.props.intl.formatMessage({ id: 'sr-close-navigation' })}
					className="mobile-nav-overlay hidden-sm hidden-md hidden-lg"
					onClick={this.handleOverlayClick.bind(this)}
				/>
				<div className="navigation">
					<div className="navigation-header"></div>
					<NavSectionsList sections={this.state.navSections} />
				</div>
			</div>
		);
	}
}
 
Navigation.propTypes = {
	intl: intlShape.isRequired
};
 
export default injectIntl(Navigation);
