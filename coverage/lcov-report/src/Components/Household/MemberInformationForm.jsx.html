class MemberRelationshipsFormBase extends Component {
 
	constructor(props) {
		super(props);
		this.state = {
			activeRow: props.currentRelationshipPerson ? this.findActiveRow(props.currentRelationshipPerson) : null
		};
	}
 
	handleFieldChange(name, value) {
		this.props.handleFieldChange(name, value);
	}
 
	handleAccordionToggle(row) {
		this.setState({ activeRow: row });
	}
 
	findActiveRow(currentRelationshipPerson) {
		var personId = currentRelationshipPerson.personId,
			activeRow = null;
		relationships
			.filter(item => {
				return typeof item.test !== 'function' || item.test(currentRelationshipPerson, this.props.formData);
			})
			.forEach((item, r) => {
				Object.keys(item.options).forEach((key, i) => {
					if (this.props.getFormValue('relationTypeCd[' + personId + ']') === key) {
						activeRow = r;
					}
				});
			});
 
		return activeRow;
	}
 
	componentWillReceiveProps(nextProps) {
		if (
			nextProps.currentRelationshipPerson
			&& (!this.props.currentRelationshipPerson
				|| (
					Object.keys(nextProps.currentRelationshipPerson).length !== Object.keys(this.props.currentRelationshipPerson).length
					|| nextProps.currentRelationshipPerson.personId !== this.props.currentRelationshipPerson.personId
				)
			)
		) {
			setTimeout(() => { this.setState({ activeRow: this.findActiveRow(nextProps.currentRelationshipPerson) }) });
		}
	}
 
	render() {
		var radioName = 'relationTypeCd[' + (
			this.props.currentRelationshipPerson
				? this.props.currentRelationshipPerson.personId
				: ''
		) + ']',
			panels = this.props.currentRelationshipPerson
				? relationships
					.filter(item => {
						return typeof item.test !== 'function' || item.test(this.props.currentRelationshipPerson, this.props.formData);
					})
					.map((item, i) => {
						var options = {};
 
						Object.keys(item.options).forEach((key, i) => {
							options[key] = this.props.intl.messageIfExists(item.options[key]);
						});
 
						return (
							<Panel key={i} control={this.props.intl.messageIfExists(item.category)}>
								<RadioInput
									id="relationship"
									name={radioName}
									onChange={this.handleFieldChange.bind(this)}
									value={this.props.getFormValue(radioName)}
									options={options}
									inline={false}
									required
								/>
							</Panel>
						);
					})
				: [],
			selectedValue = this.props.getFormValue(radioName),
			selectedDisplayKey = selectedValue
				? relationships.reduce((prev, next) => {
					return typeof next.options[selectedValue] === 'string' ? next.options[selectedValue] : prev
				}, '')
				: 'household-add-member-relationship-header-no-value',
			selectedDisplayValue = this.props.intl.messageIfExists(selectedDisplayKey);
 
		return (
			<FormColumn className="relationship-info-form" columnClassName="col-md-12">
				<h4 className={selectedDisplayKey}>
					<FormattedHTMLMessage
						id="household-add-member-relationship-header"
						values={{
							relationshipFirstName: this.props.currentRelationshipPerson ? this.props.currentRelationshipPerson.firstName : '',
							firstName: this.props.getFormValue('firstName'),
							selectedValue: selectedDisplayValue
						}}
					/>
				</h4>
				<Accordion
					activeRow={this.state.activeRow}
					onToggle={this.handleAccordionToggle.bind(this)}
				>
					{panels}
				</Accordion>
			</FormColumn>
		);
	}
}
