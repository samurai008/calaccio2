const handleNobodyOption = (value) => {
	if (value[value.length - 1] === 'nobody') {
		value = ['nobody'];
	}
	else if (value.indexOf('nobody') >= 0) {
		let removedVal = value.splice(value.indexOf('nobody'), 1);
	}
 
	return value;
};

export const PlanToFileForm = injectIntl(FormHOC((props) => {

			props.handleFieldChange(name, handleNobodyOption(value));
		},
		householdMemberSegmentedControlAdditionalOptions = {
			nobody: props.intl.messageIfExists('tax-information-plan-to-file-option-nobody')
		},

			? props.getFormValue('filingTaxes').map((personId, i) => {
