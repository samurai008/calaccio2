export const HouseholdMemberAddressForm = injectIntl(FormHOC((props) => {
	let allPhysicalAddressesAreValid = false,
		allMailingAddressesAreValid = true,
		validPhysicalAddresses = [],
		validMailingAddresses = [];
	/* keep track of ids of persons with valid physical addresses. if the number matches the number of persons selected, valid */
	const onPhysicalAddressValidate = (personId, isValid) => {
		if (isValid && validPhysicalAddresses.indexOf(personId) === -1) validPhysicalAddresses.push(personId);
		if (!isValid && validPhysicalAddresses.indexOf(personId) !== -1) validPhysicalAddresses.splice(validPhysicalAddresses.indexOf(personId), 1);
 
		allPhysicalAddressesAreValid =
			props.getFormValue('selectedPersonsNotLivingWithPC').length === validPhysicalAddresses.length;
 
		validateForm();
	};
	/* keep track of ids of persons with valid mailing addresses. if the number matches the number of persons selected -that have entered mailing address-, valid */
	const onMailingAddressValidate = (personId, isValid) => {
		let persons = props.getFormValue('selectedPersonsNotLivingWithPC');
 
		// assume people have valid mailing addresses
		if (isValid && validMailingAddresses.indexOf(personId) === -1) validMailingAddresses.push(personId);
		if (isValid === false && validMailingAddresses.indexOf(personId) !== -1) validMailingAddresses.splice(validMailingAddresses.indexOf(personId), 1);
 
		allMailingAddressesAreValid = persons.length === validMailingAddresses.length
			|| props.getFormValue('selectedPersonsNotLivingWithPC').filter(id => {
				// return only people not living with PC with a mailing address
				return props.getFormValue('memberMailingAddresses[' + id + ']', null) !== null;
			}).length === validMailingAddresses.length;
 
		validateForm();
	};
	/* send parent container aggregate validation state on changes to that state */
	const validateForm = () => {
		if(allPhysicalAddressesAreValid && allMailingAddressesAreValid){
			props.onInternalFormValid(true);
		}
	};
	let householdMembers = props.householdMembers.filter((m) => {
			return m.personId !== props.getFormValue('primaryContactId');
		}),
		forms = Array.isArray(props.getFormValue('selectedPersonsNotLivingWithPC'))
			? props.getFormValue('selectedPersonsNotLivingWithPC').map((personId, i) => {
				let addressFieldName = 'memberPhysicalAddresses[' + personId + ']',
					radioFieldName = 'memberAddressSameAsMailing[' + personId + ']',
					mailingAddressFieldName = 'memberMailingAddresses[' + personId + ']',
					person = PeopleStore.getPersonById(personId);
 
				return (
					<div key={i}>
						<FormattedMessage id="household-primary-contact-member-addresses-form-header" tagName="h3" values={person} />
						<AddressInput
							name={addressFieldName}
							id={addressFieldName}
							value={props.getFormValue(addressFieldName, {})}
							onChange={props.handleFieldChange}
							intl={props.intl}
							onValidate={onPhysicalAddressValidate.bind(null, personId)}
							onBlur={validateForm}
							required
						/>
						<RadioInput
							label={props.intl.messageIfExists('household-primary-contact-member-addresses-form-address-same-as-mailing-label', person)}
							id={radioFieldName}
							name={radioFieldName}
							onChange={props.handleFieldChange}
							value={props.getFormValue(radioFieldName)}
							options={{ Y: props.intl.messageIfExists('Yes'), N: props.intl.messageIfExists('No') }}
							required
						/>
						<ConditionalField condition={props.getFormValue(radioFieldName) === 'N'}>
							<AddressInput
								label={props.intl.messageIfExists('household-primary-contact-member-addresses-form-mailing-address-label', person)}
								aria-label={props.intl.messageIfExists('household-primary-contact-member-addresses-form-mailing-address-label', person)}
								name={mailingAddressFieldName}
								id={mailingAddressFieldName}
								value={props.getFormValue(mailingAddressFieldName, {})}
								onChange={props.handleFieldChange}
								intl={props.intl}
								onValidate={onMailingAddressValidate.bind(null, personId)}
								onBlur={validateForm}
								required={props.getFormValue(radioFieldName) === 'N'}
							/>
						</ConditionalField>
					</div>
				);
			})
			: null;
 
	return (
		<FormColumn columnClassName="col-sm-12 col-md-12">
			<h3 className="household-addresses-headersplit-split">
			<FormattedMessage id="household-primary-contact-member-addresses-header" />
			<FormattedMessage id="household-primary-contact-member-addresses-header-1" />
			<FormattedMessage id="household-primary-contact-member-addresses-header-2" values={props.primaryContact || { firstName: '' }}/>
			</h3>
			<br />
			<HouseholdMemberSegmentedControl
				members={householdMembers}
				onChange={props.handleFieldChange}
				value={props.getFormValue('selectedPersonsNotLivingWithPC')}
				name="selectedPersonsNotLivingWithPC"
				multiSelect={true}
			/>
			<br />
			<br />
			<div className="row">
				<div className="col-md-8">
					{forms}
				</div>
