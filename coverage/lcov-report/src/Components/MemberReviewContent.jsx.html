import React from 'react';
import {
	intlShape,
	injectIntl
} from 'react-intl';
import { Link } from 'react-router';
import PeopleStore from '../Stores/PeopleStore';
import { ReviewSection, ReviewRow } from './ReviewSection.jsx';
import MemberAvatar from './MemberAvatar.jsx';
import FormattedPhone from './FormattedPhone.jsx';
import ApplicationStore from '../Stores/ApplicationStore.js';
import ConditionalField from '../Components/ConditionalField.jsx';
 
const MemberReviewContent = props => {
	var person = props.person,
		/* set the already defined relationships for a person */
		definedRelationships = (person.relationships && Array.isArray(person.relationships) && person.relationships.length > 0) ? person.relationships : [],
		/* generate a set of relationships to display for the person, filling in any missing relationships with NA so it can be added */
		displayedRelationships = props.otherHouseholdMembers.map(otherPerson => {
			// default relationship to display
			let relationship = {
				relatedPersonId: otherPerson.personId,
				relationTypeCd: 'NA'
			};
			// override with real relationship if found
			if(definedRelationships.length > 0) {
				let foundRelationship = definedRelationships.filter(r => r.relatedPersonId === otherPerson.personId);
				if(foundRelationship.length === 1) {
					relationship = foundRelationship[0];
				}
			}
			return relationship;
		});
	return (
		<div className="member-review-content">
			<MemberAvatar person={person} avatarSize="md" />
			<br/>
			<ReviewSection header={props.intl.messageIfExists('member-review-header-basic-information')}>
				<ReviewRow
					id={props.id + '-basic-info'}
					linkUrl={props.editUrlBase.replace('{personId}', person.personId) + 'basic-info'}
					items={[
						{
							label: props.intl.messageIfExists('household-review--person-firstName--label'),
							value: person.firstName
						},
						{
							label: props.intl.messageIfExists('household-review--person-middleName--label'),
							value: person.middleName || props.intl.messageIfExists('household-review--generic--value-notApplicable')
						},
						{
							label: props.intl.messageIfExists('household-review--person-lastName--label'),
							value: person.lastName
						},
						{
							label: props.intl.messageIfExists('household-review--person-suffix--label'),
							value: person.suffix
								? props.intl.messageIfExists('household-review--person-suffix--value-' + person.suffix)
								: props.intl.messageIfExists('household-review--generic--value-notApplicable')
						},
						{
							label: props.intl.messageIfExists('household-review--person-dob--label'),
							value: person.dob && typeof person.dob === 'object' ? person.dob.format('L') : ''
						}
					]}
				/>
			<ConditionalField condition={ ApplicationStore.getDetails().subsidizedApplication  === 'Y' && person.age >= 18 && person.age < 21 }>
				<ReviewRow
					id={props.id + '-basic-info2'}
					linkUrl={props.editUrlBase.replace('{personId}', person.personId) + 'basic-info2'}
					items={[
						{
							label: props.intl.messageIfExists('household-review--person-schoolAttendingStatus--label'),
							value: person.schoolAttendingStatus
										? props.intl.messageIfExists( person.schoolAttendingStatus && person.schoolAttendingStatus === 'Y' ? 'yes' : 'no')
										: null
						}
					]}
				/>
				</ConditionalField>
 
				<ReviewRow
					id={props.id + '-basic-info3'}
					linkUrl={props.editUrlBase.replace('{personId}', person.personId) + 'basic-info3'}
					items={[
						{
							label: props.intl.messageIfExists('household-review--person-genderCd--label'),
							value: person.genderCd
										? props.intl.messageIfExists('household-review--person-genderCd--value-' + person.genderCd)
										: null
						}
					]}
				/>
 
				<ReviewRow
					id={props.id + '-basic-info4'}
					linkUrl={props.editUrlBase.replace('{personId}', person.personId) + 'basic-info4'}
					items={[
						{
							label: props.intl.messageIfExists('household-review--person-maritalStatus--label'),
							value: person.maritalStatus
										? props.intl.messageIfExists('household-review--person-maritalStatus--value-' + person.maritalStatus)
										: null
						}
					]}
				/>
			</ReviewSection>
 
			<ReviewSection header={props.intl.messageIfExists('member-review-header-contact-information')}>
				<ReviewRow
					id={props.id + '-basic-info5'}
					linkUrl={props.editUrlBase.replace('{personId}', person.personId) + 'basic-info5'}
					items={[
						{
							label: props.intl.messageIfExists('household-review--person-contact-homePhoneNumber-label'),
							value:  person.contact && person.contact.homePhoneNumber
										? React.createElement(FormattedPhone, { value: person.contact.homePhoneNumber })
										: props.intl.messageIfExists('household-review--generic--value-notApplicable')
						},
						{
							label: props.intl.messageIfExists('household-review--person-contact-cellPhoneNumber-label'),
							value: person.contact && person.contact.cellPhoneNumber
										? React.createElement(FormattedPhone, { value: person.contact.cellPhoneNumber })
										: props.intl.messageIfExists('household-review--generic--value-notApplicable')
						},
						{
							label: props.intl.messageIfExists('household-review--person-contact-workPhoneNumber-label'),
							value: person.contact && person.contact.workPhoneNumber
										? React.createElement(FormattedPhone, { value: person.contact.workPhoneNumber })
										: props.intl.messageIfExists('household-review--generic--value-notApplicable')
						},
						{
							label: props.intl.messageIfExists('household-review--person-contact-extension-label'),
							value: person.contact && person.contact.extension ? person.contact.extension : props.intl.messageIfExists('household-review--generic--value-notApplicable')
						},
						{
							label: props.intl.messageIfExists('household-review--person-contact-email-label'),
							value: person.contact && person.contact.email ? person.contact.email : props.intl.messageIfExists('household-review--generic--value-notApplicable')
						}
					]}
				/>
 
				<ReviewRow
					id={props.id + '-basic-info6'}
					linkUrl={props.editUrlBase.replace('{personId}', person.personId) + 'basic-info6'}
					items={[
						{
							label: props.intl.messageIfExists('household-review--person-contact-preferWrittenLang--label'),
							value: person.contact && person.contact.preferWrittenLang
										? props.intl.messageIfExists('household-review--person-contact-preferWrittenLang--value-' + person.contact.preferWrittenLang)
										: props.intl.messageIfExists('household-review--generic--value-notApplicable')
						},
						{
							label: props.intl.messageIfExists('household-review--person-contact-preferSpokenLang--label'),
							value: person.contact && person.contact.preferSpokenLang
										? props.intl.messageIfExists('household-review--person-contact-preferSpokenLang--value-' + person.contact.preferSpokenLang)
										: props.intl.messageIfExists('household-review--generic--value-notApplicable')
						}
					]}
				/>
 
			</ReviewSection>
 
			{
				props.otherHouseholdMembers.length > 0
					? (
						<ReviewSection header={props.intl.messageIfExists('member-review-header-relationships-information')}>
							{
								displayedRelationships.map((relationship, i) => {
										var otherPerson = PeopleStore.getPersonById(relationship.relatedPersonId),
											showAges = person.firstName === otherPerson.firstName;
 
										return (
											<ReviewRow
												id={props.id + '-relationship-info-' + i}
												key={i}
												linkUrl={props.editUrlBase.replace('{personId}', person.personId) + 'relationship-info/' + relationship.relatedPersonId}
												items={[
													{
														label: props.intl.messageIfExists('application-review-relationship-label', {
															firstName: person.firstName + (showAges ? ' (' + person.age + ')' : ''),
															relationshipFirstName: otherPerson.firstName + (showAges ? ' (' + otherPerson.age + ')' : ''),
														}),
														value: props.intl.messageIfExists('relationship-value-' + relationship.relationTypeCd)
													}
												]}
											/>
										);
									})
							}
						</ReviewSection>
					)
					: null
			}
 
			<ReviewSection header={props.intl.messageIfExists('member-review-header-demographic-information')}>
 
				<ReviewRow
					id={props.id + '-optional-questions'}
					linkUrl={props.editUrlBase.replace('{personId}', person.personId) + 'optional-questions'}
					items={[
						{
							label: props.intl.messageIfExists('household-review--person-personal-isHispanicOrigin--label'),
							value: person.personal && person.personal.isHispanicOrigin === 'Y' 
										? props.intl.messageIfExists('yes')
										: (person.personal && person.personal.isHispanicOrigin === 'N' ? props.intl.messageIfExists('no')
										:props.intl.messageIfExists('household-review--generic--value-notApplicable'))
						}
					].concat(
						person.personal && person.personal.isHispanicOrigin === 'Y'
							? {
								label: props.intl.messageIfExists('household-review--person-personal-ethnicity--label'),
								value: person.personal && person.personal.ethnicity
											? person.personal.ethnicity.map(code => props.intl.messageIfExists('household-review--person-personal-ethnicity--value-' + code)).join(', ')
											: props.intl.messageIfExists('household-review--generic--value-notApplicable')
							} : [],
						person.personal && person.personal.race
							? {
								label: props.intl.messageIfExists('household-review--person-personal-race--label'),
								value: person.personal && person.personal.race && (person.personal.race.length > 0)
											? person.personal.race.map(code => props.intl.messageIfExists('household-review--person-personal-race--value-' + code)).join(', ')
											: props.intl.messageIfExists('household-review--generic--value-notApplicable')
							} : []
					)}
				/>
 
			</ReviewSection>
		</div>
	);
};
 
MemberReviewContent.propTypes = {
	person: React.PropTypes.object,
	otherHouseholdMembers: React.PropTypes.array,
	intl: intlShape.isRequired,
	editUrlBase: React.PropTypes.string
};
 
MemberReviewContent.defaultProps = {
	person: {},
	otherHouseholdMembers: [],
	editUrlBase: '/household/edit/{personId}/'
};
 
 
export default injectIntl(MemberReviewContent);
