// Import dependencies
import React, { Component } from 'react';
import { FormattedMessage } from 'react-intl';
import { Form } from 'formsy-react';
import Modal, { ModalHeader } from '../Modal.jsx';
import TextInput from '../Inputs/TextInput.jsx';
 
class SecurityQuestion extends Component {
 
	constructor(props) {
		super(props);
		// Set initial state
		this.state = {
			modalIsOpen: false,
			selectedQuestion: ''
		};
	}
 
	handleClickChoose(e) { 
		e.preventDefault();
		this.setState({ modalIsOpen: true });
	}
 
	handleClickEdit(e) { 
		e.preventDefault();
		this.setState({ modalIsOpen: true });
	}
 
	handleChooseQuestion(q) {
		this.setState({ modalIsOpen: false, selectedQuestion: q ? q : this.state.selectedQuestion });
		this.props.onChangeQuestion(this.props.name, q ? q : this.state.selectedQuestion);
	}
 
	handleCloseModal() {
		this.setState({ modalIsOpen: false });
	}
 
	handleAnswer(name, val) {
		this.props.onChangeAnswer(this.props.name, val);
	}
 
	render() {
		var className = ['security-question'].concat(
				this.props.disabled ? 'disabled' : []
			),
			questions = this.props.questions.map((q, idx) => {
				var classNames = ['btn btn-default btn-block'];
				if(q === this.state.selectedQuestion) {
					classNames.push('selected')
				}
				return <button key={idx} className={classNames.join(' ')} onClick={this.handleChooseQuestion.bind(this, q)}>{q}</button>
			}),
			buttonClassName = ['btn', 'btn-default', 'btn-pill', 'btn-add', 'choose-question'].concat(
				this.props.disabled ? 'disabled' : []
			);
		return (
			<div className={className.join(' ')}>
				{this.state.selectedQuestion ?
					(
						<div className="row">
							<div className="col-xs-8">
								<label>{this.state.selectedQuestion}</label>
							</div>
							<div className="col-xs-4 text-right">
								<a href="#" role="button" className="edit btn btn-link" onClick={this.handleClickEdit.bind(this)}>
									<FormattedMessage id="security-questions-button-change" />
								</a>
							</div>
							<div className="col-xs-12 col-md-6">
								<TextInput 
									id="security-question-input"
									name="security-question-input"
									onChange={this.handleAnswer.bind(this)}
									value=""
									requiredError={this.props.requiredAnswerError}
									validations="maxLength:50"
									validationErrors={{
										maxLength: this.props.answerErrorMaxLength
									}}
									required
								/>
							</div>
						</div>
					)
					:
					(
						<div className="row vertical-align">
							<div className="col-xs-8">
								{this.props.label}
							</div>
							<div className="col-xs-4 text-right">
								<button 
									className={buttonClassName.join(' ')} 
									onClick={this.handleClickChoose.bind(this)}
									disabled={this.props.disabled}
								>
									<FormattedMessage id="security-questions-button-add" />
								</button>
							</div>
						</div>
					)
				}
				<Modal
					className="security-question-modal"
					isOpen={this.state.modalIsOpen}
					contentLabel="Choose Security Question"
					onRequestClose={this.handleCloseModal.bind(this)}
					header={
						<ModalHeader closeButton={true}>{this.props.modalHeader}</ModalHeader>
					}
				>
					<div className="questions">{questions}</div>
				</Modal>
			</div>
		);
	}
}
 
SecurityQuestion.propTypes = {
	questions: React.PropTypes.array,
	id: React.PropTypes.string,
	label: React.PropTypes.oneOfType([
		React.PropTypes.string,
		React.PropTypes.element
	]),
	modalHeader: React.PropTypes.oneOfType([
		React.PropTypes.string,
		React.PropTypes.element
	]),
	requiredAnswerError: React.PropTypes.string,
	onChangeQuestion: React.PropTypes.func,
	onChangeAnswer: React.PropTypes.func,
	disabled: React.PropTypes.bool
}
 
SecurityQuestion.defaultProps = {
	questions: [],
	onChangeQuestion: () => {},
	onChangeAnswer: () => {},
	disabled: false,
	modalHeader: ''
}
 
export default SecurityQuestion;
