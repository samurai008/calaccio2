import React from 'react';
import { FormattedMessage } from 'react-intl';
import { householdMemberShape } from '../types';
import PeopleStore from '../Stores/PeopleStore';
 
const MemberAvatar = props => {
	var className = ['member-avatar'].concat(
			props.className ? props.className : [],
			props.vertical ? 'vertical' : [],
			props.person.avatar ? props.person.avatar : []
		),
		avatarClassName = ['avatar'].concat(
			props.person.avatar ? props.person.avatar : [],
			props.avatarSize ? props.avatarSize.split(' ') : []
		),
		firstName = (props.person.firstName.length > 10) ? (props.person.firstName.substr(0, 10) + 'â€¦') : props.person.firstName,
		lastInitial = PeopleStore.getPersonById(props.person.personId) === null ? (props.person.lastName.substring(0, 1) + '.') : PeopleStore.getLastInitial(props.person.personId) ;
	
	return (
		<div className={className.join(' ')}>
			<div className={avatarClassName.join(' ')} />
			<div className="member-info">
				{
					props.showName ?
						<div className="name">{firstName + ' ' + lastInitial} <br/>  </div>
						: ''
				}
				{
					props.showAge ?
						<span className="age">
							<span className=" hidden-sm hidden-md hidden-lg">(</span>
							<span>{props.person.age}</span>
							<span className=" hidden-sm hidden-md hidden-lg">)</span>
							<span className="hidden-xs"> <FormattedMessage id="years-abbreviated" /></span>
						</span>
						: ''
				}
				{props.children}
			</div>
		</div>
	);
};
 
MemberAvatar.propTypes = {
	person: householdMemberShape.isRequired,
	vertical: React.PropTypes.bool,
	avatarSize: React.PropTypes.string,
	showName: React.PropTypes.bool,
	showAge: React.PropTypes.bool
};
 
MemberAvatar.defaultProps = {
	avatarSize: 'lg',
	showName: true,
	showAge: true
};
 
export default MemberAvatar;
