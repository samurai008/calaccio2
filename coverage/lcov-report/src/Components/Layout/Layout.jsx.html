import React from 'react';
import { Link } from 'react-router';
 
export const ContentIntro = props => {
	var className = ['intro', 'inverted-theme'].concat(
		props.className ? props.className : []
	);
 
	return (
		<div className={className.join(' ')}>
			<div className="container">
				{props.children}
			</div>
		</div>
	)
}
 
export const ContentCard = props => {
	return (
		<div className="content content-box backdrop">
			<div className="container">
				{props.children}
			</div>
		</div>
	)
}
 
export const CardNavigation = props => {
	return (
		<div className="card-navigation">
			{
				(typeof props.onClick === "function")
				? <button className="btn-link card-navigation-link" onClick={props.onClick}><span className="ic_escape" />{props.text}</button>
				: <Link className="card-navigation-link" to={props.link}><span className="ic_escape"/>{props.text}</Link>
			}
			<span className="card-navigation-content">{props.children}</span>
		</div>
	)
}
 
CardNavigation.propTypes = {
	onClick: React.PropTypes.func
};

}
