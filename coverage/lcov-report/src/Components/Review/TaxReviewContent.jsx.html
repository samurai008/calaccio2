import React from 'react';
import { FormattedMessage } from 'react-intl';
import { Link } from 'react-router';
import MemberReview from '../MemberReview.jsx';
import FormattedPhone from '../FormattedPhone.jsx';
import { ReviewSection, ReviewRow, ReviewDT, ReviewDD } from '../ReviewSection.jsx';
 
const TaxReviewContent = props => {
	var primaryFiler = props.householdMembers.find(person => person.taxInfo && person.taxInfo.primaryTaxFiler === 'Y'),
		peopleWhoPlanToFile = props.householdMembers
			.filter((person) => person.taxInfo && person.taxInfo.filingTaxes === 'Y')
			.map((person, i) => {
				return (
					<MemberReview person={person} key={i}>
						<div><FormattedMessage id="tax-review-filer-status-label" tagName="b" /></div>
						{
							person.taxInfo.taxFilingStatus
							? (
								<FormattedMessage id={'tax-filing-status-' + person.taxInfo.taxFilingStatus} />
							) : (
								<FormattedMessage id="tax-review-primary-filer-no-status" />
							)
						}
					</MemberReview>
				);
			}),
 
		peopleRequiredToFile = props.householdMembers
			.filter((person) => person.taxInfo && person.taxInfo.requiredToFileTax === 'Y')
			.map((person, i) => <MemberReview person={person} key={i} /> ),
 
		dependants = props.householdMembers
			.filter((person) => person.taxInfo && person.taxInfo.taxClaimer === 'Y')
			.map((person, i) => {
				var claimee = props.householdMembers.find(member => member.personId === person.taxInfo.claimerPersonId);
				var custodialParentPhone = person.taxInfo.custodialParentPhoneNum;
				return (
					<MemberReview person={person} key={i}>
						<p><FormattedMessage id="tax-review-dependant-claimee-header" tagName="b" values={person} /></p>
						{
							claimee
							? (
								<div>
									
									<MemberReview person={claimee} key={i}>
									</MemberReview>
								</div>
							) : custodialParentPhone ? <FormattedPhone value={person.taxInfo.custodialParentPhoneNum} />
								:(
								<FormattedMessage id="tax-review-dependants-no-claimee" />
								)
						}
					</MemberReview>
				);
			});
 
	return (
		<ReviewSection>
			<ReviewRow id={props.id + '-primary-filer'} linkUrl={props.editUrlBase + 'primary-filer'}>
				<ReviewDT className={primaryFiler ? 'block' : ''}><FormattedMessage id="tax-review-primary-filer-label" /></ReviewDT>
				{ 
					primaryFiler
					? (
						<div className="row">
							<MemberReview person={primaryFiler}>
								<div><FormattedMessage id="tax-review-primary-filer-status" tagName="b" /></div>
								{
									primaryFiler.taxInfo.taxFilingStatus
									? (
										<FormattedMessage id={'tax-filing-status-' + primaryFiler.taxInfo.taxFilingStatus} />
									) : (
										<FormattedMessage id="tax-review-primary-filer-no-status" />
									)
								}
							</MemberReview>
						</div>
					) : (
						<ReviewDD><FormattedMessage id="tax-review-primary-filer-nobody" /></ReviewDD>
					)
				}
			</ReviewRow>
			<ReviewRow id={props.id + '-plan-to-file'} linkUrl={props.editUrlBase + 'plan-to-file'}>
				<ReviewDT className={peopleWhoPlanToFile.length > 0 ? 'block' : ''}><FormattedMessage id="tax-review-plan-to-file-header" /></ReviewDT>
				{
					peopleWhoPlanToFile.length > 0 
					? <div className="row">{peopleWhoPlanToFile}</div>
					: <ReviewDD><FormattedMessage id="tax-review-plan-to-file-nobody" /></ReviewDD>
				}
				
			</ReviewRow>
			<ReviewRow id={props.id + '-required-to-file'} linkUrl={props.editUrlBase + 'required-to-file'}>
				<ReviewDT className={peopleRequiredToFile.length > 0 ? 'block' : ''}><FormattedMessage id="tax-review-required-to-file-header" /></ReviewDT>
				{
					peopleRequiredToFile.length > 0 
					? <div className="row">{peopleRequiredToFile}</div> 
					: <ReviewDD><FormattedMessage id="tax-review-required-to-file-nobody" /></ReviewDD>
				}
			</ReviewRow>
			<ReviewRow id={props.id + '-dependents'} linkUrl={props.editUrlBase + 'dependents'}>
				<ReviewDT className={dependants.length > 0 ? 'block' : ''}><FormattedMessage id="tax-review-dependants-header" tagName="h4" /></ReviewDT>
				{
					dependants.length > 0 
					? <div className="row">{dependants}</div>
					: <ReviewDD><FormattedMessage id="tax-review-dependants-nobody" /></ReviewDD>
				}
			</ReviewRow>
		</ReviewSection>
	);
}
 
TaxReviewContent.propTypes = {
	householdMembers: React.PropTypes.array,
	editUrlBase: React.PropTypes.string
};
 
TaxReviewContent.defaultProps = {
	editUrlBase: '/household/tax-information/edit/'
};
 
export default TaxReviewContent;
