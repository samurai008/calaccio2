// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import { HouseholdMemberChangeList } from './HouseholdMemberChangeList.jsx';
import PeopleStore from '../../Stores/PeopleStore.js';
import ApplicationStore from '../../Stores/ApplicationStore.js';
import MemberAvatar from '../../Components/MemberAvatar.jsx';
 
let tempAvatars = {};
 
export const ApplicationChangeList = injectIntl((props)  => {
	let changes = props.householdMemberChanges.map((changeLogPerson,idx) => {
		let fullPerson = PeopleStore.getPersonById(changeLogPerson.personId);
		if(fullPerson === null) {
			fullPerson = {};
			if(typeof tempAvatars[changeLogPerson.personId] === "undefined") {
				tempAvatars[changeLogPerson.personId] = PeopleStore.getNextAvatar();
			}
			fullPerson.avatar = tempAvatars[changeLogPerson.personId];
		}
		return (
			<div className="" key={idx}>
				<div className="avatar col-xs-12">
					<MemberAvatar 
						person={Object.assign(fullPerson, changeLogPerson)} 
						showName={true} 
						showAge={typeof fullPerson.age !== "undefined"}
						avatarSize="sm"
					/>
				</div>
				<p/>
				<HouseholdMemberChangeList 
					changeList={changeLogPerson.changeLog} 
					personId={changeLogPerson.personId}
					onChange={props.onChange}
				/>
			</div>
		) 
	});
	
	return (
		<div className="household-members-list">
				{changes}
		</div>
	);
});
 
ApplicationChangeList.propTypes = {
	householdMemberChanges: React.PropTypes.array
};
 
ApplicationChangeList.defaultProps = {
	householdMemberChanges: [],
	viewOnlyMode: false
};
