// Import Libs & Components
import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import DropDown from '../../Components/Inputs/DropDown.jsx';
import DateInput from '../../Components/Inputs/DateInput.jsx';
import DataStore from '../../Stores/DataStore.js';
import FormHOC from '../../Components/Forms/FormHOC.jsx';
 
export const ApplicationChange = injectIntl((props)  => {
	let change = Object.assign({}, props.changeItem),
		ahbxLocale = document.cookie.match(/(^|;)\s*ahbx_locale\s*=\s*([^;]+)/),
		currentLocale = (ahbxLocale) ? ahbxLocale.pop() : 'en',
		dropDownOptions = DataStore.getChangeTypesByCategory(props.changeItem.changeCategory).reduce((opts, type) => {
			opts.push({value: type, label: props.intl.messageIfExists("application-signature-changelog-change-type-" + type)});
			return opts;
		}, []),
		previousValue = typeof props.changeItem.values === "undefined" 
        ? props.changeItem.previousValue 
        : typeof props.changeItem.values[currentLocale].previous === "undefined"
		? ""
		: props.changeItem.values[currentLocale].previous,
        changedValue = typeof props.changeItem.values === "undefined"
        ? props.changeItem.changedValue
        : typeof props.changeItem.values[currentLocale].changed === "undefined"
		? ""
		: props.changeItem.values[currentLocale].changed,
		changeElement = props.changeItem.changeElement[currentLocale],
		changeCategory = props.changeItem.changeCategory,
		/* on change, pass the needed structure & lookup info back to the top level to change the log in place */
		handleReasonChange = (name, value) => {
			change[name] = value;
			props.onChange('changeLogDelta', { personId: props.personId, index: props.changeItemKey, changeLog: change});
		};
	return (
		<div className="household-member-change">
			<FormattedMessage 
				id="application-signature-changelog-change-category-label" 
				values={{changeCategory: props.intl.messageIfExists("application-signature-changelog-change-category-" + changeCategory)}} 
				tagName="h5" />
			<FormattedMessage id="application-signature-changelog-change-type-label" values={{changeType: changeElement}} tagName="p"/> 
			<FormattedMessage id="application-signature-changelog-previous-label" values={{previousValue: previousValue}} tagName="p"/>
			<FormattedMessage id="application-signature-changelog-changed-label" values={{newValue: changedValue}} tagName="p"/>
			<div className="col-xs-12 col-sm-8 changelog-field">
			<DropDown
				className="application-signature-changelog-dropdown"
				label={<FormattedMessage id="application-signature-changelog-reason-label" />}
				aria-label={<FormattedMessage id="application-signature-changelog-reason-label"/>}
				customAriaLabel={props.intl.messageIfExists("application-signature-changelog-reason-label")}
				name="changeType"
				id={"changeType-" + props.changeItem.changeCategory + "-" + props.changeItemKey}
				options={dropDownOptions}
				onChange={handleReasonChange}
				value={props.changeItem.changeType}
				requiredError={props.intl.messageIfExists('application-signature-changelog-reason-required-error')}
				required
			/>
			</div>
			<div className="col-xs-12 col-sm-4 changelog-field">
			<DateInput
				label={props.intl.messageIfExists('application-signature-changelog-event-date-label')}
				aria-label={props.intl.messageIfExists('application-signature-changelog-event-date-label')}
				id={"eventDate-" + props.changeItem.changeCategory + "-" + props.changeItemKey}
				name="eventDate"
				onChange={handleReasonChange}
				showIcon={true}
				datepickerEnabled={true} 
				value={props.changeItem.eventDate}
				validations={{isValidDate: true}} 
				validationErrors={{isValidDate: props.intl.messageIfExists('application-signature-changelog-event-date-required-error')}}
				requiredError={props.intl.messageIfExists('application-signature-changelog-event-date-required-error')}
				required
			/>
			</div>
		</div>
	);
});
 
ApplicationChange.propTypes = {
	changeItemKey: React.PropTypes.number,
	changeItem: React.PropTypes.object,
	personId: React.PropTypes.string
};
 
ApplicationChange.defaultProps = {
	changeItemKey: 0,
	changeItem: {},
	personId: ''
};
