import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import Formsy from 'formsy-react';
import FormHOC from '../../Components/Forms/FormHOC.jsx';
import MultiSelectCheckBoxes from '../../Components/Inputs/MultiSelectCheckBoxes.jsx';
import RadioInput from '../../Components/Inputs/RadioInput.jsx';
 
 
 
export const SogiForm = injectIntl(FormHOC((props) => {
 
	const multiSelectOptions = {
		genderOptions: {
			'F': 'sogi-gender-option-label-F',
			'M': 'sogi-gender-option-label-M',
			'T': 'sogi-gender-option-label-TFM',
			'G': 'sogi-gender-option-label-G',
			'B': 'sogi-gender-option-label-B',
			'A': 'sogi-gender-option-label-A'
		},
		orientationOptions: {
			'S': 'sogi-orientation-label-S',
			'G': 'sogi-orientation-label-G',
			'B': 'sogi-orientation-label-B',
			'Q': 'sogi-orientation-label-Q',
			'A': 'sogi-orientation-label-A',
			'U': 'sogi-orientation-label-U'
		}
	};
 
	Object.keys(multiSelectOptions).forEach(option => {
		Object.keys(multiSelectOptions[option]).forEach((code) => {
			multiSelectOptions[option][code] = props.intl.messageIfExists(multiSelectOptions[option][code]);
		});
	});
 
	return (
		<div className="sogi-form">
			<RadioInput
				label={props.intl.messageIfExists('sogi-gender-label')}
				beforeInput={<p className="label-note">{props.intl.messageIfExists('sogi-gender-label-note')}</p>}
				id="sogi.genderIdentificationCode"
				name="sogi.genderIdentificationCode"
				onChange={props.handleFieldChange}
				value={props.getFormValue('sogi.genderIdentificationCode')}
				options={multiSelectOptions.genderOptions}
				inline={false}
			/>
			<RadioInput
				label={props.intl.messageIfExists('sogi-sex-label')}
				id="sogi.birthCertificateGenderCode"
				name="sogi.birthCertificateGenderCode"
				onChange={props.handleFieldChange}
				value={props.getFormValue('sogi.birthCertificateGenderCode')}
				options={{F: props.intl.messageIfExists('sogi-gender-option-label-F'),M: props.intl.messageIfExists('sogi-gender-option-label-M') }}
			/>
			<RadioInput
				label={props.intl.messageIfExists('sogi-orientation-label')}
				id="sogi.sexualOrientationCode"
				name="sogi.sexualOrientationCode"
				onChange={props.handleFieldChange}
				value={props.getFormValue('sogi.sexualOrientationCode')}
				options={multiSelectOptions.orientationOptions}
				inline={false}
			/>
		</div>
	);
 
}));
