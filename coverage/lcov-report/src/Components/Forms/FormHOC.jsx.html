export const deepGet = (path, obj) => {
	const nested = path.split('.').reduce(function(prev, curr) {
		return prev ? prev[curr] : undefined
	}, obj);
	// fallback to flat objects
	return (typeof nested !== "undefined") ? nested : obj[path];
};
export const deepSet = (path, val) => {
	let obj = {};
	const keys = path.split('.');
	if(keys.length > 0) {
		const lastKey = keys.pop();
		const lastObj = keys.reduce((obj, key) =>
				obj[key] = obj[key] || {},
			obj);
		lastObj[lastKey] = val;
		return obj;
	}
	// fallback to flat path
	obj[path] = val;
	return obj;
};
