import React from 'react';
import { ContentIntro, ContentCard } from '../../Components/Layout/Layout.jsx';
import { intlShape, injectIntl, FormattedMessage, FormattedDate, FormattedHTMLMessage } from 'react-intl';
import FormHOC from './FormHOC.jsx';
import RadioInput from '../Inputs/RadioInput.jsx';
 
const OptInForm = injectIntl(FormHOC((props) => {
	return (
		<div>
			<RadioInput
				className="opt-in-radio"
				label={<FormattedMessage id="opt-in-label" />}
				name="subsidizedApplication"
				id="radio"
				onChange={props.handleFieldChange}
				value={props.getFormValue('subsidizedApplication')}
				options={{
					Y: props.intl.messageIfExists('opt-in-radio-yes'),
					N: props.intl.messageIfExists('opt-in-radio-no'),
				}}					
				required
			/>
		</div>
	);
}));
 
const OptInFormPage = injectIntl((props) => {
	var classNames = ['opt-in'].concat(
			props.className ? props.className.split(' ') : []
		),
		handleSubmitClick = e => {
			e.preventDefault();
			props.handleSubmitForm();
		};
	return (
		<div className={classNames.join(' ')}>
			<ContentIntro>
				<div className="illustration ic_xray_crutch_desktop hidden-xs" />
				<div className="illustration ic_xray_crutch_mobile hidden-sm hidden-md hidden-lg" />
				<h1><FormattedMessage id="opt-in-header" /></h1>
				{
					props.intl.doesMessageExist('opt-in-text') ? (
						<p><FormattedMessage id="opt-in-text" /></p>
					) : null
				}
			</ContentIntro>
 
			<ContentCard>
				<div className="row">
					<div className="col-xs-12 col-sm-8 col-sm-push-2 col-md-12 col-md-push-0">
						<OptInForm
							formData={props.formData}
							onFormValid={props.onFormValid} 
							onFormInvalid={props.onFormInValid}
							onChange={props.onChange}
						/>
						<div className="text-center row">
							<div className="col-xs-12 col-sm-4 col-sm-push-4 col-md-4 col-md-push-4">
								<button
									id="opt-in-button-text"
									type="submit"
									className="btn btn-primary btn-block"
									onClick={handleSubmitClick}
									disabled={!props.formIsValid}
								>
									<FormattedMessage id="opt-in-button-text" />
								</button>
							</div>
						</div>
					</div>
				</div>
			</ContentCard>
		</div>
	);
});
 
OptInFormPage.propTypes = {
	handleSubmitForm: React.PropTypes.func
};
 
OptInFormPage.defaultProps = {
	handleSubmitForm: e => {}
};
 
export { OptInForm as default, OptInFormPage };
