	render() {
		var children = React.Children.map(this.props.children, (child, i) => {
				return React.cloneElement(child, {
					onRequestToggle: this.handleToggleRequest.bind(this),
					open: this.state.activeRow === i,

		return (

const Panel = props => {
 
	var className = ['accordion-panel'].concat(
			props.open ? 'open' : [],
			props.className ? props.className : []
		);
 
	const handleHeaderClick = e => {
		e.preventDefault();
		props.onRequestToggle(props.index);
	}
 
	return (
		<div className={className.join(' ')}>
			<div className="accordion-panel-control">
				<a href="#" onClick={handleHeaderClick}>
					{props.control}
				</a>
			</div>
			{ 
				props.open
				? (<div className="accordion-panel-content">{props.children}</div>)
				: null
			}
		</div>
	);
}
 
Panel.propTypes = {
	control: React.PropTypes.oneOfType([
		React.PropTypes.string,
		React.PropTypes.element
	]),
	open: React.PropTypes.bool,
	index: React.PropTypes.number,
	onRequestToggle: React.PropTypes.func
};
 
Panel.defaultProps = {
	open: false
};
 
export { Accordion as default, Panel };
