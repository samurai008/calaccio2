import React, { Component } from 'react';
import PhoneInput from '../../Components/Inputs/PhoneInput.jsx';
import RadioInput from '../../Components/Inputs/RadioInput.jsx';
import NumericInput from '../../Components/Inputs/NumericInput.jsx';
import ConditionalField from '../../Components/ConditionalField.jsx';
import { FormattedMessage, FormattedHTMLMessage } from 'react-intl';
const HouseholdMemberPhoneInput = (props) => {
	return (
	<span className="multiple-phone-input">
		<PhoneInput
			label={props.intl.messageIfExists('household-add-member-contact-home-phone-label', props.i18nValues)}
			aria-label={props.intl.messageIfExists('household-add-member-contact-home-phone-label', props.i18nValues)}
			id="homephone"
			name={(props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'homePhoneNumber'}
			value={props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'homePhoneNumber')}
			onChange={props.handleFieldChange}
			validations="minLength:10,maxLength:14,doesNotStartWith:0"
			validationErrors={{
				minLength: props.intl.messageIfExists('account-creation-create-contact-method-phone-error'),
				maxLength: props.intl.messageIfExists('account-creation-create-contact-method-phone-error'),
				doesNotStartWith:  props.intl.messageIfExists('account-creation-create-contact-method-phone-error')
			}}
			optional={
				props.phoneNumbersOptional === true
				|| (props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'workPhoneNumber').length >= 10
				|| props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'cellPhoneNumber').length >= 10)}
			required={
				props.phoneNumbersOptional === false
				&& props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'workPhoneNumber').length < 10
				&& props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'cellPhoneNumber').length < 10
			}
			requiredError={props.intl.messageIfExists('account-creation-create-contact-method-phone-error')}
		/>
		<PhoneInput
			label={props.intl.messageIfExists('household-add-member-contact-cell-phone-label', props.i18nValues)}
			aria-label={props.intl.messageIfExists('household-add-member-contact-cell-phone-label', props.i18nValues)}
			id="cellphone"
			name={(props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'cellPhoneNumber'}
			value={props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'cellPhoneNumber')}
			onChange={props.handleFieldChange}
			validations="minLength:10,maxLength:14,doesNotStartWith:0"
			validationErrors={{
				minLength: props.intl.messageIfExists('account-creation-create-contact-method-phone-error'),
				maxLength: props.intl.messageIfExists('account-creation-create-contact-method-phone-error'),
				doesNotStartWith:  props.intl.messageIfExists('account-creation-create-contact-method-phone-error') 
			}}
			optional={
				props.phoneNumbersOptional === true
				|| (props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'workPhoneNumber').length >= 10
				|| props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'homePhoneNumber').length >= 10)
			}
			required={
				props.phoneNumbersOptional === false
				&& props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'workPhoneNumber').length < 10
				&& props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'homePhoneNumber').length < 10
			}
			requiredError={props.intl.messageIfExists('account-creation-create-contact-method-phone-error')}
		/>
		<ConditionalField condition={
			props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'cellPhoneNumber').length >= 10
			&& props.dataObjPrefix === "primaryContactData.contact"
		}>
			<RadioInput
				label={props.intl.messageIfExists('household-add-member-contact-phone-consent-label', props.i18nValues)}
				aria-label={props.intl.formatMessage({ id: 'household-add-member-contact-phone-consent-label' }, props.i18nValues)}
				id={'isCoverageDetailsRequired'}
				name={(props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'isCoverageDetailsRequired'}
				onChange={props.handleFieldChange}
				options={{ Y: props.intl.messageIfExists('Yes'), N: props.intl.messageIfExists('No') }}
				value={props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'isCoverageDetailsRequired')}
				requiredError={props.intl.messageIfExists('household-add-member-contact-phone-consent-required-error')}
				required={props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'cellPhoneNumber').length >= 10}
			/>
			<ConditionalField condition={props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'isCoverageDetailsRequired') === "Y"}>
				<FormattedHTMLMessage id="household-add-member-contact-phone-consent-hint" tagName="p" />
				<br />
				<br />
			</ConditionalField>
		</ConditionalField>
		<PhoneInput
			label={props.intl.messageIfExists('household-add-member-contact-work-phone-label', props.i18nValues)}
			aria-label={props.intl.messageIfExists('household-add-member-contact-work-phone-label', props.i18nValues)}
			id="workphone"
			name={(props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'workPhoneNumber'}
			value={props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'workPhoneNumber')}
			onChange={props.handleFieldChange}
			validations="minLength:10,maxLength:14,doesNotStartWith:0"
			validationErrors={{
				minLength: props.intl.messageIfExists('account-creation-create-contact-method-phone-error'),
				maxLength: props.intl.messageIfExists('account-creation-create-contact-method-phone-error'),
				doesNotStartWith:  props.intl.messageIfExists('account-creation-create-contact-method-phone-error') 
			}}
			optional={
				props.phoneNumbersOptional === true
				|| (props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'homePhoneNumber').length >= 10
				|| props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'cellPhoneNumber').length >= 10)
			}
			required={
				props.phoneNumbersOptional === false
				&& props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'cellPhoneNumber').length < 10
				&& props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'homePhoneNumber').length < 10
			}
			requiredError={props.intl.messageIfExists('account-creation-create-contact-method-phone-error')}
		/>
 
		<NumericInput
			className="extension"
			label={props.intl.messageIfExists('household-add-member-contact-extension-label', props.i18nValues)}
			aria-label={props.intl.messageIfExists('household-add-member-contact-extension-label', props.i18nValues)}
			id="extension"
			name={(props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'extension'}
			value={props.getFormValue((props.dataObjPrefix ? props.dataObjPrefix + '.' : '') + 'extension')}
			onChange={props.handleFieldChange}
			pattern="[0-9]{0,5}"
			maxLength="5"
			optional={true}
		/>
	</span>
	);
};
 
HouseholdMemberPhoneInput.propTypes = {
	intl: React.PropTypes.object,
	handleFieldChange: React.PropTypes.func,
	getFormValue: React.PropTypes.func,
	phoneNumbersOptional: React.PropTypes.bool,
	dataObjPrefix: React.PropTypes.string
};
 
HouseholdMemberPhoneInput.defaultProps = {
	phoneNumbersOptional: true
};
 
export default HouseholdMemberPhoneInput;
