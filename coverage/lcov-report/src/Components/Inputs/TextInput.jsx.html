import React, { Component } from 'react';
import { HOC } from 'formsy-react';
import { FormattedMessage } from 'react-intl';
import InputShape from './InputShape.jsx';
import ToolTip from '../ToolTip.jsx';
 
class TextInput extends Component {
 
	constructor(props) {
		super(props);
		// Set initial state
		this.state = {};
 
		this.focus = this.focus.bind(this);
	}
 
	componentWillMount() {
		this.props.setValue(this.props.formatInput(this.props.value));
	}
 
	handleInputChange(e) {
		var val = this.props.formatInput(e.target.value);
		// if pattern, limit value to matches
		if(this.props.pattern) val = val.match(new RegExp(this.props.pattern)).join();
		this.props.setValue(val);
		this.props.onChange(this.props.name, val);
	}
 
	handleInputFocus(e) {
		this.refs.container.handleInputFocus(e);
	}
 
	handleInputBlur(e) {
		this.refs.container.handleInputBlur(e);
	}
 
	handleInputChangeOnPaste(pasteAllowed, e){
		if (pasteAllowed !== true){
			e.preventDefault();
		}
	}
 
	focus() {
		this.textInput.focus();
	}
 
	render() {
		var inputProps = ['type', 'name', 'maxLength', 'id', 'placeholder', 'required', 'disabled', 'aria-label'].reduce((prev, next) => {
				if(this.props[next]) prev[next] = this.props[next];
				return prev;
			}, {});
 
		inputProps['aria-label'] = this.props['aria-label'] ? this.props['aria-label'] : ((!this.props.label && this.props.inlineLabel) ? this.props.inlineLabel : this.props.label);
 
		// ariaLabel has to be a string
		if(typeof inputProps['aria-label'] !== 'string') {
			inputProps['aria-label'] = null;
		}
 
		return (
			<InputShape ref="container" {...this.props}>
				{this.props.beforeInput}
				<input
					{...inputProps}
					onPaste={this.handleInputChangeOnPaste.bind(this, this.props.allowCopyPaste)}
					ref={input => this.textInput = input }
					value={this.props.getValue()}
					htmlFor={this.props.id}
					onChange={this.handleInputChange.bind(this)}
					onFocus={this.handleInputFocus.bind(this)}
					onBlur={this.handleInputBlur.bind(this)}
					className="form-control"
					onKeyUp={this.props.onKeyUp}
					onKeyDown={this.props.onKeyDown}
				/>
				{this.props.children}
			</InputShape>
		);
	}
}
 
TextInput.propTypes = {
	value: React.PropTypes.string,
	id: React.PropTypes.string.isRequired,
	name: React.PropTypes.string.isRequired,
	type: React.PropTypes.string,
	placeholder: React.PropTypes.string,
	formatInput: React.PropTypes.func,
	pattern: React.PropTypes.string,
	onChange: React.PropTypes.func,
	onFocus: React.PropTypes.func,
	onBlur: React.PropTypes.func,
	beforeInput: React.PropTypes.object,
	allowCopyPaste: React.PropTypes.bool,
	onKeyDown: React.PropTypes.func,
	onKeyUp: React.PropTypes.func
};
 
TextInput.defaultProps = {
	className: '',
	type: 'text',
	required: null,
	formatInput: val => val,
	onChange: () => {},
	onFocus: () => {},
	onBlur: () => {},
	beforeInput: null,
	allowCopyPaste: true,
	onKeyDown: () => {},
	onKeyUp: () => {}
};
 
export default HOC(TextInput);
