class DropDown extends Component {
	constructor(props) {
		super(props);

		};
		this.handleInputBlur = this.handleInputBlur.bind(this);

	handleSelectChange(val) {
		this.props.setValue(val.value);
		this.props.onChange(this.props.name, val.value);
	}
 
	handleInputBlur(e) {
		if(this.props.required || this.props.validations) {
			this.refs.container.handleInputBlur(e);
		}
	}
 
	handleInputChange(name, val) {
		this.props.onChange(this.props.name, this.props.getValue(), val);
	}
 
	render() {
		var className = ['drop-down'].concat(
				this.props.className ? this.props.className.split(' ') : []
			),
			options = [],

				clearable: false
			},
			// customFor may accept an array at a future date.
			customFor = typeof this.props.customFor === 'string' ? [this.props.customFor] : this.props.customFor,
			customInput = customFor.indexOf(this.props.getValue()) >= 0 ?

							<TextInput
								id={'drop-down-custom-' + this.props.id}
								name={'drop-down-custom-' + this.props.id}
								aria-label={this.props.customAriaLabel}
								placeholder={this.props.customPlaceholder}
								onChange={this.handleInputChange.bind(this)}
								value={this.props.customValue}
								required={this.props.required}
							/>
						) : '';
 
		selectProps = ['name', 'id', 'required', 'disabled', 'options', 'placeholder', 'clearable'].reduce((prev, next) => {
			if(this.props[next]) prev[next] = this.props[next];
			return prev;
		}, selectProps);

		return (
			<InputShape

}
