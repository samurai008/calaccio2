import React, { Component } from 'react';
import { HOC } from 'formsy-react';
import InputShape from './InputShape.jsx';
 
class CheckBox extends Component {
 
	constructor(props) {
		super(props);
	}
 
	handleChange(e) {
		var value = !!this.refs.checkbox.checked;
		this.props.setValue(value);
		this.props.onChange(this.props.name, value);
	}
 
	render() {
		var className = ['checkbox-input'].concat(
				this.props.className ? this.props.className.split(' ') : []
			),
			inputProps = {
				ref: 'checkbox',
				type: 'checkbox',
				name: this.props.name,
				id: this.props.id,
				checked: this.props.getValue(),
				onChange: this.handleChange.bind(this),
				'aria-labelledby': this.props.name + '-label',
				disabled: this.props.disabled
			};
 
		return (
			<InputShape 
				{...this.props}
				label={null}
				className={className.join(' ')} 
			>
				<label className="checkbox-inline">
					<input {...inputProps} />
					<span className="check"></span>
					<span id={inputProps.name + '-label'}>{this.props.label}</span>
				</label>
			</InputShape>
		);
	}
}
 
CheckBox.propTypes = {
	options: React.PropTypes.object.isRequired,
	value: React.PropTypes.oneOfType([
		React.PropTypes.bool,
		React.PropTypes.string
	]),
	id: React.PropTypes.string.isRequired,
	name: React.PropTypes.string.isRequired,
	onChange: React.PropTypes.func,
	disabled: React.PropTypes.bool
};
 
CheckBox.defaultProps = {
	className: '',
	options: {},
	value: '',
	onChange: val => {},
	disabled: false
};
 
export default HOC(CheckBox);
