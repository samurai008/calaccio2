import React, { Component } from 'react';
import TextInput from './TextInput.jsx';
import isInteger from 'is-integer';
 
class CurrencyInput extends Component {
	
	constructor(props) {
		super(props);
	}
 
	formatInput(val) {
		var endsInZero = val && (val.substr(val.length - 3).indexOf('.') >= 0 && val.substr(val.length - 1) === '0'),
			hasPeriod = val && val.substr(val.length - 3).indexOf('.') >= 0,
			simpleNum = val ? val.replace(this.props.allowNegativeValue ? /([,])/g : /([,-])/g, '') : '',
			num = parseFloat(simpleNum),
			formattedNum = this.props.allowNegativeValue && simpleNum === '-' ? '-' : (
								isNaN(num) ? '' : (num.toLocaleString('en-US', { maximumFractionDigits: 2 }) + (isInteger(num) && hasPeriod ? '.' : '')) + (endsInZero ? '0' : '')
							);
		return formattedNum;
	}
 
	handleBlur(e) {
		// Add a zero if there's only one decimal	
		var num = this.props.onBlur(e) || e.target.value;
		if(num.substr(num.length - 2, 1) === '.') {
			num = num + '0';
		}
		return num;
	}
 
	render() {
		var className = ['currency-input'].concat(
							this.props.className ? this.props.className.split(' ') : []
						);
		
		return (
			<TextInput 
				{...this.props} 
				className={className.join(' ')}
				type="text" 
				addOn="$"
				formatInput={this.formatInput.bind(this)} 
				onBlur={this.handleBlur.bind(this)}
			/>
		);
	}
}
 
CurrencyInput.PropTypes = {
	allowNegativeValue: React.PropTypes.bool,
	onBlur: React.PropTypes.func
};
 
CurrencyInput.defaultProps = {
	className: '',
	allowNegativeValue: false,
	onBlur: e => e.target.value
};
 
export default CurrencyInput;
