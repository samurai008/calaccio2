import React, { Component } from 'react';
import Formsy from 'formsy-react';
import TextInput from './TextInput.jsx';
 
Formsy.addValidationRule('doesNotStartWith', (values, value, otherValue) => {
	let first = value.charAt(0) === '(' ? value.charAt(1) : value.charAt(0);
	return typeof value === 'string' && parseInt(first) !== otherValue;
});
 
class PhoneInput extends Component {
 
	constructor(props) {
		super(props);
	}
 
	formatInput(val) {
		var digitsOnly = ('' + val).replace(/\D/g, '').substr(0, 10),
			matches = digitsOnly.match(/^(\d{3})(\d{3})(\d{4})$/);
		return (!matches) ? digitsOnly : '(' + matches[1] + ') ' + matches[2] + '-' + matches[3];
	}
 
	render() {
		var className = ['phone-input'].concat(
							this.props.className ? this.props.className.split(' ') : []
						);
		
		return (
			<TextInput 
				{...this.props} 
				className={className.join(' ')}
				type="text" 
				formatInput={this.formatInput.bind(this)} 
				/* formsy internal ref to input component */
				innerRef={(component) => { this.inputRef = component; }}
			/>
		);
	}
}
 
PhoneInput.PropTypes = {
};
 
PhoneInput.defaultProps = {
	className: '',
};
 
export default Formsy.HOC(PhoneInput);
