	updateUploadedFiles() {
		if (this.useModal === true){
			let uploadedDocs =  DocumentUploadStore.getSubsetDocuments(this.props.documentType);
			uploadedDocs.forEach((uploaded) => {
				let match = this.props.documents.find((doc) => {
					return doc.value === uploaded.documentUploadType.value;

				if (match && match.value) {
					uploaded.documentUploadType.label  = match.label;

				uploadedFiles: uploadedDocs

					uploaded.documentUploadType.label  = match.label;

				uploadedFiles: uploadedDocs
