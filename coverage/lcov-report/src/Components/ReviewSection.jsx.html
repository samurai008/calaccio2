import React from 'react';
import { Link } from 'react-router';
import { FormattedMessage } from 'react-intl';
import ApplicationActions from '../Actions/ApplicationActions.js';
/*
	Each child passed in will be wrapped in an <li>. 
	If you need multiple elements in a section, but sure to wrap them in a parent element when you pass them in.
*/
 
export const ReviewSection = props => {
	var className = ['review-content', 'review-section'].concat(
			props.className ? props.className.split(' ') : []
		);
	return (
		<div className={className.join(' ')}>
			{
				props.header ? (
					<h4>{props.header}</h4>
				) : null
			}
			<div className="section">
				<ul className="list-group">
					{
						React.Children.toArray(props.children)
							.filter(child => child !== null)
							.map((child, i) => <li key={i} className="list-group-item">{child}</li>)
					}
				</ul>
			</div>
		</div>
	);
};
 
export const ReviewRow = props => {
	var className = ['review-row'].concat(
			props.className ? props.className.split(' ') : []
		);
	return (
		<div className={className.join(' ')}>
			{ props.linkUrl
				? (
					<Link id={props.id + '-edit'} className="pull-right edit" to={ props.linkUrl } onClick={ApplicationActions.startRACEdit.bind(ApplicationActions)}>
						<FormattedMessage id={props.linkTextKey} />
					</Link>
				)
				: null
			}
			{ 
				props.items
					? props.items.map((item, i) => {
						return (
							<ReviewItem key={i}>
								<ReviewDT>{item.label}</ReviewDT>
								<ReviewDD>{item.value}</ReviewDD>
							</ReviewItem>
						);
					})
					: null
			}
			{ 
				props.children 
			}
		</div>
	);
};
 
ReviewRow.propTypes = {
	linkUrl: React.PropTypes.string,
	linkTextKey: React.PropTypes.string,
	items: React.PropTypes.array
};
 
ReviewRow.defaultProps = {
	linkTextKey: 'household-review--generic--edit-member-linkText'
};
 
export const ReviewItem = props => {
	return (
		<div className="item-container">
			{props.children}
		</div>
	);
};
 
export const ReviewDT = props => {
	var className = ['item', 'item__label'].concat(
			props.className ? props.className.split(' ') : []
		);
	return (
		<div className={className.join(' ')}>
			{props.children}
		</div>
	);
};
 
export const ReviewDD = props => {
	var className = ['item', 'item__value'].concat(
			props.className ? props.className.split(' ') : []
		);
	return (
		<div className={className.join(' ')}>
			{props.children}
		</div>
	);
};
