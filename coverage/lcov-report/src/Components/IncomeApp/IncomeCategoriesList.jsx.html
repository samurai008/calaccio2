import React from 'react';
import { FormattedMessage, FormattedHTMLMessage } from 'react-intl';
import { categoryShape } from '../../types.js';
 
const IncomeCategoriesList = (props) => {
	var onRequestAdd = categoryType => {
			props.onRequestAdd(categoryType);
		},
		categories = props.items.map((cat, key) => <li key={key}><IncomeCategory id={props.id + '-' + cat.categoryType} item={cat} onRequestAdd={onRequestAdd} /></li>);
	return (
		<div className="income-categories-list">
			<ul>{categories}</ul>
		</div>
	);
}
 
IncomeCategoriesList.propTypes = {
	onRequestAdd: React.PropTypes.func,
	items: React.PropTypes.array
};
 
IncomeCategoriesList.defaultProps = {
	onRequestAdd: categoryType => {},
	items: []
};
 
const IncomeCategory = (props) => {
	var classNames = ['income-category'],
		onAdd = e => {
			e.preventDefault();
			props.onRequestAdd(props.item.categoryType);
		};
 
	classNames.push(props.item.categoryType);
 
	return (
		<div className={classNames.join(' ')}>
			<div className="row">
				<div className="col-xs-8">
					<div className="title-column">
						<FormattedHTMLMessage id={props.item.categoryType + '-category-title'} />
					</div>
				</div>
				<div className="col-xs-4 text-center">
					<button id={props.id + '-add'} className="btn btn-pill btn-add" onClick={onAdd}>
						<FormattedMessage id="2.8.2.1-select-income-sub-category-modal-button1" />
						<span className="sr-only">
							<FormattedHTMLMessage id={props.item.categoryType + '-category-title'} />
						</span>
					</button>
				</div>
			</div>
		</div>
	);
}
 
IncomeCategory.propTypes = {
	item: categoryShape.isRequired,
	onRequestAdd: React.PropTypes.func
};
 
IncomeCategory.defaultProps = {
	onRequestAdd: categoryType => {}
};
 
export default IncomeCategoriesList;
