import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage, FormattedDate } from 'react-intl';
import ToolTip from '../ToolTip.jsx';
import MoneyValue, { IncomeValuePerTimeframe } from '../MoneyValue.jsx';
import RecordSummaryRow from './RecordSummaryRow.jsx';
import { incomeTypeShape } from '../../types.js';
 
class IncomeType extends Component {
	constructor(props) {
		super(props);
	}
 
	handleClickAdd(e) {
		e.preventDefault();
		if(this.props.viewOnlyMode === true) return;
		this.props.onAdd(this.props.item.incomeType);
	}
 
	handleClickEdit(incomeId, e) {
		e.preventDefault();
		this.props.onEdit(this.props.item.incomeType, incomeId);
	}
 
	render() {
		var incomes;
		if(this.props.showSummaryRow) {
			incomes = this.props.incomes.map((income, key) => {
				return (
					<li key={key}>
						<RecordSummaryRow
							id={this.props.id + '-record-summary'}
							item={income}
							showEditColumn={true}
							onEdit={this.handleClickEdit.bind(this, income.id)}
						/>
					</li>
				)
			});
		}
		return (
			<div className="income-type">
				<div className="row">
					<div className="col-xs-8 col-sm-9">
						<h4 className="title">
							<FormattedMessage id={this.props.item.incomeType + '-income-type-title'} />
							<ToolTip
								id={this.props.item.incomeType}
								text={
									this.props.person && this.props.person.memberOfIndianTribeIndicator === 'Y' && this.props.intl.doesMessageExist(this.props.item.incomeType + '-income-type-tooltip-ai-na')
									? this.props.intl.formatMessage({ id: this.props.item.incomeType + '-income-type-tooltip-ai-na' })
									: this.props.intl.messageIfExists(this.props.item.incomeType + '-income-type-tooltip')
								}
							/>
						</h4>
						<p><FormattedMessage id={this.props.item.incomeType + '-income-type-subtitle'} /></p>
					</div>
					<div className="col-xs-4 col-sm-3 text-right">
						<button
							id={this.props.id + '-add'}
							className="btn btn-pill btn-add"
							onClick={this.handleClickAdd.bind(this)}
							aria-label={this.props.intl.messageIfExists('2.1-employment-&-self-employment-income-button1') + ' ' + this.props.intl.messageIfExists(this.props.item.incomeType + '-income-type-title')}
							disabled={this.props.viewOnlyMode}
						>
							<FormattedMessage id="2.1-employment-&-self-employment-income-button1" />
						</button>
					</div>
				</div>
				<div className="incomes">
					<ul>{incomes}</ul>
				</div>
			</div>
		)
	}
}
 
IncomeType.propTypes = {
	item: incomeTypeShape,
	showSummaryRow: React.PropTypes.bool,
	intl: intlShape.isRequired,
	viewOnlyMode: React.PropTypes.bool,
	person: React.PropTypes.object,
	incomes: React.PropTypes.array,
	onAdd: React.PropTypes.func,
	onEdit: React.PropTypes.func
};
 
IncomeType.defaultProps = {
	item: {},
	showSummaryRow: false,
	viewOnlyMode: false,
	onAdd: () => {},
	onEdit: () => {}
};
 
export default injectIntl(IncomeType);
