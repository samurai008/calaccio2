import React from 'react';
import { FormattedMessage } from 'react-intl';
import { IncomeValuePerTimeframe } from '../MoneyValue.jsx';
import { recordShape } from '../../types.js';
 
const RecordSummaryRow = props => {
	var classNames = ['record-summary-row', 'row'],
		sourceClasses = [(props.showEditColumn ? 'col-xs-9' : 'col-xs-6'), 'col-sm-6'],
		valueClasses = ['amount', (props.showEditColumn ? 'col-xs-9' : 'col-xs-6'), 'col-sm-' + (props.showEditColumn ? '4' : '6')],
		editClasses = ['col-sm-2', 'text-right'];
 
	return (
		<div className={classNames.join(' ')}>
			<div className={sourceClasses.join(' ')}>
				{props.item.incomeSource}
			</div>
			<div className={valueClasses.join(' ')}>
				<IncomeValuePerTimeframe item={props.item} displayNegative={props.displayNegative} useSlash={props.currencyUseSlash} />
			</div>
			{
				props.showEditColumn ? (
					<div className={editClasses.join(' ')}>
						<a id={props.id + '-edit'} href="#" role="button" className="edit" onClick={props.onEdit} onKeyDown={e => { if(e.keyCode === 32) props.onEdit(); }}>
							<FormattedMessage id={props.viewOnlyMode ? "global-element-view-button-text" : "global-element-edit-button-text"} />
						</a>
					</div>
				) : ''
			}
		</div>
	);
};
 
RecordSummaryRow.propTypes = {
	item: recordShape.isRequired,
	showEditColumn: React.PropTypes.bool,
	onEdit: React.PropTypes.func,
	currencyUseSlash: React.PropTypes.bool,
	displayNegative: React.PropTypes.bool
};
 
RecordSummaryRow.defaultProps = {
	showEditColumn: false,
	onEdit: () => {},
	currencyUseSlash: true,
	displayNegative: false
};
 
export default RecordSummaryRow;
