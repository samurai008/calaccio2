// Import Libs & Components
import React, { Component } from 'react';
import { FormattedMessage } from 'react-intl';
import MoneyValue, { IncomeValuePerTimeframe } from '../MoneyValue.jsx';
import RecordSummaryRow from './RecordSummaryRow.jsx';
import ConditionalField from '../ConditionalField.jsx';
 
class HouseholdMemberIncomeSummary extends Component {
 
	constructor(props) {
		super(props);
		// Set initial state
		this.state = {};
	}
 
	render() {
		var classNames = ['income-summary'],
			incomes = this.props.incomes.map((item, key) => <RecordSummaryRow key={key} item={item} currencyUseSlash={this.props.currencyUseSlash} />),
			deductions = this.props.deductions.map((item, key) => <RecordSummaryRow key={key} item={item} displayNegative={true} currencyUseSlash={this.props.currencyUseSlash} />);
 
		let isMoreThanZero = false;
		this.props.incomes.forEach(function(i){
			isMoreThanZero = isMoreThanZero || (i.amount > 0);
		});
		this.props.deductions.forEach(function(d){
			isMoreThanZero = isMoreThanZero ||  (d.amount > 0);
		});
		
		return (
 
			<ConditionalField condition={isMoreThanZero === true} >
				<div className={classNames.join(' ')}>
					{
						incomes.length > 0
						?
							<div className="incomes">
								<h4><FormattedMessage id="incomeSummary.incomeTitle" /></h4>
								{incomes}
							</div>
						: null
					}
					{
						deductions.length > 0
						?
							<div className="deductions">
								<h4><FormattedMessage id="incomeSummary.deductionsTitle" /></h4>
								{deductions}
							</div>
						: null
					}
				</div>
			</ConditionalField>
		);
	}
}
 
HouseholdMemberIncomeSummary.propTypes = {
	incomes: React.PropTypes.array,
	deductions: React.PropTypes.array,
	currencyUseSlash: React.PropTypes.bool
}
 
HouseholdMemberIncomeSummary.defaultProps = {
	incomes: [],
	deductions: [],
	currencyUseSlash: true
}
 
export default HouseholdMemberIncomeSummary;
