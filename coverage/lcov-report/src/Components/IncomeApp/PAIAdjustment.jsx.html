import React, { Component } from 'react';
import { intlShape, injectIntl, FormattedMessage, FormattedHTMLMessage } from 'react-intl';
import { Form } from 'formsy-react';
import PeopleStore from '../../Stores/PeopleStore';
import CurrencyInput from '../Inputs/CurrencyInput.jsx';
import { CancelModal } from '../../Components/Modal.jsx';
import moment from '../../TimeshifterMoment';
 
class PAIAdjustment extends Component {
 
	constructor(props) {
		super(props);
 
		this.state = {
			newVal: 0
		};
	}
 
	onChangeValue(name, val) {
		this.setState({ newVal: parseFloat(val.replace(/[^0-9.]/g, '') || '0') });
	}
 
	onCancel(e) {
		e.preventDefault();
		Eif (this.state.newVal === 0) {
			this.props.onRequestCancel();
		}
		else {
		this.setState({ confirmCancelModalIsOpen: true });
		}
	}
 
	closeCancelModal() {
		this.setState({ confirmCancelModalIsOpen: false });
	}
 
	handleConfirmCancel() {
		this.props.onRequestCancel();
	}
 
	onSave(e) {
		e.preventDefault();
		this.props.onRequestSave(this.state.newVal);
	}
 
	render() {
		// SIR 97645 replaced Date usage with moment for Timeshifter support
		var messageValues = {
				paiAmount: this.props.intl.formatNumber(this.props.amount, { style: 'currency', currency: 'USD' }),
				coverageYear: this.props.intl.formatDate(moment().year(this.props.enrollmentYear), { year: 'numeric' })
			};
 
		return (
			<div className="pai-adjustment">
 
				<FormattedHTMLMessage
					id="2.8.4-pai-adjustment-modal-body-copy-dynamic"
					values={messageValues} />
				<Form>
					<CurrencyInput
						inlineLabel={'in ' + messageValues.coverageYear}
						name="newPAI"
						id="newPAIInput"
						maxLength="13"
						onChange={this.onChangeValue.bind(this)}
						value={this.props.amount.toString()}
						disabled={this.props.viewOnlyMode}
						/>
				</Form>
				<br/>
				<div className="footer text-center modal-buttons">
					<button className="btn btn-default" onClick={this.onCancel.bind(this)}><FormattedMessage id="2.8.4-pai-adjustment-modal-button1" /></button>
					<button className="btn btn-primary" onClick={this.onSave.bind(this)} disabled={this.state.newVal === 0}><FormattedMessage id="2.8.4-pai-adjustment-modal-button2" /></button>
				</div>
					<CancelModal
					isOpen={this.state.confirmCancelModalIsOpen}
					onRequestClose={this.closeCancelModal.bind(this)}
					onRequestConfirm={this.handleConfirmCancel.bind(this)}
					contentLabel={this.props.intl.messageIfExists('modals.cancel.header')}
				/>
			</div>
		);
	}
}
 
PAIAdjustment.propTypes = {
	amount: React.PropTypes.number,
	viewOnlyMode: React.PropTypes.bool,
	enrollmentYear: React.PropTypes.number.isRequired,

	onRequestSave: React.PropTypes.func,
	intl: intlShape.isRequired
};
 
PAIAdjustment.defaultProps = {
	amount: 0,
	viewOnlyMode: false,
	onRequestCancel: () => {},

};
 
export default injectIntl(PAIAdjustment);
