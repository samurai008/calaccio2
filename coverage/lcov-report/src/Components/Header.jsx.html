//A component for the constant Header which appears in all Income Section pages
 
import React from 'react';
import { FormattedMessage } from 'react-intl';
import LayoutActions from '../Actions/LayoutActions';
import NavActions from '../Actions/NavActions';
import LocaleMenu from './LocaleMenu.jsx';
import FAB from '../Components/FAB.jsx';
 
const Header = props => {
 
	const className = ['main-header'].concat(
			props.showSaveAndExit ? 'has-save-and-exit' : [],
			props.className ? props.className.split(' ') : []
		),
		handleHelpClick = e => {
			e.preventDefault();
			LayoutActions.toggleFAB();
		},
		handleSaveClick = e => {
			e.preventDefault();
			NavActions.saveAndExit();
		},
 
		handleLogOutUserClick = e => {
			e.preventDefault();
			NavActions.logOutUser();
		};
 
	return (
		<header className={className.join(' ')} role="banner">
			<div className="container">
				<div className="row">
					<div className="col-xs-3 col-sm-2"><h1 className="logo">Covered California</h1></div>
					<div className="btn-column col-xs-9 col-sm-10 text-right">
						<LocaleMenu isOpen={props.localeMenuIsOpen} />
 
						<FAB isOpen={props.fabIsOpen} />
					
						{ props.showSaveAndExit
							? (
								<button id="save-exit-btn" className="btn btn-header" onClick={handleSaveClick}>
									<span className="hidden-xs"><FormattedMessage id="global-element-header-save-exit-link" /></span>
									<span className="hidden-sm hidden-md hidden-lg"><FormattedMessage id="global-element-header-save-exit-link-mobile" /></span>
								</button>
							) : null
						}
						{ props.showLogOutUser
							? (
								<button id="logout-btn" className="btn btn-header" onClick={handleLogOutUserClick}>
									<span className="hidden-xs"><FormattedMessage id="global-element-header-log-out" /></span>
									<span className="hidden-sm hidden-md hidden-lg"><FormattedMessage id="global-element-header-log-out-mobile" /></span>
								</button>
							) : null
						}
 
					</div>
				</div>
			</div>
		</header>
	);
}
 
Header.propTypes = {
	showSaveAndExit: React.PropTypes.bool
};
 
Header.defaultProps = {
	showSaveAndExit: true
};
 
export default Header;
